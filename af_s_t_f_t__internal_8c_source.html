56 void afSTFTlib_init
57 (
58  void** handle,
59  int hopSize,
60  int inChannels,
61  int outChannels,
62  int LDmode,
63  int hybridMode
64 )
65 {
66  int k, ch, dsFactor;
67  float eq;
68 
69  *handle = malloc(sizeof(afSTFTlib_internal_data));
70 
71  /* Basic initializations */
72  afSTFTlib_internal_data *h = (afSTFTlib_internal_data*)(*handle);
73 
74  h->inChannels = inChannels;
75  h->outChannels = outChannels;
76  h->hopSize = hopSize;
77  dsFactor = 1024/hopSize;
78  h->hLen = 10240/dsFactor;
79  h->totalHops=10;
80  h->hopIndexIn=0;
81  h->hopIndexOut=0;
82  h->LDmode = LDmode;
83  h->protoFilter = (float*)malloc(sizeof(float)*h->hLen);
84  h->protoFilterI = (float*)malloc(sizeof(float)*h->hLen);
85  h->inBuffer = (float**)malloc(sizeof(float*)*h->inChannels);
86  h->outBuffer = (float**)malloc(sizeof(float*)*h->outChannels);
87  h->fftProcessFrameTD = (float*)calloc(sizeof(float),h->hopSize*2);
88 #ifdef AFSTFT_USE_SAF_UTILITIES
89  saf_rfft_create(&(h->hSafFFT), h->hopSize*2);
90  h->fftProcessFrameFD = calloc((h->hopSize+1), sizeof(float_complex));
91  h->tempHopBuffer = malloc(h->hopSize*sizeof(float));
92 #else
93  switch (hopSize) {
94  case 32:
95  h->log2n=6;
96  break;
97  case 64:
98  h->log2n=7;
99  break;
100  case 128:
101  h->log2n=8;
102  break;
103  case 256:
104  h->log2n=9;
105  break;
106  case 512:
107  h->log2n=10;
108  break;
109  case 1024:
110  h->log2n=11;
111  break;
112  default:
113  // No other modes defined
114  return;
115  break;
116  }
117  h->fftProcessFrameFD = (float*)calloc(sizeof(float),(h->hopSize+1)*2);
118  vtInitFFT(&(h->vtFFT), h->fftProcessFrameTD, h->fftProcessFrameFD, h->log2n);
119 #endif
120 
121  /* Normalization to ensure 0dB gain */
122  if (h->LDmode==0) {
123 #ifdef AFSTFT_USE_SAF_UTILITIES
124  eq = 2.0f/sqrtf(5.487604141f);
125 #else
126  eq = 1.0f/sqrtf((float)h->hopSize*5.487604141f);
127 #endif
128  for (k=0; k<h->hLen; k++) {
129  h->protoFilter[h->hLen-k-1] = __afSTFT_protoFilter1024[k*dsFactor]*eq;
130  h->protoFilterI[h->hLen-k-1] = __afSTFT_protoFilter1024[k*dsFactor]*eq;
131  }
132  }
133  else
134  {
135 #ifdef AFSTFT_USE_SAF_UTILITIES
136  eq = 2.0f/sqrtf(4.544559956f);
137 #else
138  eq = 1.0f/sqrtf((float)h->hopSize*4.544559956f);
139 #endif
140  for (k=0; k<h->hLen; k++) {
141  h->protoFilter[h->hLen-k-1] = __afSTFT_protoFilter1024LD[k*dsFactor]*eq;
142  h->protoFilterI[k]=__afSTFT_protoFilter1024LD[k*dsFactor]*eq;
143  }
144  }
145  for(ch=0;ch<h->inChannels;ch++)
146  h->inBuffer[ch] = (float*)calloc(h->hLen,sizeof(float));
147 
148  for(ch=0;ch<h->outChannels;ch++)
149  h->outBuffer[ch] = (float*)calloc(h->hLen,sizeof(float));
150 
151  /* Initialize the hybrid filter memory etc. */
152  h->hybridMode=hybridMode;
153  if (h->hybridMode)
154  afHybridInit(&(h->h_afHybrid), h->hopSize, h->inChannels,h->outChannels);
155 }

157 void afSTFTlib_channelChange
158 (
159  void* handle,
160  int new_inChannels,
161  int new_outChannels
162 )
163 {
164  afSTFTlib_internal_data *h = (afSTFTlib_internal_data*)(handle);
165  afHybrid *hyb_h = h->h_afHybrid;
166 
167  int i, ch, sample;
168  if(h->inChannels!=new_inChannels){
169  for(i=new_inChannels; i<h->inChannels; i++)
170  free(h->inBuffer[i]);
171  h->inBuffer = (float**)realloc(h->inBuffer, sizeof(float*)*new_inChannels);
172  for(i=h->inChannels; i<new_inChannels; i++)
173  h->inBuffer[i] = (float*)calloc(h->hLen,sizeof(float));
174  }
175 
176  if(h->outChannels!=new_outChannels){
177  for(i=new_outChannels; i<h->outChannels; i++)
178  free(h->outBuffer[i]);
179  h->outBuffer = (float**)realloc(h->outBuffer, sizeof(float*)*new_outChannels);
180  for(i=h->outChannels; i<new_outChannels; i++)
181  h->outBuffer[i] = (float*)calloc(h->hLen,sizeof(float));
182  }
183 
184  if (h->hybridMode) {
185  hyb_h = h->h_afHybrid;
186  if (hyb_h->inChannels != new_inChannels) {
187  for (ch = new_inChannels; ch < hyb_h->inChannels; ch++) {
188  for (sample = 0; sample < 7; sample++) {
189  free(hyb_h->analysisBuffer[ch][sample].re);
190  free(hyb_h->analysisBuffer[ch][sample].im);
191  }
192  free(hyb_h->analysisBuffer[ch]);
193  }
194  hyb_h->analysisBuffer = (complexVector**) realloc(hyb_h->analysisBuffer, sizeof(complexVector*) * new_inChannels);
195  for (ch = hyb_h->inChannels; ch < new_inChannels; ch++) {
196  hyb_h->analysisBuffer[ch] = (complexVector*) malloc(sizeof(complexVector) * 7);
197  for (sample = 0; sample < 7; sample++) {
198  hyb_h->analysisBuffer[ch][sample].re = (float*) calloc(sizeof(float), h->hopSize + 1);
199  hyb_h->analysisBuffer[ch][sample].im = (float*) calloc(sizeof(float), h->hopSize + 1);
200  }
201  }
202  }
203  }
204  h->inChannels = new_inChannels;
205  h->outChannels = new_outChannels;
206  if (h->hybridMode){
207  hyb_h->inChannels = new_inChannels;
208  hyb_h->outChannels = new_outChannels;
209  }

236 void afSTFTlib_forward
237 (
238  void* handle,
239  float** inTD,
240  complexVector* outFD
241 )
242 {
243  afSTFTlib_internal_data *h = (afSTFTlib_internal_data*)(handle);
244  int ch,k,j,hopIndex_this,hopIndex_this2;
245  float *p1,*p2,*p3;
246 #ifndef AFSTFT_USE_SAF_UTILITIES
247  float *p4;
248 #endif
249  int lr;
250 
251  for (ch=0;ch<h->inChannels;ch++)
252  {
253  /* Copy the input frame into the memory buffer */
254  hopIndex_this2 = h->hopIndexIn;
255  p1=&(h->inBuffer[ch][hopIndex_this2*h->hopSize]);
256  p2=inTD[ch];
257  memcpy((void*)p1,(void*)p2,sizeof(float)*(h->hopSize));
258 
259  hopIndex_this2++;
260  if (hopIndex_this2 >= h->totalHops)
261  {
262  hopIndex_this2 = 0;
263  }
264 
265  /* Apply prototype filter to the collected data in the memory buffer, and fold the result (for the FFT operation). */
266  p1 = h->fftProcessFrameTD;
267 #ifdef AFSTFT_USE_SAF_UTILITIES
268  memset(p1, 0, h->hopSize*2*sizeof(float));
269 #else
270  vtClr(p1, h->hopSize*2);
271 #endif
272  lr=0; /* Left or right part of the frame */
273  hopIndex_this = hopIndex_this2;
274  for (k=0;k<h->totalHops;k++)
275  {
276  p1=&(h->inBuffer[ch][h->hopSize*hopIndex_this]);
277  p2=&(h->protoFilter[k*h->hopSize]);
278  if (lr==1)
279  {
280  p3=&(h->fftProcessFrameTD[h->hopSize]);
281  lr=0;
282  }
283  else
284  {
285  p3=&(h->fftProcessFrameTD[0]);
286  lr=1;
287  }
288 #ifdef AFSTFT_USE_SAF_UTILITIES
289  //utility_svvmuladd(p1, p2, h->hopSize, p3); /* somehow slower... */
290  for (j=0;j<h->hopSize;j+=4){
291  p3[j] += (p1[j])*(p2[j]);
292  p3[j+1] += (p1[j+1])*(p2[j+1]);
293  p3[j+2] += (p1[j+2])*(p2[j+2]);
294  p3[j+3] += (p1[j+3])*(p2[j+3]);
295  }
296 #else
297  vtVma(p1, p2, p3, h->hopSize); /* Vector multiply-add */
298 #endif
299  hopIndex_this++;
300  if (hopIndex_this >= h->totalHops)
301  {
302  hopIndex_this = 0;
303  }
304  }
305 
306  /* Apply FFT and copy the data to the output vector */
307 #ifdef AFSTFT_USE_SAF_UTILITIES
308  saf_rfft_forward(h->hSafFFT, h->fftProcessFrameTD, h->fftProcessFrameFD);
309 // for(k = 0; k<h->hopSize+1; k++){
310 // outFD[ch].re[k] = crealf(h->fftProcessFrameFD[k]);
311 // outFD[ch].im[k] = cimagf(h->fftProcessFrameFD[k]);
312 // }
313  cblas_scopy(h->hopSize+1, (float*)h->fftProcessFrameFD, 2, outFD[ch].re, 1);
314  cblas_scopy(h->hopSize+1, (float*)h->fftProcessFrameFD + 1, 2, outFD[ch].im, 1);
315 #else
316  vtRunFFT(h->vtFFT,1);
317  outFD[ch].re[0]=h->fftProcessFrameFD[0];
318  outFD[ch].im[0]=0.0f; /* DC im = 0 */
319  outFD[ch].re[h->hopSize]=h->fftProcessFrameFD[h->hopSize];
320  outFD[ch].im[h->hopSize]=0.0f; /* Nyquist im = 0 */
321  p1 = outFD[ch].re + 1;
322  p2 = outFD[ch].im + 1;
323  p3 = h->fftProcessFrameFD + 1;
324  p4 = h->fftProcessFrameFD + 1 + h->hopSize;
325  memcpy((void*)p1,(void*)p3,sizeof(float)*(h->hopSize - 1));
326  memcpy((void*)p2,(void*)p4,sizeof(float)*(h->hopSize - 1));
327 #endif
328  }
329  h->hopIndexIn++;
330  if (h->hopIndexIn >= h->totalHops)
331  {
332  h->hopIndexIn = 0;
333  }
334 
335  /* Subdivide lowest bands with half-band filters if hybrid mode is enabled */
336  if (h->hybridMode)
337  {
338  afHybridForward(h->h_afHybrid, outFD);
339  }
340 }

342 void afSTFTlib_inverse
343 (
344  void* handle,
345  complexVector* inFD,
346  float** outTD
347 )
348 {
349  afSTFTlib_internal_data *h = (afSTFTlib_internal_data*)(handle);
350  int ch,k,j,hopIndex_this,hopIndex_this2;
351  float *p1,*p2,*p3;
352 #ifndef AFSTFT_USE_SAF_UTILITIES
353  float *p4;
354 #endif
355  int lr;
356 
357  /* Combine subdivided lowest bands if hybrid mode is enabled */
358  if (h->hybridMode)
359  {
360  afHybridInverse(h->h_afHybrid, inFD);
361  }
362 
363  for (ch=0;ch<h->outChannels;ch++)
364  {
365  /* Copy data from input to internal memory */
366  hopIndex_this2 = h->hopIndexOut;
367 
368  /* Inverse FFT */
369 #ifdef AFSTFT_USE_SAF_UTILITIES
370  //for(k = 0; k<h->hopSize+1; k++)
371  // h->fftProcessFrameFD[k] = cmplxf(inFD[ch].re[k], inFD[ch].im[k]);
372  cblas_scopy(h->hopSize+1, inFD[ch].re, 1, (float*)h->fftProcessFrameFD, 2);
373  cblas_scopy(h->hopSize+1, inFD[ch].im, 1, (float*)h->fftProcessFrameFD + 1, 2);
374 
375  /* The low delay mode requires this procedure corresponding to the circular shift of the data in the time domain */
376  if (h->LDmode == 1)
377  for (k=1; k<h->hopSize; k+=2)
378  h->fftProcessFrameFD[k] = crmulf(h->fftProcessFrameFD[k], -1.0f);
379 
380  saf_rfft_backward(h->hSafFFT, h->fftProcessFrameFD, h->fftProcessFrameTD);
381 #else
382  h->fftProcessFrameFD[0] = inFD[ch].re[0]; /* DC */
383  h->fftProcessFrameFD[h->hopSize] = inFD[ch].re[h->hopSize]; /* Nyquist */
384  p1 = inFD[ch].re + 1;
385  p2 = inFD[ch].im + 1;
386  p3 = h->fftProcessFrameFD + 1;
387  p4 = h->fftProcessFrameFD + 1 + h->hopSize;
388  memcpy((void*)p3,(void*)p1,sizeof(float)*(h->hopSize - 1));
389  memcpy((void*)p4,(void*)p2,sizeof(float)*(h->hopSize - 1));
390 
391  /* The low delay mode requires this procedure corresponding to the circular shift of the data in the time domain */
392  if (h->LDmode == 1) {
393  for (k=1;k<h->hopSize;k+=2) {
394  *p3 = -*p3;
395  *p4 = -*p4;
396  p3+=2;
397  p4+=2;
398  }
399  }
400 
401  vtRunFFT(h->vtFFT, -1);
402 #endif
403 
404  /* Clear buffer at the pointer location and increment the pointer */
405  p1 = &(h->outBuffer[ch][hopIndex_this2*h->hopSize]);
406 #ifdef AFSTFT_USE_SAF_UTILITIES
407  memset(p1, 0, h->hopSize*sizeof(float));
408 #else
409  vtClr(p1,h->hopSize);
410 #endif
411  hopIndex_this2++;
412  if (hopIndex_this2 >= h->totalHops)
413  {
414  hopIndex_this2=0;
415  }
416  hopIndex_this = hopIndex_this2;
417 
418  lr=0; /* Left or right part of the frame */
419  for (k=0;k<h->totalHops;k++)
420  {
421  /* Apply the prototype filter to the repeated version of the IFFT'd data. */
422  p1=&(h->outBuffer[ch][h->hopSize*hopIndex_this]);
423  p2=&(h->protoFilterI[k*h->hopSize]);
424 
425  if (lr==1)
426  {
427  p3=&(h->fftProcessFrameTD[h->hopSize]);
428  lr=0;
429  }
430  else
431  {
432  p3=&(h->fftProcessFrameTD[0]);
433  lr=1;
434  }
435 
436  /* Overlap-add to the existing data in the memory buffer (from previous frames). */
437 #ifdef AFSTFT_USE_SAF_UTILITIES
438  for (j=0;j<h->hopSize;j+=4){
439  p1[j] += (p2[j])*(p3[j]);
440  p1[j+1] += (p2[j+1])*(p3[j+1]);
441  p1[j+2] += (p2[j+2])*(p3[j+2]);
442  p1[j+3] += (p2[j+3])*(p3[j+3]);
443  }
444 #else
445  vtVma(p2, p3, p1, h->hopSize); /* Vector multiply-add */
446 #endif
447 
448  hopIndex_this++;
449  if (hopIndex_this >= h->totalHops)
450  {
451  hopIndex_this = 0;
452  }
453  }
454 
455  /* Copy a frame from work memory to the output */
456  p1 = outTD[ch];
457  p2 = &(h->outBuffer[ch][h->hopSize*hopIndex_this]);
458  memcpy((void*)p1,(void*)p2,sizeof(float)*(h->hopSize));
459 
460  }
461  h->hopIndexOut++;
462  if (h->hopIndexOut >= h->totalHops)
463  {
464  h->hopIndexOut=0;
465  }
466 
467 }

510 (
511  void** handle,
512  int hopSize,
513  int inChannels,

515 )

525  h->loopPointer=0;
526  for (ch=0;ch<h->inChannels;ch++)
527  {
528  h->analysisBuffer[ch] = (complexVector*)malloc(sizeof(complexVector)*7);
529  for (sample=0;sample<7;sample++)
530  {
531  h->analysisBuffer[ch][sample].re=(float*)calloc(sizeof(float),h->hopSize+1);
532  h->analysisBuffer[ch][sample].im=(float*)calloc(sizeof(float),h->hopSize+1);
533  }

537 void afHybridForward
538 (
539  void* handle,
540  complexVector* FD
541 )
542 {
543  afHybrid *h = (afHybrid*)(handle);
544  int ch,band,sample,realImag;
545  float *pr1, *pr2, *pi1, *pi2;
546  float re,im;
547  int sampleIndices[7];
548  int loopPointerThis;
549  h->loopPointer++;
550  if( h->loopPointer == 7)
551  {
552  h->loopPointer = 0;
553  }
554 
555  for (ch=0;ch<h->inChannels;ch++)
556  {
557  /* Copy data from input to the memory buffer */
558  pr1 = FD[ch].re;
559  pi1 = FD[ch].im;
560  pr2 = h->analysisBuffer[ch][h->loopPointer].re;
561  pi2 = h->analysisBuffer[ch][h->loopPointer].im;
562  memcpy((void*)pr2,(void*)pr1,sizeof(float)*(h->hopSize+1));
563  memcpy((void*)pi2,(void*)pi1,sizeof(float)*(h->hopSize+1));
564 
565  /* Get the pointer to a position corresponding to the group delay of the linear-phase half-band filter. */
566  loopPointerThis = h->loopPointer - 3;
567  if( loopPointerThis < 0)
568  {
569  loopPointerThis += 7;
570  }
571  pr1 = FD[ch].re;
572  pr2 = h->analysisBuffer[ch][loopPointerThis].re;
573  for (realImag=0;realImag<2;realImag++)
574  {
575  /* The 0.5 multipliers are the center coefficients of the half-band FIR filters. Data is duplicated for the half-bands. */
576  *pr1 = *pr2;
577  *(pr1+1) = *(pr2+1)*0.5f;
578  *(pr1+2) = *(pr1+1);
579  *(pr1+3) = *(pr2+2)*0.5f;
580  *(pr1+4) = *(pr1+3);
581  *(pr1+5) = *(pr2+3)*0.5f;
582  *(pr1+6) = *(pr1+5);
583  *(pr1+7) = *(pr2+4)*0.5f;
584  *(pr1+8) = *(pr1+7);
585 
586  /* The rest of the bands are shifted upwards in the frequency indices, and delayed by the group delay of the half-band filters */
587  memcpy((void*)(pr1+9),(void*)(pr2+5),sizeof(float)*(h->hopSize-4));
588 
589  /* Repeat process for the imaginary part, at next iteration. */
590  pr1 = FD[ch].im;
591  pr2 = h->analysisBuffer[ch][loopPointerThis].im;
592  }
593 
594  for (sample=0;sample<7;sample++)
595  {
596  sampleIndices[sample]=h->loopPointer+1+sample;
597  if(sampleIndices[sample] > 6)
598  {
599  sampleIndices[sample]-=7;
600  }
601 
602  }
603  for (band=1; band<5; band++)
604  {
605  /* The rest of the half-band FIR filtering is implemented below. The real<->imaginary shifts are for shifting the half-band filter spectra. */
606  re = -COEFF1*h->analysisBuffer[ch][sampleIndices[6]].im[band];
607  im = COEFF1*h->analysisBuffer[ch][sampleIndices[6]].re[band];
608  re -= COEFF2*h->analysisBuffer[ch][sampleIndices[4]].im[band];
609  im += COEFF2*h->analysisBuffer[ch][sampleIndices[4]].re[band];
610  re += COEFF2*h->analysisBuffer[ch][sampleIndices[2]].im[band];
611  im -= COEFF2*h->analysisBuffer[ch][sampleIndices[2]].re[band];
612  re += COEFF1*h->analysisBuffer[ch][sampleIndices[0]].im[band];
613  im -= COEFF1*h->analysisBuffer[ch][sampleIndices[0]].re[band];
614 
615  /* The addition or subtraction process below provides the upper and lower half-band spectra (the coefficient 0.5 had the same sign for both bands).
616  The half-band orders are switched for bands=1,3 with respect to band=2,4, because of the organization of the spectral data at the downsampled frequency band signals. As the result of the order switching, the bands are organized by the ascending spectral position. */
617  if (band == 1 || band== 3)
618  {
619  FD[ch].re[band*2-1] -= re;
620  FD[ch].im[band*2-1] -= im;
621  FD[ch].re[band*2] += re;
622  FD[ch].im[band*2] += im;
623  }
624  else
625  {
626  FD[ch].re[band*2-1] += re;
627  FD[ch].im[band*2-1] += im;
628  FD[ch].re[band*2] -= re;
629  FD[ch].im[band*2] -= im;
630  }
631 
632  }
