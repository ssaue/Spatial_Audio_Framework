38 static void smbFft(float *fftBuffer, int fftFrameSize, long sign)
39 /*
40  FFT routine, (C)1996 S.M.Bernsee. Sign = -1 is FFT, 1 is iFFT (inverse)
41  Fills fftBuffer[0...2*fftFrameSize-1] with the Fourier transform of the
42  time domain data in fftBuffer[0...2*fftFrameSize-1]. The FFT array takes
43  and returns the cosine and sine parts in an interleaved manner, ie.
44  fftBuffer[0] = cosPart[0], fftBuffer[1] = sinPart[0], asf. fftFrameSize
45  must be a power of 2. It expects a complex input signal (see footnote 2),
46  ie. when working with 'common' audio signals our input signal has to be
47  passed as {in[0],0.,in[1],0.,in[2],0.,...} asf. In that case, the transform
48  of the frequencies of interest is in fftBuffer[0...fftFrameSize].
49 */
50 {
51  float wr, wi, arg, *p1, *p2, temp;
52  float tr, ti, ur, ui, *p1r, *p1i, *p2r, *p2i;
53  long i, bitm, j, le, le2, k;
54 
55  for (i = 2; i < 2*fftFrameSize-2; i += 2) {
56  for (bitm = 2, j = 0; bitm < 2*fftFrameSize; bitm <<= 1) {
57  if (i & bitm) j++;
58  j <<= 1;
59  }
60  if (i < j) {
61  p1 = fftBuffer+i; p2 = fftBuffer+j;
62  temp = *p1; *(p1++) = *p2;
63  *(p2++) = temp; temp = *p1;
64  *p1 = *p2; *p2 = temp;
65  }
66  }
67  for (k = 0, le = 2; k < (int)(log(fftFrameSize)/log(2.)+.5); k++) {
68  le <<= 1;
69  le2 = le>>1;
70  ur = 1.0;
71  ui = 0.0;
72  arg = M_PI / (le2>>1);
73  wr = cos(arg);
74  wi = sign*sin(arg);
75  for (j = 0; j < le2; j += 2) {
76  p1r = fftBuffer+j; p1i = p1r+1;
77  p2r = p1r+le2; p2i = p2r+1;
78  for (i = j; i < 2*fftFrameSize; i += le) {
79  tr = *p2r * ur - *p2i * ui;
80  ti = *p2r * ui + *p2i * ur;
81  *p2r = *p1r - tr; *p2i = *p1i - ti;
82  *p1r += tr; *p1i += ti;
83  p1r += le; p1i += le;
84  p2r += le; p2i += le;
85  }
86  tr = ur*wr - ui*wi;
87  ui = ur*wi + ui*wr;
88  ur = tr;
89  }
90  }
91 }

122 void smb_pitchShift_create
123 (
124  void** hSmb,
125  int nCH,
126  int fftFrameSize,
127  int osamp,
128  float sampleRate
129 )
130 {
131  *hSmb = malloc(sizeof(smb_pitchShift_data));
132  smb_pitchShift_data *h = (smb_pitchShift_data*)(*hSmb);
133  int i;
134 
135  h->fftFrameSize = fftFrameSize;
136  h->osamp = osamp;
137  h->sampleRate = sampleRate;
138  h->nCH = nCH;
139  h->pitchShiftFactor = 1.0f;
140 
141  /* internals */
142  saf_fft_create(&(h->hFFT), fftFrameSize);
143  h->stepSize = fftFrameSize/h->osamp;
144  h->inFifoLatency = fftFrameSize - (h->stepSize);
145  h->gRover = (int*)malloc1d(nCH * sizeof(int));
146  for(i=0; i<nCH; i++)
147  h->gRover[i] = h->inFifoLatency;
148  h->window = (float*)malloc1d(fftFrameSize*sizeof(float));
149  for (i = 0; i < fftFrameSize; i++)
150  h->window[i] = -0.5f*cosf(2.0f*SAF_PI*(float)i/(float)fftFrameSize)+0.5f;
151  h->gInFIFO = (float**)calloc2d(nCH,fftFrameSize,sizeof(float));
152  h->gOutFIFO = (float**)calloc2d(nCH,fftFrameSize,sizeof(float));
153 #ifdef SMB_ENABLE_SAF_FFT
154  h->gFFTworksp_td = (float_complex**)calloc2d(nCH,fftFrameSize,sizeof(float_complex));
155  h->gFFTworksp_fd = (float_complex**)calloc2d(nCH,fftFrameSize,sizeof(float_complex));
156 #else
157  h->gFFTworksp = (float**)calloc2d(nCH,2*fftFrameSize,sizeof(float));
158 #endif
159  h->gLastPhase = (float**)calloc2d(nCH,fftFrameSize/2+1,sizeof(float));
160  h->gSumPhase = (float**)calloc2d(nCH,fftFrameSize/2+1,sizeof(float));
161  h->gOutputAccum = (float**)calloc2d(nCH,2*(h->fftFrameSize),sizeof(float));
162  h->gAnaFreq = (float**)calloc2d(nCH,fftFrameSize,sizeof(float));
163  h->gAnaMagn = (float**)calloc2d(nCH,fftFrameSize,sizeof(float));
164  h->gSynFreq = (float**)malloc2d(nCH,fftFrameSize,sizeof(float));
165  h->gSynMagn = (float**)malloc2d(nCH,fftFrameSize,sizeof(float));
166 }

210 void smb_pitchShift_apply
211 (
212  void* hSmb,
213  float pitchShift,
214  int frameSize,
215  float *indata,
216  float *outdata
217 )
218 {
219  smb_pitchShift_data *h = (smb_pitchShift_data*)(hSmb);
220  float magn, phase, tmp, real, imag;
221  float freqPerBin, expct;
222  int ch, i, k, qpd, index, fftFrameSize2;
223 
224  /* set up some handy variables */
225  fftFrameSize2 = h->fftFrameSize/2;
226  freqPerBin = h->sampleRate/(float)h->fftFrameSize;
227  expct = 2.0f*SAF_PI*(float)(h->stepSize)/(float)h->fftFrameSize;
228 
229  /* flush buffers */
230  if(h->pitchShiftFactor!=pitchShift){
231  h->pitchShiftFactor = pitchShift;
232  for(ch=0; ch<h->nCH; ch++){
233  memset(h->gOutputAccum[ch], 0, h->stepSize*sizeof(float));
234  memset(h->gLastPhase[ch], 0, (h->fftFrameSize/2+1)*sizeof(float));
235  memset(h->gSumPhase[ch], 0, (h->fftFrameSize/2+1)*sizeof(float));
236  }
237  }
238 
239  /* main processing loop */
240  for(ch=0; ch<h->nCH; ch++){
241  for (i = 0; i < frameSize; i++){
242  /* As long as we have not yet collected enough data just read in */
243  h->gInFIFO[ch][h->gRover[ch]] = indata[ch*frameSize+i];
244  outdata[ch*frameSize+i] = h->gOutFIFO[ch][h->gRover[ch]-(h->inFifoLatency)];
245  h->gRover[ch]++;
246 
247  /* now we have enough data for processing */
248  if (h->gRover[ch] >= h->fftFrameSize) {
249  h->gRover[ch] = h->inFifoLatency;
250 
251 #ifdef SMB_ENABLE_SAF_FFT
252  for (k = 0; k < h->fftFrameSize;k++)
253  h->gFFTworksp_td[ch][k] = cmplxf(h->gInFIFO[ch][k] * (float)h->window[k], 0.0f);
254  saf_fft_forward(h->hFFT, h->gFFTworksp_td[ch], h->gFFTworksp_fd[ch]);
255 #else
256  /* do windowing and re,im interleave */
257  for (k = 0; k < h->fftFrameSize;k++) {
258  h->gFFTworksp[ch][2*k] = h->gInFIFO[ch][k] * h->window[k];
259  h->gFFTworksp[ch][2*k+1] = 0.;
260  }
261 
262  /* do transform */
263  smbFft(h->gFFTworksp[ch], h->fftFrameSize, -1);
264 #endif
265 
266  /* ***************** ANALYSIS ******************* */
267  for (k = 0; k <= fftFrameSize2; k++) {
268 #ifdef SMB_ENABLE_SAF_FFT
269  real = crealf(h->gFFTworksp_fd[ch][k]);
270  imag = cimagf(h->gFFTworksp_fd[ch][k]);
271 #else
272  /* de-interlace FFT buffer */
273  real = h->gFFTworksp[ch][2*k];
274  imag = h->gFFTworksp[ch][2*k+1];
275 #endif
276  /* compute magnitude and phase */
277  magn = 2.0f*sqrtf(real*real + imag*imag);
278  phase = atan2f(imag,real);
279 
280  /* compute phase difference */
281  tmp = phase - h->gLastPhase[ch][k];
282  h->gLastPhase[ch][k] = phase;
283 
284  /* subtract expected phase difference */
285  tmp -= (float)k*expct;
286 
287  /* map delta phase into +/- Pi interval */
288  qpd = (int)(tmp/SAF_PI);
289  if (qpd >= 0) qpd += qpd&1;
290  else qpd -= qpd&1;
291  tmp -= SAF_PI*(float)qpd;
292 
293  /* get deviation from bin frequency from the +/- Pi interval */
294  tmp = (float)h->osamp*tmp/(2.0f*SAF_PI);
295 
296  /* compute the k-th partials' true frequency */
297  tmp = (float)k*freqPerBin + tmp*freqPerBin;
298 
299  /* store magnitude and true frequency in analysis arrays */
300  h->gAnaMagn[ch][k] = magn;
301  h->gAnaFreq[ch][k] = tmp;
302 
303  }
304 
305  /* ***************** PROCESSING ******************* */
306  /* this does the actual pitch shifting */
307  memset(h->gSynMagn[ch], 0, h->fftFrameSize*sizeof(float));
308  memset(h->gSynFreq[ch], 0, h->fftFrameSize*sizeof(float));
309  for (k = 0; k <= fftFrameSize2; k++) {
310  index = (int)((float)k*(h->pitchShiftFactor));
311  if (index <= fftFrameSize2) {
312  h->gSynMagn[ch][index] += (h->gAnaMagn[ch][k]);
313  h->gSynFreq[ch][index] = h->gAnaFreq[ch][k] * (h->pitchShiftFactor);
314  }
315  }
316 
317  /* ***************** SYNTHESIS ******************* */
318  /* this is the synthesis step */
319  for (k = 0; k <= fftFrameSize2; k++) {
320  /* get magnitude and true frequency from synthesis arrays */
321  magn = h->gSynMagn[ch][k];
322  tmp = h->gSynFreq[ch][k];
323 
324  /* subtract bin mid frequency */
325  tmp -= (float)k*freqPerBin;
326 
327  /* get bin deviation from freq deviation */
328  tmp /= freqPerBin;
329 
330  /* take osamp into account */
331  tmp = 2.0f*SAF_PI*tmp/(h->osamp);
332 
333  /* add the overlap phase advance back in */
334  tmp += (float)k*expct;
335 
336  /* accumulate delta phase to get bin phase */
337  h->gSumPhase[ch][k] += tmp;
338  phase = h->gSumPhase[ch][k];
339 
340 #ifdef SMB_ENABLE_SAF_FFT
341  h->gFFTworksp_fd[ch][k] = cmplxf(magn*cosf(phase), magn*sinf(phase));
342 #else
343  /* get real and imag part and re-interleave */
344  h->gFFTworksp[ch][2*k] = magn*cos(phase);
345  h->gFFTworksp[ch][2*k+1] = magn*sin(phase);
346 #endif
347  }
348 
349 #ifdef SMB_ENABLE_SAF_FFT
350  for (k = fftFrameSize2+1; k < h->fftFrameSize; k++)
351  h->gFFTworksp_fd[ch][k] = cmplxf(0.0f, 0.0f);
352 
353  saf_fft_backward(h->hFFT, h->gFFTworksp_fd[ch], h->gFFTworksp_td[ch]);
354 
355  for(k=0; k < h->fftFrameSize; k++)
356  h->gOutputAccum[ch][k] += 2.0f*(h->window[k])*crealf(h->gFFTworksp_td[ch][k])/((h->osamp));
357 #else
358  /* zero negative frequencies */
359  for (k = h->fftFrameSize+2; k < 2*(h->fftFrameSize); k++)
360  h->gFFTworksp[ch][k] = 0.;
361 
362  /* do inverse transform */
363  smbFft(h->gFFTworksp[ch], h->fftFrameSize, 1);
364 
365  /* do windowing and add to output accumulator */
366  for(k=0; k < h->fftFrameSize; k++)
367  h->gOutputAccum[ch][k] += 2.*(h->window[k])*(h->gFFTworksp[ch][2*k])/(fftFrameSize2*(h->osamp));
368 #endif
369  for (k = 0; k < h->stepSize; k++)
370  h->gOutFIFO[ch][k] = h->gOutputAccum[ch][k];
371 
372  /* shift accumulator */
373  memmove(h->gOutputAccum[ch], h->gOutputAccum[ch] + (h->stepSize), h->fftFrameSize*sizeof(float));
374 
375  /* move input FIFO */
376  for (k = 0; k < h->inFifoLatency; k++)
377  h->gInFIFO[ch][k] = h->gInFIFO[ch][k+h->stepSize];
378  }
379  }
380  }
381 }
