38 void estimateITDs
39 (
40  float* hrirs /* N_dirs x NUM_EARS x hrir_len */,
41  int N_dirs,
42  int hrir_len,
43  int fs,
44  float* itds_s
45 )
46 {
47  int i, n, j, k, maxIdx, xcorr_len;
48  float maxVal, itd_bounds, fc, Q, K, KK, D, wn, Wz1[2], Wz2[2], b[3], a[3];
49  float* xcorr_LR, *ir_L, *ir_R, *hrir_lpf;
50 
51  /* calculate LPF coefficients, 2nd order IIR design equations from DAFX (2nd ed) p50 */
52  fc = 750.0f;
53  Q = 0.7071f;
54  K = tanf(SAF_PI * fc/(float)fs);
55  KK = K * K;
56  D = KK * Q + K + Q;
57  b[0] = (KK * Q) / D; b[1] = (2.0f * KK * Q) / D; b[2] = (KK * Q) / D;
58  a[0] = 1.0f; a[1] = (2.0f * Q * (KK - 1.0f)) / D; a[2] = (KK * Q - K + Q) / D;
59 
60  /* determine the ITD via the cross-correlation between the LPF'd left and right HRIR signals */
61  xcorr_len = 2*(hrir_len)-1;
62  itd_bounds = sqrtf(2.0f)/2e3f;
63  xcorr_LR = (float*)malloc1d(xcorr_len*sizeof(float));
64  ir_L = (float*)malloc1d(hrir_len*sizeof(float));
65  ir_R = (float*)malloc1d(hrir_len*sizeof(float));
66  hrir_lpf = (float*)malloc1d(hrir_len*2*sizeof(float));
67  for(i=0; i<N_dirs; i++){
68  /* apply lpf */
69  memset(Wz1, 0, 2*sizeof(float));
70  memset(Wz2, 0, 2*sizeof(float));
71  for (n=0; n<hrir_len; n++){
72  for(j=0; j<NUM_EARS; j++){
73  /* biquad difference equation (Direct form 2) */
74  wn = hrirs[i*NUM_EARS*hrir_len + j*hrir_len + n] - a[1] * Wz1[j] - a[2] * Wz2[j];
75  hrir_lpf[n*2+j] = b[0] * wn + b[1]*Wz1[j] + b[2]*Wz2[j];
76 
77  /* shuffle delays */
78  Wz2[j] = Wz1[j];
79  Wz1[j] = wn;
80  }
81  }
82 
83  /* xcorr between L and R */
84  for(k=0; k<hrir_len; k++){
85  ir_L[k] = hrir_lpf[k*2+0];
86  ir_R[k] = hrir_lpf[k*2+1];
87  }
88  cxcorr(ir_L, ir_R, xcorr_LR, hrir_len, hrir_len);
89  maxIdx = 0;
90  maxVal = 0.0f;
91  for(j=0; j<xcorr_len; j++){
92  if(xcorr_LR[j] > maxVal){
93  maxIdx = j;
94  maxVal = xcorr_LR[j];
95  }
96  }
97  itds_s[i] = ((float)hrir_len-(float)maxIdx-1.0f)/(float)fs;
98  itds_s[i] = itds_s[i] > itd_bounds ? itd_bounds : itds_s[i];
99  itds_s[i] = itds_s[i] < -itd_bounds ? -itd_bounds : itds_s[i];
100  }
101 
102  free(xcorr_LR);
103  free(ir_L);
104  free(ir_R);
105  free(hrir_lpf);
106 }

137 void HRIRs2HRTFs
138 (
139  float* hrirs, /* N_dirs x NUM_EARS x hrir_len */
140  int N_dirs,
141  int hrir_len,
142  int fftSize,
143  float_complex* hrtfs /* (fftSize/2+1) x NUM_EARS x N_dirs */
144 )
145 {
146  int i, j, k, nBins;
147  void* hSafFFT;
148  float* hrir_pad;
149  float_complex* hrtf;
150 
151  //TODO: if fftSize is shorter than hrir_len, maybe truncate based on the median peak?
152  /* Perform FFT */
153  nBins = fftSize/2 + 1;
154  saf_rfft_create(&hSafFFT, fftSize);
155  hrir_pad = calloc1d(fftSize, sizeof(float));
156  hrtf = malloc1d(nBins*sizeof(float_complex));
157  for(i=0; i<N_dirs; i++){
158  for(j=0; j<NUM_EARS; j++){
159  memcpy(hrir_pad, &hrirs[i*NUM_EARS*hrir_len+j*hrir_len], MIN(fftSize, hrir_len)*sizeof(float));
160  saf_rfft_forward(hSafFFT, hrir_pad, hrtf);
161  for(k=0; k<nBins; k++)
162  hrtfs[k*NUM_EARS*N_dirs + j*N_dirs + i] = hrtf[k];
163  }
164  }
165 
166  saf_rfft_destroy(&hSafFFT);
167  free(hrir_pad);
168  free(hrtf);
169 }

171 void diffuseFieldEqualiseHRTFs
172 (
173  int N_dirs,
174  float* itds_s,
175  float* centreFreq,
176  int N_bands,
177  float* weights,
178  int applyEQ,
179  int applyPhase,
180  float_complex* hrtfs /* N_bands x #NUM_EARS x N_dirs */
181 )
182 {
183  /* Anything to do at all? */
184  if(applyEQ + applyPhase)
185  {
186  int i, j, nd, band;
187  float* ipd, *hrtf_diff, *_weights;
188 
189  /* diffuse-field equalise */
190  if(applyEQ){
191  hrtf_diff = calloc1d(N_bands*NUM_EARS, sizeof(float));
192  if(weights == NULL){
193  _weights = malloc1d(N_dirs*sizeof(float));
194  for(int idx=0; idx < N_dirs; idx++)
195  _weights[idx] = 4.f*SAF_PI / (float)N_dirs;
196  }
197  else
198  _weights = weights;
199  for(band=0; band<N_bands; band++)
200  for(i=0; i<NUM_EARS; i++)
201  for(j=0; j<N_dirs; j++)
202  hrtf_diff[band*NUM_EARS + i] += _weights[j]/(4.f*SAF_PI) * powf(cabsf(hrtfs[band*NUM_EARS*N_dirs + i*N_dirs + j]), 2.0f);
203  for(band=0; band<N_bands; band++)
204  for(i=0; i<NUM_EARS; i++)
205  hrtf_diff[band*NUM_EARS + i] = sqrtf(hrtf_diff[band*NUM_EARS + i]);
206  for(band=0; band<N_bands; band++)
207  for(i=0; i<NUM_EARS; i++)
208  for(nd=0; nd<N_dirs; nd++)
209  hrtfs[band*NUM_EARS*N_dirs + i*N_dirs + nd] = ccdivf(hrtfs[band*NUM_EARS*N_dirs + i*N_dirs + nd], cmplxf(hrtf_diff[band*NUM_EARS + i] + 2.23e-8f, 0.0f));
210  free(hrtf_diff);
211  if(weights==NULL)
212  free(_weights);
213  }
214 
215  /* Create complex HRTFs by introducing the interaural phase differences
216  * (IPDs) to the HRTF magnitude responses */
217  if(applyPhase){
218  /* convert ITDs to phase differences -pi..pi */
219  ipd = malloc1d(N_bands*N_dirs*sizeof(float));
220  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasTrans, N_bands, N_dirs, 1, 1.0,
221  centreFreq, 1,
222  itds_s, 1, 0.0,
223  ipd, N_dirs);
224  for(i=0; i<N_bands; i++)
225  for(j=0; j<N_dirs; j++)
226  ipd[i*N_dirs+j] = (matlab_fmodf(2.0f*SAF_PI*ipd[i*N_dirs+j] + SAF_PI, 2.0f*SAF_PI) - SAF_PI)/2.0f; /* /2 here, not later */
227 
228  for(band=0; band<N_bands; band++){
229  for(nd=0; nd<N_dirs; nd++){
230  hrtfs[band*NUM_EARS*N_dirs + 0*N_dirs + nd] = crmulf( cexpf(cmplxf(0.0f, ipd[band*N_dirs + nd])), cabsf(hrtfs[band*NUM_EARS*N_dirs + 0*N_dirs + nd]) );
231  hrtfs[band*NUM_EARS*N_dirs + 1*N_dirs + nd] = crmulf( cexpf(cmplxf(0.0f,-ipd[band*N_dirs + nd])), cabsf(hrtfs[band*NUM_EARS*N_dirs + 1*N_dirs + nd]) );
232  }
233  }
234  free(ipd);
235  }
236  }

240 void interpHRTFs
241 (
242  float_complex* hrtfs, /* N_bands x 2 x N_hrtf_dirs */
243  float* itds,
244  float* freqVector,
245  float* vbap_gtable,
246  int N_hrtf_dirs,
247  int N_bands,
248  int N_interp_dirs,
249  float_complex* hrtfs_interp /* pre-alloc, N_bands x 2 x N_interp_dirs */
250 )
251 {
252  int i, band;
253  float* itd_interp, *mags_interp, *ipd_interp;
254  float** mags;
255 
256  mags = (float**)malloc1d(N_bands*sizeof(float*));
257  itd_interp = malloc1d(N_interp_dirs*sizeof(float));
258  mags_interp = malloc1d(N_interp_dirs*NUM_EARS*sizeof(float));
259  ipd_interp = malloc1d(N_interp_dirs*sizeof(float));
260 
261  /* calculate HRTF magnitudes */
262  for(band=0; band<N_bands; band++){
263  mags[band] = malloc1d(NUM_EARS * N_hrtf_dirs*sizeof(float));
264  for(i=0; i< NUM_EARS * N_hrtf_dirs ; i++)
265  mags[band][i] = cabsf(hrtfs[band*NUM_EARS * N_hrtf_dirs + i]);
266  }
267 
268  /* interpolate ITDs */
269  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, N_interp_dirs, 1, N_hrtf_dirs, 1.0f,
270  vbap_gtable, N_hrtf_dirs,
271  itds, 1, 0.0f,
272  itd_interp, 1);
273  for(band=0; band<N_bands; band++){
274  /* interpolate HRTF magnitudes */
275  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasTrans, N_interp_dirs, NUM_EARS, N_hrtf_dirs, 1.0f,
276  vbap_gtable, N_hrtf_dirs,
277  mags[band], N_hrtf_dirs, 0.0f,
278  mags_interp, NUM_EARS);
279 
280  /* convert ITDs to phase differences -pi..pi */
281  for(i=0; i<N_interp_dirs; i++)
282  ipd_interp[i] = (matlab_fmodf(2.0f*SAF_PI*freqVector[band]*itd_interp[i] + SAF_PI, 2.0f*SAF_PI) - SAF_PI)/2.0f; /* /2 here, not later */
283 
284  /* reintroduce the interaural phase differences (IPD) */
285  for(i=0; i<N_interp_dirs; i++){
286  hrtfs_interp[band*NUM_EARS*N_interp_dirs + 0* N_interp_dirs +i] = ccmulf( cmplxf(mags_interp[i*NUM_EARS+0],0.0f), cexpf(cmplxf(0.0f, ipd_interp[i])) );
287  hrtfs_interp[band*NUM_EARS*N_interp_dirs + 1* N_interp_dirs +i] = ccmulf( cmplxf(mags_interp[i*NUM_EARS+1],0.0f), cexpf(cmplxf(0.0f,-ipd_interp[i])) );
288  }
289  }
290 
291  free(itd_interp);
292  for(band=0; band<N_bands; band++)
293  free(mags[band]);
294  free(mags);
295  free(mags_interp);
296  free(ipd_interp);
297 }

299 void binauralDiffuseCoherence
300 (
301  float_complex* hrtfs, /* N_bands x 2 x N_hrtf_dirs */
302  float* itds,
303  float* freqVector,
304  int N_hrtf_dirs,
305  int N_bands,
306  float* HRTFcoh
307 )
308 {
309  int i, j;
310  float* ipd;
311  float_complex *hrtf_ipd_lr;
312 
313  /* convert ITDs to phase differences -pi..pi */
314  ipd = malloc1d(N_bands*N_hrtf_dirs*sizeof(float));
315  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasTrans, N_bands, N_hrtf_dirs, 1, 1.0,
316  freqVector, 1,
317  itds, 1, 0.0,
318  ipd, N_hrtf_dirs);
319  for(i=0; i<N_bands; i++)
320  for(j=0; j<N_hrtf_dirs; j++)
321  ipd[i*N_hrtf_dirs+j] = (matlab_fmodf(2.0f*SAF_PI*ipd[i*N_hrtf_dirs+j] + SAF_PI, 2.0f*SAF_PI) - SAF_PI);
322 
323  /* compute complex coherence */
324  hrtf_ipd_lr = calloc1d(N_bands, sizeof(float_complex));
325  for(i=0; i<N_bands; i++){
326  for(j=0; j<N_hrtf_dirs; j++)
327  hrtf_ipd_lr[i] = ccaddf(hrtf_ipd_lr[i], crmulf(cexpf(crmulf(cmplxf(0.0f, 1.0f), ipd[i*N_hrtf_dirs+j])),
328  cabsf(hrtfs[i*NUM_EARS*N_hrtf_dirs + 0*N_hrtf_dirs + j])*
329  cabsf(hrtfs[i*NUM_EARS*N_hrtf_dirs + 1*N_hrtf_dirs + j])));
330  hrtf_ipd_lr[i] = ccdivf(hrtf_ipd_lr[i], cmplxf((float)N_hrtf_dirs, 0.0f));
331  }
332 
333  /* due to almost axisymmetry of ITD, the coherence is almost real */
334  for(i=0; i<N_bands; i++)
335  HRTFcoh[i] = crealf(hrtf_ipd_lr[i]) < 0.0f ? 0.0f : crealf(hrtf_ipd_lr[i]);
336  HRTFcoh[0] = 1.0f; /* force 1 at DC */
337 
338  free(ipd);
339  free(hrtf_ipd_lr);
340 }
