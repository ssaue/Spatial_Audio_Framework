32 static void applyWindowingFunction
33 (
34  WINDOWING_FUNCTION_TYPES type,
35  int winlength,
36  float* x
37 )
38 {
39  int i, N;
40 
41  /* if winlength is odd -> symmetric window (mid index has value=1) */
42  if ( !(winlength % 2 == 0) )
43  N = winlength-1;
44  /* otherwise, if winlength is even (index: winlength/2+1 = 1.0, but first
45  * value != last value) */
46  else
47  N = winlength;
48 
49  switch(type){
50  case WINDOWING_FUNCTION_RECTANGULAR:
51  break;
52 
53  case WINDOWING_FUNCTION_HAMMING:
54  for(i=0; i<winlength; i++)
55  x[i] *= 0.54f - 0.46f * (cosf(2.0f*SAF_PI*(float)i/(float)N)); /* more wide-spread coefficient values */
56  /* optimal equiripple coefficient values: */
57  /*x[i] *= 0.53836f - 0.46164f * (cosf(2.0f*M_PI*(float)i/(float)N));*/
58  break;
59 
60  case WINDOWING_FUNCTION_HANN:
61  for(i=0; i<winlength; i++)
62  x[i] *= 0.5f - 0.5f * (cosf(2.0f*SAF_PI*(float)i/(float)N));
63  break;
64 
65  case WINDOWING_FUNCTION_BARTLETT:
66  for(i=0; i<winlength; i++)
67  x[i] *= 1.0f - 2.0f * fabsf((float)i-((float)N/2.0f))/(float)N;
68  break;
69 
70  case WINDOWING_FUNCTION_BLACKMAN:
71  for(i=0; i<winlength; i++){
72  x[i] *= 0.42659f -
73  0.49656f *cosf(2.0f*SAF_PI*(float)i/(float)N) +
74  0.076849f*cosf(4.0f*SAF_PI*(float)i/(float)N);
75  }
76  break;
77 
78  case WINDOWING_FUNCTION_NUTTALL:
79  for(i=0; i<winlength; i++){
80  x[i] *= 0.355768f -
81  0.487396f*cosf(2.0f*SAF_PI*(float)i/(float)N) +
82  0.144232f*cosf(4.0f*SAF_PI*(float)i/(float)N) -
83  0.012604f*cosf(6.0f*SAF_PI*(float)i/(float)N);
84  }
85  break;
86 
87  case WINDOWING_FUNCTION_BLACKMAN_NUTTALL:
88  for(i=0; i<winlength; i++){
89  x[i] *= 0.3635819f -
90  0.4891775f*cosf(2.0f*SAF_PI*(float)i/(float)N) +
91  0.1365995f*cosf(4.0f*SAF_PI*(float)i/(float)N) +
92  0.0106411f*cosf(4.0f*SAF_PI*(float)i/(float)N);
93  }
94  break;
95 
96  case WINDOWING_FUNCTION_BLACKMAN_HARRIS:
97  for(i=0; i<winlength; i++){
98  x[i] *= 0.35875f -
99  0.48829f*cosf(2.0f*SAF_PI*(float)i/(float)N) +
100  0.14128f*cosf(4.0f*SAF_PI*(float)i/(float)N) +
101  0.01168f*cosf(4.0f*SAF_PI*(float)i/(float)N);
102  }
103  break;
104  }
105 }

109 static void applyIIR_1
110 (
111  float* in_signal,
112  int nSamples,
113  float* b,
114  float* a,
115  float* wz,
116  float* out_signal
117 )
118 {
119  int n;
120  float wn;
121 
122  /* difference equation (Direct form 2) */
123  for (n=0; n<nSamples; n++){
124  /* numerator */
125  wn = in_signal[n] - (a[1] * wz[0]);
126 
127  /* denominator */
128  out_signal[n] = b[0] * wn + b[1] * wz[0];
129 
130  /* shuffle delays */
131  wz[0] = wn;
132  }

224 void flattenMinphase
225 (
226  float* x,
227  int len
228 )
229 {
230  int i;
231  float_complex* ctd_tmp, *tdi_f, *tdi_f_labs, *dt_min_f;
232  void* hFFT;
233 
234  /* prep */
235  ctd_tmp = malloc1d(len*sizeof(float_complex));
236  tdi_f = malloc1d(len*sizeof(float_complex));
237  tdi_f_labs = malloc1d(len*sizeof(float_complex));
238  dt_min_f = malloc1d(len*sizeof(float_complex));
239  saf_fft_create(&hFFT, len);
240 
241  /* fft */
242  for(i=0; i<len; i++)
243  ctd_tmp[i] = cmplxf(x[i], 0.0f);
244  saf_fft_forward(hFFT, (float_complex*)ctd_tmp, (float_complex*)tdi_f);
245 
246  /* take log(cabs()) */
247  for(i=0; i<len; i++)
248  tdi_f_labs[i] = cmplxf(logf(cabsf(tdi_f[i])), 0.0f);
249 
250  /* Hilbert to acquire discrete-time analytic signal */
251  hilbert(tdi_f_labs, len, dt_min_f);
252 
253  /* compute minimum-phase response, and apply to tdi_f to flatten it to unity magnitude */
254  for(i=0; i<len; i++)
255  dt_min_f[i] = ccdivf(tdi_f[i], cexpf(conjf(dt_min_f[i])));
256 
257  /* ifft */
258  saf_fft_backward(hFFT, dt_min_f, ctd_tmp);
259 
260  /* overwrite input with EQ'd version */
261  for(i=0; i<len; i++)
262  x[i] = crealf(ctd_tmp[i]);
263 
264  /* tidy up */
265  saf_fft_destroy(&hFFT);
266  free(ctd_tmp);
267  free(tdi_f);
268  free(tdi_f_labs);
269  free(dt_min_f);
270 }

337 void biQuadCoeffs
338 (
339  BIQUAD_FILTER_TYPES filterType,
340  float fc,
341  float fs,
342  float Q,
343  float gain_dB,
344  float b[3],
345  float a[3]
346 )
347 {
348  float K, KK, D, V0, A, w0, alpha, a0;
349 
350  a[0] = 1.0f;
351 
352  /* calculate the IIR filter coefficients */
353  switch (filterType){
354  case BIQUAD_FILTER_LPF:
355  /* Filter design equations - DAFX (2nd ed) p50 */
356  K = tanf(SAF_PI * fc/fs);
357  KK = K * K;
358  D = KK * Q + K + Q;
359  b[0] = (KK * Q)/D;
360  b[1] = (2.0f * KK * Q)/D;
361  b[2] = b[0];
362  a[1] = (2.0f * Q * (KK - 1.0f))/D;
363  a[2] = (KK * Q - K + Q)/D;
364  break;
365 
366  case BIQUAD_FILTER_LPF_EQCB:
367  /* Filter design equations - https://webaudio.github.io/Audio-EQ-Cookbook/audio-eq-cookbook.html */
368  w0 = 2.0f*SAF_PI*fc/fs;
369  alpha = sinf(w0)/(2.0f*Q);
370  b[0] = (1.0f - cosf(w0))/2.0f;
371  b[1] = 1.0f - cosf(w0);
372  b[2] = b[0];
373  a0 = 1.0f + alpha;
374  a[1] = -2.0f*cosf(w0);
375  a[2] = 1.0f - alpha;
376 
377  /* Scale by a0, since applyBiQuadFilter() and applyIIR() assume a0 = 1.0 */
378  b[0] /= a0;
379  b[1] /= a0;
380  b[2] /= a0;
381  a[1] /= a0;
382  a[2] /= a0;
383  break;
384 
385  case BIQUAD_FILTER_HPF:
386  /* Filter design equations - DAFX (2nd ed) p50 */
387  K = tanf(SAF_PI * fc/fs);
388  KK = K * K;
389  D = KK * Q + K + Q;
390  b[0] = (Q)/D;
391  b[1] = -(2.0f * Q)/D;
392  b[2] = b[0];
393  a[1] = (2.0f * Q * (KK - 1.0f))/D;
394  a[2] = (KK * Q - K + Q)/D;
395  break;
396 
397  case BIQUAD_FILTER_HPF_EQCB:
398  /* Filter design equations - https://webaudio.github.io/Audio-EQ-Cookbook/audio-eq-cookbook.html */
399  w0 = 2.0f*SAF_PI*fc/fs;
400  alpha = sinf(w0)/(2.0f*Q);
401  b[0] = (1.0f + cosf(w0))/2.0f;
402  b[1] = -(1.0f + cosf(w0));
403  b[2] = b[0];
404  a0 = 1.0f + alpha;
405  a[1] = -2.0f*cosf(w0);
406  a[2] = 1.0f - alpha;
407 
408  /* Scale by a0, since applyBiQuadFilter() and applyIIR() assume a0 = 1.0 */
409  b[0] /= a0;
410  b[1] /= a0;
411  b[2] /= a0;
412  a[1] /= a0;
413  a[2] /= a0;
414  break;
415 
416  case BIQUAD_FILTER_LOW_SHELF:
417  /* Filter design equations - DAFX (2nd ed) p64 */
418  K = tanf(SAF_PI * fc/fs);
419  V0 = powf(10.0f, (gain_dB/20.0f));
420  if (V0 < 1.0f)
421  V0 = 1.0f/V0;
422  KK = K * K;
423  if (gain_dB > 0.0f){
424  D = 1.0f + sqrtf(2.0f) * K + KK;
425  b[0] = (1.0f + sqrtf(2.0f * V0) * K + V0 * KK)/D;
426  b[1] = (2.0f*(V0*KK - 1.0f))/D;
427  b[2] = (1.0f - sqrtf(2.0f * V0) * K + V0 * KK)/D;
428  a[1] = (2.0f * (KK - 1.0f))/D;
429  a[2] = (1.0f - sqrtf(2.0f) * K + KK)/D;
430  }
431  else{
432  D = V0 + sqrtf(2.0f*V0)*K + KK;
433  b[0] = (V0*(1.0f + sqrtf(2.0f)*K + KK))/D;
434  b[1] = (2.0f*V0*(KK - 1.0f))/D;
435  b[2] = (V0*(1.0f - sqrtf(2.0f)*K + KK))/D;
436  a[1] = (2.0f * (KK - V0))/D;
437  a[2] = (V0 - sqrtf(2.0f*V0)*K + KK)/D;
438  }
439  break;
440 
441  case BIQUAD_FILTER_LOW_SHELF_EQCB:
442  /* Filter design equations - https://webaudio.github.io/Audio-EQ-Cookbook/audio-eq-cookbook.html */
443  A = powf(10.0f, gain_dB/40.0f);
444  w0 = 2.0f*SAF_PI*fc/fs;
445  alpha = sinf(w0)/(2.0f*Q);
446  b[0] = A*( (A+1.0f) - (A-1.0f) * cosf(w0) + 2.0f*sqrtf(A)*alpha );
447  b[1] = 2.0f*A*( (A-1.0f) - (A+1.0f) * cosf(w0) );
448  b[2] = A*( (A+1.0f) - (A-1.0f) * cosf(w0) - 2.0f*sqrtf(A)*alpha );
449  a0 = (A+1.0f) + (A-1.0f) * cosf(w0) + 2.0f*sqrtf(A)*alpha;
450  a[1] = -2.0f*( (A-1.0f) + (A+1.0f) * cosf(w0) );
451  a[2] = (A+1.0f) + (A-1.0f) * cosf(w0) - 2.0f*sqrtf(A)*alpha;
452 
453  /* Scale by a0, since applyBiQuadFilter() and applyIIR() assume a0 = 1.0 */
454  b[0] /= a0;
455  b[1] /= a0;
456  b[2] /= a0;
457  a[1] /= a0;
458  a[2] /= a0;
459  break;
460 
461  case BIQUAD_FILTER_HI_SHELF:
462  /* Filter design equations - DAFX (2nd ed) p64 */
463  K = tanf(SAF_PI * fc/fs);
464  V0 = powf(10.0f, (gain_dB/20.0f));
465  if (V0 < 1.0f)
466  V0 = 1.0f/V0;
467  KK = K * K;
468  if (gain_dB > 0.0f){
469  D = 1.0f + sqrtf(2.0f) * K + KK;
470  b[0] = (V0 + sqrtf(2.0f * V0) * K + KK)/D;
471  b[1] = (2.0f*(KK - V0))/D;
472  b[2] = (V0 - sqrtf(2.0f * V0) * K + KK)/D;
473  a[1] = (2.0f*(KK - 1.0f))/D;
474  a[2] = (1.0f - sqrtf(2.0f) * K + KK)/D;
475  }
476  else{
477  D = 1.0f + sqrtf(2.0f*V0) * K + V0*KK;
478  b[0] = (V0*(1.0f + sqrtf(2.0f)*K + KK))/D;
479  b[1] = (2.0f*V0*(KK - 1.0f))/D;
480  b[2] = (V0*(1.0f - sqrtf(2.0f)*K + KK))/D;
481  a[1] = (2.0f * (V0*KK - 1.0f))/D;
482  a[2] = (1.0f - sqrtf(2.0f*V0)*K + V0*KK)/D;
483  }
484  break;
485 
486  case BIQUAD_FILTER_HI_SHELF_EQCB:
487  /* Filter design equations - https://webaudio.github.io/Audio-EQ-Cookbook/audio-eq-cookbook.html */
488  A = powf(10.0f, gain_dB/40.0f);
489  w0 = 2.0f*SAF_PI*fc/fs;
490  alpha = sinf(w0)/(2.0f*Q);
491  b[0] = A*( (A+1.0f) + (A-1.0f) * cosf(w0) + 2.0f*sqrtf(A)*alpha );
492  b[1] = -2.0f*A*( (A-1.0f) + (A+1.0f) * cosf(w0) );
493  b[2] = A*( (A+1.0f) + (A-1.0f) * cosf(w0) - 2.0f*sqrtf(A)*alpha );
494  a0 = (A+1.0f) - (A-1.0f) * cosf(w0) + 2.0f*sqrtf(A)*alpha;
495  a[1] = 2.0f*( (A-1.0f) - (A+1.0f) * cosf(w0) );
496  a[2] = (A+1.0f) - (A-1.0f) * cosf(w0) - 2.0f*sqrtf(A)*alpha;
497 
498  /* Scale by a0, since applyBiQuadFilter() and applyIIR() assume a0 = 1.0 */
499  b[0] /= a0;
500  b[1] /= a0;
501  b[2] /= a0;
502  a[1] /= a0;
503  a[2] /= a0;
504  break;
505 
506  case BIQUAD_FILTER_PEAK:
507  /* Filter design equations - DAFX (2nd ed) p66 */
508  K = tanf(SAF_PI * fc/fs);
509  V0 = powf(10.0f, (gain_dB/20.0f));
510  KK = K * K;
511  if (gain_dB > 0.0f){
512  D = 1.0f + (K/Q) + KK;
513  b[0] = (1.0f + (V0/Q) * K + KK)/D;
514  b[1] = (2.0f*(KK - 1.0f))/D;
515  b[2] = (1.0f - (V0/Q) * K + KK)/D;
516  a[1] = b[1];
517  a[2] = (1.0f - (K/Q) + KK)/D;
518  }
519  else {
520  D = 1.0f + (K/(V0*Q)) + KK;
521  b[0] = (1.0f + (K/Q) + KK)/D;
522  b[1] = (2.0f*(KK - 1.0f))/D;
523  b[2] = (1.0f - (K/Q) + KK)/D;
524  a[1] = b[1];
525  a[2] = (1.0f - (K/(V0*Q)) + KK)/D;
526  }
527  break;
528 
529  case BIQUAD_FILTER_PEAK_EQCB:
530  /* Filter design equations - https://webaudio.github.io/Audio-EQ-Cookbook/audio-eq-cookbook.html */
531  A = powf(10.0f, gain_dB/40.0f);
532  w0 = 2.0f*SAF_PI*fc/fs;
533  alpha = sinf(w0)/(2.0f*Q);
534  b[0] = 1.0f + alpha*A;
535  b[1] = -2.0f*cosf(w0);
536  b[2] = 1.0f - alpha*A;
537  a0 = 1.0f + alpha/A;
538  a[1] = b[1];
539  a[2] = 1.0f - alpha/A;
540 
541  /* Scale by a0, since applyBiQuadFilter() and applyIIR() assume a0 = 1.0 */
542  b[0] /= a0;
543  b[1] /= a0;
544  b[2] /= a0;
545  a[1] /= a0;
546  a[2] /= a0;
547  break;
548  }

551 void applyBiQuadFilter
552 (
553  float b[3],
554  float a[3],
555  float w_z_12[2],
556  float* signal,
557  int nSamples
558 )
559 {
560  int n;
561  float wn;
562 
563  /* biquad difference equation (Direct form 2) */
564  for(n=0; n<nSamples; n++){
565  wn = signal[n] - a[1] * w_z_12[0] - a[2] * w_z_12[1];
566  signal[n] = b[0] * wn + b[1] * w_z_12[0] + b[2] * w_z_12[1];
567  /* shuffle delays */
568  w_z_12[1] = w_z_12[0];
569  w_z_12[0] = wn;
570  }

573 void evalBiQuadTransferFunction
574 (
575  float b[3],
576  float a[3],
577  float* freqs,
578  int nFreqs,
579  float fs,
580  int mag2dB,
581  float* magnitude,
582  float* phase_rad
583 )
584 {
585  int ff;
586  float w, denom_real, denom_imag, num_real, num_imag;
587 
588  for(ff=0; ff<nFreqs; ff++){
589  w = tanf(SAF_PI * freqs[ff]/fs);
590  /* substituting Euler, z = e^(-jwn) = cos(wn) + j*sin(wn), into:
591  * H(z) = (b0 + b1*z^(-1) + b2*z^(-2)) / (1 + a1*z^(-1) + a2*z^(-2)): */
592  denom_real = 1.0f + a[1]*cosf(w) + a[2]*cosf(2.0f*w);
593  denom_imag = a[1]*sinf(w) + a[2]*sinf(2*w);
594  num_real = b[0] + b[1]*cosf(w) + b[2]*cosf(2.0f*w);
595  num_imag = b[1]*sinf(w) + b[2]*sinf(2.0f*w);
596 
597  if(magnitude!=NULL){
598  magnitude[ff] = sqrtf( (powf(num_real, 2.0f) + powf(num_imag, 2.0f)) / (powf(denom_real, 2.0f) + powf(denom_imag, 2.0f) + 2.23e-7f) );
599  if(mag2dB)
600  magnitude[ff] = 20.0f*log10f(magnitude[ff]);
601  }
602  if(phase_rad!=NULL)
603  phase_rad[ff] = atan2f(num_imag,num_real) - atan2f(denom_imag, denom_real);
604  }
605 }

607 void applyIIR
608 (
609  float* in_signal,
610  int nSamples,
611  int nCoeffs,
612  float* b,
613  float* a,
614  float* wz,
615  float* out_signal
616 )
617 {
618  int n, i;
619  float wn;
620 
621  /* For compiler speed-ups */
622  switch(nCoeffs){
623  case 1: assert(0); /* just divide in_signal by b[0]... */
624  case 2: applyIIR_1(in_signal, nSamples, b, a, wz, out_signal); return;
625  case 3: applyIIR_2(in_signal, nSamples, b, a, wz, out_signal); return;
626  case 4: applyIIR_3(in_signal, nSamples, b, a, wz, out_signal); return;
627  }
628 
629  /* difference equation (Direct form 2) */
630  for (n=0; n<nSamples; n++){
631  /* numerator */
632  wn = in_signal[n];
633  for (i=1; i<nCoeffs;i++)
634  wn = wn - (a[i] * wz[i-1]);
635 
636  /* denominator */
637  out_signal[n] = b[0] * wn;
638  for (i=1; i<nCoeffs; i++)
639  out_signal[n] = out_signal[n] + (b[i] * wz[i-1]);
640 
641  /* shuffle delays */
642  switch(nCoeffs-1){
643  case 10: wz[9] = wz[8]; /* fall through */
644  case 9: wz[8] = wz[7]; /* fall through */
645  case 8: wz[7] = wz[6]; /* fall through */
646  case 7: wz[6] = wz[5]; /* fall through */
647  case 6: wz[5] = wz[4]; /* fall through */
648  case 5: wz[4] = wz[3]; /* fall through */
649  case 4: wz[3] = wz[2]; /* fall through */
650  case 3: wz[2] = wz[1]; /* fall through */
651  case 2: wz[1] = wz[0]; /* fall through */
652  case 1: wz[0] = wn; break;
653  default: assert(0); /* A 5th order BPF/BSF or 10th order LPF/HPF?
654  * Sorry, I gotta put a stop to that... */
655  }
656  }

659 void butterCoeffs
660 (
661  BUTTER_FILTER_TYPES filterType,
662  int order,
663  float cutoff1,
664  float cutoff2,
665  float sampleRate,
666  double* b_coeffs,
667  double* a_coeffs
668 )
669 {
670  int i, j, k, np, tmp_len, numStates, oddPolesFLAG, nCoeffs;
671  double wlow, whi, w0, wl, w1, BW, Wn1, q;
672  double den[3];
673  double* c_state, *r, *b_coeffs_real;
674  double** a_state, **bf_ss, **tmp1, **tmp2, **a_bili;
675  double_complex kaT, kbT;
676  double_complex den_cmplx[3];
677  double_complex* proto, *proto_tmp, *a_coeffs_cmplx, *kern, *rcmplx, *b_coeffs_cmplx;
678 
679  wlow = (double)cutoff1/((double)sampleRate/2.0);
680  whi = (double)cutoff2/((double)sampleRate/2.0);
681  w0 = 4.0 * tan(SAF_PI*wlow/2.0);
682  Wn1 = 0.0;
683 
684  /* Compute prototype for Nth order Butterworth analogue lowpass filter */
685  if (order%2 != 0){/* ISODD */
686  tmp_len = (int)((float)order/2.0f); /* floor */
687  np = 2*tmp_len+1;
688  proto = malloc1d(np*sizeof(double_complex));
689  proto[np-1] = cmplx(-1.0,0.0);
690  }
691  else{ /* ISEVEN */
692  tmp_len = order/2;
693  np = 2*tmp_len;
694  proto = malloc1d(np*sizeof(double_complex));
695  }
696  proto_tmp = malloc1d(np*sizeof(double_complex));
697  for(i=1, j=0; i<=order-1; i+=2, j++)
698  proto_tmp[j] = cexp(cmplx(0.0, SAF_PI*(double)i/(2.0*(double)order) + SAF_PI/2.0) );
699  for (i=0; i<tmp_len; i++){
700  proto[2*i] = proto_tmp[i];
701  proto[2*i+1] = conj(proto_tmp[i]);
702  }
703 
704  /* Transform prototype into state space */
705  numStates = np;
706  cmplxPairUp(proto, proto_tmp, np);
707  memcpy(proto, proto_tmp, np*sizeof(double_complex));
708  free(proto_tmp);
709  a_state = (double**)calloc2d(numStates,numStates,sizeof(double));
710  c_state = malloc1d(numStates*sizeof(double));
711  if (np%2 != 0){/* ISODD */
712  a_state[0][0] = creal(proto[np-1]);
713  c_state[0] = 1.0;
714  np--;
715  oddPolesFLAG = 1;
716  }
717  else
718  oddPolesFLAG = 0;
719 
720  /* Adjust indices as needed */
721  for(i=1; i<np; i+=2){
722  polyz_v(&proto[i-1], den_cmplx, 2);
723  for(j=0; j<3; j++)
724  den[j] = creal(den_cmplx[j]);
725  j = oddPolesFLAG ? i-1 : i-2;
726 
727  if(j==-1){
728  a_state[0][0] = -den[1];
729  a_state[0][1] = -den[2];
730  a_state[1][0] = 1.0;
731  a_state[1][1] = 0.0;
732  c_state[0] = 0.0;
733  c_state[1] = 1.0;
734  }
735  else{
736  for(k=0; k<j+1; k++)
737  a_state[j+1][k] = c_state[k];
738  a_state[j+1][j+1] = -den[1];
739  a_state[j+1][j+2] = -den[2];
740  a_state[j+2][j+1] = 1.0;
741  a_state[j+2][j+2] = 0.0;
742 
743  for(k=0; k<j+1; k++)
744  c_state[k] = 0.0;
745  c_state[j+1] = 0.0;
746  c_state[j+2] = 1.0;
747  }
748  }
749 
750  /* Transform lowpass filter into the desired filter (while in state space) */
751  bf_ss = NULL;
752  switch(filterType){
753  case BUTTER_FILTER_HPF:
754  utility_dinv(FLATTEN2D(a_state), FLATTEN2D(a_state), numStates);
755  /* fall through */
756  case BUTTER_FILTER_LPF:
757  bf_ss = (double**)malloc2d(numStates,numStates,sizeof(double));
758  for(i=0; i<numStates; i++)
759  for(j=0; j<numStates; j++)
760  bf_ss[i][j] = w0*(a_state[i][j]);
761  break;
762  case BUTTER_FILTER_BSF:
763  utility_dinv(FLATTEN2D(a_state), FLATTEN2D(a_state), numStates);
764  /* fall through */
765  case BUTTER_FILTER_BPF:
766  numStates = numStates*2;
767  w1 = 4.0*tan(SAF_PI*whi/2.0);
768  BW = w1 - w0;
769  Wn1 = sqrt(w0*w1);
770  q = Wn1/BW;
771  bf_ss = (double**)calloc2d(numStates,numStates,sizeof(double));
772  for(i=0; i<numStates/2; i++)
773  for(j=0; j<numStates/2; j++)
774  bf_ss[i][j] = Wn1 * (a_state[i][j]) /q;
775  for(i=numStates/2; i<numStates; i++)
776  for(j=0; j<numStates/2; j++)
777  bf_ss[i][j] = (i-numStates/2) == j ? -Wn1 : 0.0;
778  for(i=0; i<numStates/2; i++)
779  for(j=numStates/2; j<numStates; j++)
780  bf_ss[i][j] = i == (j-numStates/2) ? Wn1 : 0.0;
781  break;
782  }
783  assert(bf_ss != NULL);
784  nCoeffs = numStates+1;
785 
786  /* Bilinear transformation to find the discrete equivalent of the filter */
787  tmp1 = (double**)malloc2d(numStates,numStates,sizeof(double));
788  tmp2 = (double**)malloc2d(numStates,numStates,sizeof(double));
789  a_bili = (double**)malloc2d(numStates,numStates,sizeof(double));
790  for(i=0; i<numStates; i++){
791  for(j=0; j<numStates; j++){
792  tmp1[i][j] = (i==j ? 1.0f : 0.0f) + bf_ss[i][j]*0.25;
793  tmp2[i][j] = (i==j ? 1.0f : 0.0f) - bf_ss[i][j]*0.25;
794  }
795  }
796  utility_dglslv(FLATTEN2D(tmp2), numStates, FLATTEN2D(tmp1), numStates, FLATTEN2D(a_bili));
797 
798  /* Compute the filter coefficients for the numerator and denominator */
799  a_coeffs_cmplx = malloc1d(nCoeffs*sizeof(double_complex));
800  polyd_m(FLATTEN2D(a_bili), a_coeffs_cmplx, numStates);
801  rcmplx = NULL;
802  r = NULL;
803  switch(filterType){
804  default:
805  /* fall through */
806  case BUTTER_FILTER_LPF:
807  r = malloc1d(numStates*sizeof(double));
808  for(i=0; i<numStates; i++)
809  r[i] = -1.0;
810  wl = 0.0;
811  break;
812  case BUTTER_FILTER_HPF:
813  r = malloc1d(numStates*sizeof(double));
814  for(i=0; i<numStates; i++)
815  r[i] = 1.0;
816  wl = SAF_PI;
817  break;
818  case BUTTER_FILTER_BPF:
819  r = malloc1d(numStates*sizeof(double));
820  wl = 2.0*atan2(Wn1, 4.0);
821  for(i=0; i<order;i++)
822  r[i] = 1.0;
823  for(; i<2*order;i++)
824  r[i] = -1.0;
825  break;
826  case BUTTER_FILTER_BSF:
827  rcmplx = malloc1d(numStates*sizeof(double_complex));
828  Wn1 = 2.0*atan2(Wn1,4.0);
829  wl = 0.0;
830  for(i=0; i<numStates;i++)
831  rcmplx[i] = cexp(cmplx(0.0, Wn1*pow(-1.0,(double)i)));
832  break;
833  }
834  b_coeffs_real = malloc1d(nCoeffs*sizeof(double));
835  if(filterType == BUTTER_FILTER_BSF){
836  b_coeffs_cmplx = malloc1d(nCoeffs*sizeof(double_complex));
837  polyz_v(rcmplx, b_coeffs_cmplx, numStates);
838  for(i=0; i<nCoeffs; i++)
839  b_coeffs_real[i] = creal(b_coeffs_cmplx[i]);
840  free(b_coeffs_cmplx);
841  }
842  else
843  polyd_v(r, b_coeffs_real, numStates);
844  kern = calloc1d(nCoeffs,sizeof(double_complex));
845  kaT = cmplx(0.0,0.0);
846  kbT = cmplx(0.0,0.0);
847  for(i=0; i<nCoeffs; i++){
848  kern[i] = cexp(cmplx(0.0,-wl*(double)i));
849  kaT = ccadd(kaT, crmul(kern[i],creal(a_coeffs_cmplx[i])));
850  kbT = ccadd(kbT, crmul(kern[i],b_coeffs_real[i]));
851  }
852 
853  /* output */
854  for(i=0; i<nCoeffs; i++){
855  b_coeffs[i] = creal(crmul(ccdiv(kaT,kbT), b_coeffs_real[i]));
856  a_coeffs[i] = creal(a_coeffs_cmplx[i]);
857  }
858 
859  /* clean-up */
860  free(proto);
861  free(a_state);
862  free(c_state);
863  free(bf_ss);
864  free(tmp1);
865  free(tmp2);
866  free(a_bili);
867  free(a_coeffs_cmplx);
868  free(b_coeffs_real);
869  free(kern);
870 }

893 void faf_IIRFilterbank_create
894 (
895  void** phFaF,
896  int order,
897  float* fc,
898  int nCutoffFreq,
899  float sampleRate,
900  int maxNumSamples
901 )
902 {
903  *phFaF = malloc1d(sizeof(faf_IIRFB_data));
904  faf_IIRFB_data *fb = (faf_IIRFB_data*)(*phFaF);
905  double b_lpf[4], a_lpf[4], b_hpf[4], a_hpf[4], r[7], revb[4], reva[4], q[4];
906  double tmp[7], tmp2[7];
907  double_complex d1[3], d2[3], d1_num[3], d2_num[3];
908  double_complex z[3], A[3][3], ztmp[7], ztmp2[7];
909  int i, j, f, filtLen, d1_len, d2_len;
910 
911  assert( (order==1) || (order==3) );
912  assert(nCutoffFreq>1);
913  filtLen = order + 1;
914  fb->filtOrder = order;
915  fb->filtLen = filtLen;
916 
917  /* Number of bands is always one more than the number of cut-off
918  * frequencies */
919  fb->nFilters = nCutoffFreq;
920  fb->nBands = nCutoffFreq + 1;
921 
922  /* Allocate memory for filter coefficients and delay buffers */
923  fb->b_hpf = (float**)malloc2d(nCutoffFreq, filtLen, sizeof(float));
924  fb->a_hpf = (float**)malloc2d(nCutoffFreq, filtLen, sizeof(float));
925  fb->b_lpf = (float**)malloc2d(nCutoffFreq, filtLen, sizeof(float));
926  fb->a_lpf = (float**)malloc2d(nCutoffFreq, filtLen, sizeof(float));
927  fb->wz_hpf = (float***)calloc3d(fb->nBands, nCutoffFreq, order, sizeof(float));
928  fb->wz_lpf = (float***)calloc3d(fb->nBands, nCutoffFreq, order, sizeof(float));
929  fb->wz_apf1 = (float***)calloc3d(fb->nBands, nCutoffFreq, order, sizeof(float));
930  fb->wz_apf2 = (float***)calloc3d(fb->nBands, nCutoffFreq, order, sizeof(float));
931  fb->maxNSamplesToExpect = maxNumSamples;
932  fb->tmp = malloc1d(maxNumSamples*sizeof(float));
933  fb->tmp2 = malloc1d(maxNumSamples*sizeof(float));
934 
935  /* Compute low-pass and complementary high-pass filter coefficients for each
936  * cut-off frequency */
937  for(f=0; f<nCutoffFreq; f++){
938  /* Low-pass filter */
939  butterCoeffs(BUTTER_FILTER_LPF, order, fc[f], 0.0f, sampleRate, (double*)b_lpf, (double*)a_lpf);
940 
941  /* IIR power complementary filter design (i.e. High-pass) */
942  for(i=0; i<filtLen; i++){
943  reva[i] = a_lpf[filtLen-i-1];
944  revb[i] = b_lpf[filtLen-i-1];
945  }
946  convd(revb, b_lpf, filtLen, filtLen, tmp);
947  convd(a_lpf, reva, filtLen, filtLen, tmp2);
948  for(i=0; i<2*filtLen-1; i++)
949  r[i] = tmp[i] - tmp2[i];
950  q[0] = sqrt(-r[0]/-1.0);
951  q[1] = -r[1]/(2.0*-1.0*q[0]);
952  if(order==3){
953  //q[3]=conj(-1.0*q[0]);
954  //q[2]=conj(-1.0*q[1]);
955  q[3] = -1.0*q[0];
956  q[2] = -1.0*q[1];
957  }
958  for(i=0; i<filtLen; i++)
959  q[i] = b_lpf[i] - q[i];
960 
961  /* Find roots of polynomial */
962  if(order==1)
963  z[0] = cmplx(-q[1]/q[0], 0.0);
964  else{ /* 3rd order */
965  memset(A, 0, 9*sizeof(double_complex));
966  A[0][0] = cmplx(-q[1]/q[0], 0.0);
967  A[0][1] = cmplx(-q[2]/q[0], 0.0);
968  A[0][2] = cmplx(-q[3]/q[0], 0.0);
969  A[1][0] = cmplx(1.0, 0.0);
970  A[2][1] = cmplx(1.0, 0.0);
971  utility_zeig((double_complex*)A, 3, NULL, NULL, NULL, (double_complex*)z);
972  }
973 
974  /* Separate the zeros inside the unit circle and the ones outside to
975  * form the allpass functions */
976  d1[0] = cmplx(1.0, 0.0);
977  d2[0] = cmplx(1.0, 0.0);
978  d1_len = d2_len = 1;
979  for(i=0; i<order; i++){
980  if (cabs(z[i]) < 1.0){
981  ztmp[0] = cmplx(1.0, 0.0);
982  ztmp[1] = crmul(z[i], -1.0);
983  convz(d2,ztmp,d2_len,2,ztmp2);
984  d2_len++;
985  for(j=0; j<d2_len; j++)
986  d2[j] = ztmp2[j];
987  }
988  else{
989  ztmp[0] = cmplx(1.0, 0.0);
990  ztmp[1] = ccdiv(cmplx(-1.0, 0.0), conj(z[i]));
991  convz(d1,ztmp,d1_len,2,ztmp2);
992  d1_len++;
993  for(j=0; j<d1_len; j++)
994  d1[j] = ztmp2[j];
995  }
996  }
997 
998  /* Convert coupled allpass filter to transfer function form (code from:
999  * https://github.com/nsk1001/Scilab-functions written by Nagma Samreen
1000  * Khan) */
1001  for(i=0; i<d1_len; i++)
1002  d1_num[i] = conj(d1[d1_len-i-1]);
1003  for(i=0; i<d2_len; i++)
1004  d2_num[i] = conj(d2[d2_len-i-1]);
1005  convz(d1_num, d2, d1_len, d2_len, ztmp);
1006  convz(d2_num, d1, d2_len, d1_len, ztmp2);
1007  for(i=0; i<filtLen; i++){
1008  b_hpf[i] = -0.5 * creal(ccsub(ztmp[filtLen-i-1], ztmp2[filtLen-i-1]));
1009  a_hpf[i] = a_lpf[i];
1010  }
1011 
1012  /* Store in single precision for run-time */
1013  for(i=0; i<filtLen; i++){
1014  fb->b_hpf[f][i] = (float)b_hpf[i];
1015  fb->a_hpf[f][i] = (float)a_hpf[i];
1016  fb->b_lpf[f][i] = (float)b_lpf[i];
1017  fb->a_lpf[f][i] = (float)a_lpf[i];
1018  }
1019  }
1020 }

1022 void faf_IIRFilterbank_apply
1023 (
1024  void* hFaF,
1025  float* inSig,
1026  float** outBands,
1027  int nSamples
1028 )
1029 {
1030  faf_IIRFB_data *fb = (faf_IIRFB_data*)(hFaF);
1031  int band,j;
1032 
1033  assert(nSamples <= fb->maxNSamplesToExpect);
1034 
1035  /* Copy input signal to all output bands/channels */
1036  for(band=0; band<fb->nBands; band++)
1037  memcpy(outBands[band], inSig, nSamples*sizeof(float));
1038 
1039  /* Band 0 */
1040  for (j = 0; j<fb->nFilters; j++)
1041  applyIIR(outBands[0], nSamples, fb->filtLen, fb->b_lpf[j], fb->a_lpf[j], fb->wz_lpf[0][j], outBands[0]);
1042 
1043  /* Band 1 */
1044  applyIIR(outBands[1], nSamples, fb->filtLen, fb->b_hpf[0], fb->a_hpf[0], fb->wz_hpf[1][0], outBands[1]);
1045  for (j = 1; j<fb->nFilters; j++)
1046  applyIIR(outBands[1], nSamples, fb->filtLen, fb->b_lpf[j], fb->a_lpf[j], fb->wz_lpf[1][j], outBands[1]);
1047 
1048  /* All-pass filters (bands 2..N-1) */
1049  for (band = 2; band < fb->nBands; band++){
1050  for (j=0; j<=band-2; j++){
1051  applyIIR(outBands[band], nSamples, fb->filtLen, fb->b_lpf[j], fb->a_lpf[j], fb->wz_apf1[band][j], fb->tmp);
1052  applyIIR(outBands[band], nSamples, fb->filtLen, fb->b_hpf[j], fb->a_hpf[j], fb->wz_apf2[band][j], fb->tmp2);
1053  utility_svvadd(fb->tmp, fb->tmp2, nSamples, outBands[band]);
1054  }
1055  }
1056 
1057  /* Bands 2..N-2 */
1058  for(band = 2; band< fb->nBands-1; band++){
1059  /* high-pass filter */
1060  applyIIR(outBands[band], nSamples, fb->filtLen, fb->b_hpf[band-1], fb->a_hpf[band-1], fb->wz_hpf[band][band-1], outBands[band]);

1116 void FIRCoeffs
1117 (
1118  FIR_FILTER_TYPES filterType,
1119  int order,
1120  float fc1,
1121  float fc2, /* only needed for band-pass/stop */
1122  float fs,
1123  WINDOWING_FUNCTION_TYPES windowType,
1124  int scalingFLAG,
1125  float* h_filt
1126 )
1127 {
1128  int i, h_len;
1129  float ft1, ft2, h_sum, f0;
1130  float_complex h_z_sum;
1131 
1132  h_len = order + 1;
1133  ft1 = fc1/(fs*2.0f);
1134 
1135  /* compute filter weights */
1136  if(order % 2 == 0){
1137  /* if order is multiple of 2 */
1138  switch(filterType){
1139  case FIR_FILTER_LPF:
1140  for(i=0; i<h_len; i++)
1141  h_filt[i] = i==order/2 ? 2.0f*ft1 : sinf(2.0f*SAF_PI*ft1*(float)(i-order/2)) / (SAF_PI*(float)(i-order/2));
1142  break;
1143 
1144  case FIR_FILTER_HPF:
1145  for(i=0; i<h_len; i++)
1146  h_filt[i] = i==order/2 ? 1.0f - 2.0f*ft1 : -sinf(2.0f*ft1*SAF_PI*(float)(i-order/2)) / (SAF_PI*(float)(i-order/2));
1147  break;
1148 
1149  case FIR_FILTER_BPF:
1150  ft2 = fc2/(fs*2.0f);
1151  for(i=0; i<h_len; i++){
1152  h_filt[i] = i==order/2 ? 2.0f*(ft2-ft1) :
1153  sinf(2.0f*SAF_PI*ft2*(float)(i-order/2)) / (SAF_PI*(float)(i-order/2)) - sinf(2.0f*SAF_PI*ft1*(float)(i-order/2)) / (SAF_PI*(float)(i-order/2));
1154  }
1155  break;
1156 
1157  case FIR_FILTER_BSF:
1158  ft2 = fc2/(fs*2.0f);
1159  for(i=0; i<h_len; i++){
1160  h_filt[i] = i==order/2 ? 1.0f - 2.0f*(ft2-ft1) :
1161  sinf(2.0f*SAF_PI*ft1*(float)(i-order/2)) / (SAF_PI*(float)(i-order/2)) - sinf(2.0f*SAF_PI*ft2*(float)(i-order/2)) / (SAF_PI*(float)(i-order/2));
1162  }
1163  break;
1164  }
1165  }
1166  else
1167  assert(0); /* please specify an even value for the filter 'order' argument */
1168 
1169  /* Apply windowing function */
1170  applyWindowingFunction(windowType, h_len, h_filt);
1171 
1172  /* Scaling, to ensure pass-band is truely at 1 (0dB).
1173  * [1] "Programs for Digital Signal Processing", IEEE Press John Wiley &
1174  * Sons, 1979, pg. 5.2-1.
1175  */
1176  if(scalingFLAG){
1177  switch(filterType){
1178  case FIR_FILTER_LPF:
1179  case FIR_FILTER_BSF:
1180  h_sum = 0.0f;
1181  for(i=0; i<h_len; i++)
1182  h_sum += h_filt[i];
1183  for(i=0; i<h_len; i++)
1184  h_filt[i] /= h_sum;
1185  break;
1186 
1187  case FIR_FILTER_HPF:
1188  f0 = 1.0f;
1189  h_z_sum = cmplxf(0.0f, 0.0f);
1190  for(i=0; i<h_len; i++)
1191  h_z_sum = ccaddf(h_z_sum, crmulf(cexpf(cmplxf(0.0f, -2.0f*SAF_PI*(float)i*f0/2.0f)), h_filt[i]));
1192  h_sum = cabsf(h_z_sum);
1193  for(i=0; i<h_len; i++)
1194  h_filt[i] /= h_sum;
1195  break;
1196 
1197  case FIR_FILTER_BPF:
1198  f0 = (fc1/fs+fc2/fs)/2.0f;
1199  h_z_sum = cmplxf(0.0f, 0.0f);
1200  for(i=0; i<h_len; i++)
1201  h_z_sum = ccaddf(h_z_sum, crmulf(cexpf(cmplxf(0.0f, -2.0f*SAF_PI*(float)i*f0/2.0f)), h_filt[i]));
1202  h_sum = cabsf(h_z_sum);
1203  for(i=0; i<h_len; i++)
1204  h_filt[i] /= h_sum;
1205  break;
1206  }
1207  }
1208 }

1212  int order,
1213  float* fc, /* cut-off frequencies; nCutoffFreq x 1 */

1216  WINDOWING_FUNCTION_TYPES windowType,
