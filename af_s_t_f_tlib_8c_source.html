81 static void afAnalyse
82 (
83  float* inTD/* nSamplesTD x nCH */,
84  int nSamplesTD,
85  int nCH,
86  int hopSize,
87  int LDmode,
88  int hybridmode,
89  float_complex* outTF /* out_nBands x nTimeslots x nCH */
90 )
91 {
92  int t, ch, sample, band;
93  void* hSTFT;
94  float_complex*** FrameTF;
95  float** tempFrameTD;
96  int nTimeSlots, nBands;
97 
98  nBands = hopSize + (hybridmode ? 5 : 1);
99  nTimeSlots = nSamplesTD/hopSize;
100 
101  /* allocate memory */
102  afSTFT_create(&(hSTFT), nCH, 1, hopSize, LDmode, hybridmode, AFSTFT_TIME_CH_BANDS);
103  FrameTF = (float_complex***)malloc3d(nTimeSlots, nCH, nBands, sizeof(float_complex));
104  tempFrameTD = (float**)calloc2d(nCH, nTimeSlots*hopSize, sizeof(float));
105 
106  /* perform TF transform */
107  for(ch=0; ch<nCH; ch++)
108  for(sample=0; sample<nSamplesTD; sample++)
109  tempFrameTD[ch][sample] = inTD[sample* nCH + ch];
110  afSTFT_forward(hSTFT, tempFrameTD, nTimeSlots*hopSize, FrameTF);
111 
112  /* save result to output */
113  for (band = 0; band < nBands; band++)
114  for (t = 0; t < nTimeSlots; t++)
115  for (ch = 0; ch < nCH; ch++)
116  outTF[band * nTimeSlots * nCH + t * nCH + ch] = FrameTF[t][ch][band];
117 
118  /* clean-up */
119  afSTFT_destroy(&hSTFT);
120  free(FrameTF);
121  free(tempFrameTD);
122 }

146 void afSTFT_create
147 (
148  void ** const phSTFT,
149  int nCHin,
150  int nCHout,
151  int hopsize,
152  int lowDelayMode,
153  int hybridmode,
154  AFSTFT_FDDATA_FORMAT format
155 )
156 {
157  *phSTFT = malloc1d(sizeof(afSTFT_data));
158  afSTFT_data *h = (afSTFT_data*)(*phSTFT);
159  int ch;
160 
161  if(hybridmode)
162  assert(hopsize==64 || hopsize==128 || hopsize==256);
163  assert(1024 % hopsize == 0 );
164 
165  h->nCHin = nCHin;
166  h->nCHout = nCHout;
167  h->hopsize = hopsize;
168  h->hybridmode = hybridmode;
169  h->nBands = hybridmode ? hopsize+5 : hopsize+1; /* hybrid mode incurs an additional 4 bands */
170  if(lowDelayMode)
171  h->afSTFTdelay = hybridmode ? 7*hopsize : 4*hopsize; /* hybrid mode incurs 3 additional hops of latency */
172  else
173  h->afSTFTdelay = hybridmode ? 12*hopsize : 9*hopsize; /* hybrid mode incurs 3 additional hops of latency */
174  h->format = format;
175 
176  /* init afSTFT core */
177  afSTFTlib_init(&(h->hInt), hopsize, nCHin, nCHout, lowDelayMode, hybridmode);
178 
179  /* temp buffers */
180  if(nCHout>0){
181  h->STFTOutputFrameTF = malloc1d(nCHout * sizeof(complexVector));
182  for(ch=0; ch < nCHout; ch++) {
183  h->STFTOutputFrameTF[ch].re = (float*)calloc1d(h->nBands, sizeof(float));
184  h->STFTOutputFrameTF[ch].im = (float*)calloc1d(h->nBands, sizeof(float));
185  }
186  }
187  else
188  h->STFTOutputFrameTF = NULL;
189  if(nCHout > 0 || nCHin > 0)
190  h->tempHopFrameTD = (float**)malloc2d( MAX(nCHin, nCHout), hopsize, sizeof(float));
191  if(nCHin>0){
192  h->STFTInputFrameTF = malloc1d(nCHin * sizeof(complexVector));
193  for(ch=0; ch < nCHin; ch++) {
194  h->STFTInputFrameTF[ch].re = (float*)calloc1d(h->nBands, sizeof(float));
195  h->STFTInputFrameTF[ch].im = (float*)calloc1d(h->nBands, sizeof(float));
196  }
197  }
198  else
199  h->STFTInputFrameTF = NULL;
200 }

233 void afSTFT_forward
234 (
235  void * const hSTFT,
236  float** dataTD,
237  int framesize,
238  float_complex*** dataFD
239 )
240 {
241  afSTFT_data *h = (afSTFT_data*)(hSTFT);
242  int ch, t, nHops, band;
243 
244  assert(framesize % h->hopsize == 0); /* framesize must be multiple of hopsize */
245  nHops = framesize/h->hopsize;
246 
247  /* Loop over hops */
248  for(t=0; t < nHops; t++) {
249  /* forward transform */
250  for(ch = 0; ch < h->nCHin; ch++)
251  utility_svvcopy(&(dataTD[ch][t*(h->hopsize)]), (h->hopsize), h->tempHopFrameTD[ch]);
252  afSTFTlib_forward(h->hInt, h->tempHopFrameTD, h->STFTInputFrameTF);
253 
254  /* store */
255  switch(h->format){
256  case AFSTFT_BANDS_CH_TIME:
257  for(band=0; band<h->nBands; band++)
258  for(ch=0; ch < h->nCHin; ch++)
259  dataFD[band][ch][t] = cmplxf(h->STFTInputFrameTF[ch].re[band], h->STFTInputFrameTF[ch].im[band]);
260  break;
261  case AFSTFT_TIME_CH_BANDS:
262  for(band=0; band<h->nBands; band++)
263  for(ch=0; ch < h->nCHin; ch++)
264  dataFD[t][ch][band] = cmplxf(h->STFTInputFrameTF[ch].re[band], h->STFTInputFrameTF[ch].im[band]);
265  break;
266  }
267  }
268 }

468 )
469 {

509 void afSTFT_FIRtoFilterbankCoeffs
510 (
511  float* hIR /*N_dirs x nCH x ir_len*/,
512  int N_dirs,
513  int nCH,
514  int ir_len,
515  int hopSize,
516  int LDmode,
517  int hybridmode,
518  float_complex* hFB /* nBands x nCH x N_dirs */
519 )
520 {
521  int i, j, t, nd, nm, nTimeSlots, ir_pad, nBands;
522  int* maxIdx;
523  float maxVal, idxDel, irFB_energy, irFB_gain, phase;
524  float* centerImpulse, *centerImpulseFB_energy, *ir;
525  float_complex cross;
526  float_complex* centerImpulseFB, *irFB;
527 
528  nBands = hopSize + (hybridmode ? 5 : 1);
529  ir_pad = 1024;//+512;
530  nTimeSlots = (MAX(ir_len,hopSize)+ir_pad)/hopSize;
531  maxIdx = calloc1d(nCH,sizeof(int));
532  centerImpulse = calloc1d(MAX(ir_len,hopSize)+ir_pad, sizeof(float));
533 
534  /* pick a direction to estimate the center of FIR delays */
535  for(j=0; j<nCH; j++){
536  maxVal = 2.23e-13f;
537  for(i=0; i<ir_len; i++){
538  if(hIR[j*ir_len + i] > maxVal){
539  maxVal = hIR[j*ir_len + i];
540  maxIdx[j] = i;
541  }
542  }
543  }
544  idxDel = 0.0f;
545  for(j=0; j<nCH; j++)
546  idxDel += (float)maxIdx[j];
547  idxDel /= (float)nCH;
548  idxDel = (idxDel + 1.5f);
549 
550  /* ideal impulse at mean delay */
551  centerImpulse[(int)idxDel] = 1.0f;
552 
553  /* analyse impulse with the filterbank */
554  centerImpulseFB = malloc1d(nBands*nTimeSlots*nCH*sizeof(float_complex));
555  afAnalyse(centerImpulse, MAX(ir_len,hopSize)+ir_pad, 1, hopSize, LDmode, hybridmode, centerImpulseFB);
556  centerImpulseFB_energy = calloc1d(nBands, sizeof(float));
557  for(i=0; i<nBands; i++)
558  for(t=0; t<nTimeSlots; t++)
559  centerImpulseFB_energy[i] += powf(cabsf(centerImpulseFB[i*nTimeSlots + t]), 2.0f);
560 
561  /* initialise FB coefficients */
562  ir = calloc1d( (MAX(ir_len,hopSize)+ir_pad) * nCH, sizeof(float));
563  irFB = malloc1d(nBands*nCH*nTimeSlots*sizeof(float_complex));
564  for(nd=0; nd<N_dirs; nd++){
565  for(j=0; j<ir_len; j++)
566  for(i=0; i<nCH; i++)
567  ir[j*nCH+i] = hIR[nd*nCH*ir_len + i*ir_len + j];
568  afAnalyse(ir, MAX(ir_len,hopSize)+ir_pad, nCH, hopSize, LDmode, hybridmode, irFB);
569  for(nm=0; nm<nCH; nm++){
570  for(i=0; i<nBands; i++){
571  irFB_energy = 0;
572  for(t=0; t<nTimeSlots; t++)
573  irFB_energy += powf(cabsf(irFB[i*nTimeSlots*nCH + t*nCH + nm]), 2.0f); /* out_nBands x nTimeslots x nCH */
574  irFB_gain = sqrtf(irFB_energy/MAX(centerImpulseFB_energy[i], 2.23e-8f));
575  cross = cmplxf(0.0f,0.0f);
576  for(t=0; t<nTimeSlots; t++)
577  cross = ccaddf(cross, ccmulf(irFB[i*nTimeSlots*nCH + t*nCH + nm], conjf(centerImpulseFB[i*nTimeSlots + t])));
578  phase = atan2f(cimagf(cross), crealf(cross));
579  hFB[i*nCH*N_dirs + nm*N_dirs + nd] = crmulf( cexpf(cmplxf(0.0f, phase)), irFB_gain);
580  }
581  }
582  }
583 
584  /* clean-up */
585  free(maxIdx);
586  free(centerImpulse);
587  free(centerImpulseFB_energy);
588  free(centerImpulseFB);
589  free(ir);
590  free(irFB);
591 }
