37  * https://cboard.cprogramming.com/c-programming/170381-cross-platform-wait-sleep.html */

39  /* For Windows (32- and 64-bit) */
40 # include <windows.h>
41 # define SAF_SLEEP(msecs) Sleep(msecs)
42 #elif defined(__unix) || defined(__APPLE__)
43  /* For linux, OSX, and other unixes */
44 # ifndef _POSIX_C_SOURCE
45 # define _POSIX_C_SOURCE 199309L /* or greater */
46 # endif
47 # include <time.h>
48 # define SAF_SLEEP(msecs) do { \
49  struct timespec ts; \
50  ts.tv_sec = msecs/1000; \
51  ts.tv_nsec = msecs%1000*1000; \
52  nanosleep(&ts, NULL); \
53 } while (0)

55 # error "Unknown system"
