68 void kiss_fftr(kiss_fftr_cfg st,const kiss_fft_scalar *timedata,kiss_fft_cpx *freqdata)
69 {
70  /* input buffer timedata is stored row-wise */
71  int k,ncfft;
72  kiss_fft_cpx fpnk,fpk,f1k,f2k,tw,tdc;
73 
74  if ( st->substate->inverse) {
75  fprintf(stderr,"kiss fft usage error: improper alloc\n");
76  exit(1);
77  }
78 
79  ncfft = st->substate->nfft;
80 
81  /*perform the parallel fft of two real signals packed in real,imag*/
82  kiss_fft( st->substate , (const kiss_fft_cpx*)timedata, st->tmpbuf );
83  /* The real part of the DC element of the frequency spectrum in st->tmpbuf
84  * contains the sum of the even-numbered elements of the input time sequence
85  * The imag part is the sum of the odd-numbered elements
86  *
87  * The sum of tdc.r and tdc.i is the sum of the input time sequence.
88  * yielding DC of input time sequence
89  * The difference of tdc.r - tdc.i is the sum of the input (dot product) [1,-1,1,-1...
90  * yielding Nyquist bin of input time sequence
91  */
92 
93  tdc.r = st->tmpbuf[0].r;
94  tdc.i = st->tmpbuf[0].i;
95  C_FIXDIV(tdc,2);
96  CHECK_OVERFLOW_OP(tdc.r ,+, tdc.i);
97  CHECK_OVERFLOW_OP(tdc.r ,-, tdc.i);
98  freqdata[0].r = tdc.r + tdc.i;
99  freqdata[ncfft].r = tdc.r - tdc.i;
100 #ifdef USE_SIMD
101  freqdata[ncfft].i = freqdata[0].i = _mm_set1_ps(0);
102 #else
103  freqdata[ncfft].i = freqdata[0].i = 0;
104 #endif
105 
106  for ( k=1;k <= ncfft/2 ; ++k ) {
107  fpk = st->tmpbuf[k];
108  fpnk.r = st->tmpbuf[ncfft-k].r;
109  fpnk.i = - st->tmpbuf[ncfft-k].i;
110  C_FIXDIV(fpk,2);
111  C_FIXDIV(fpnk,2);
112 
113  C_ADD( f1k, fpk , fpnk );
114  C_SUB( f2k, fpk , fpnk );
115  C_MUL( tw , f2k , st->super_twiddles[k-1]);
116 
117  freqdata[k].r = HALF_OF(f1k.r + tw.r);
118  freqdata[k].i = HALF_OF(f1k.i + tw.i);
119  freqdata[ncfft-k].r = HALF_OF(f1k.r - tw.r);
120  freqdata[ncfft-k].i = HALF_OF(tw.i - f1k.i);
121  }
122 }

124 void kiss_fftri(kiss_fftr_cfg st,const kiss_fft_cpx *freqdata,kiss_fft_scalar *timedata)
125 {
126  /* input buffer timedata is stored row-wise */
127  int k, ncfft;
128 
129  if (st->substate->inverse == 0) {
130  fprintf (stderr, "kiss fft usage error: improper alloc\n");
131  exit (1);
132  }
133 
134  ncfft = st->substate->nfft;
135 
136  st->tmpbuf[0].r = freqdata[0].r + freqdata[ncfft].r;
137  st->tmpbuf[0].i = freqdata[0].r - freqdata[ncfft].r;
138  C_FIXDIV(st->tmpbuf[0],2);
139 
140  for (k = 1; k <= ncfft / 2; ++k) {
141  kiss_fft_cpx fk, fnkc, fek, fok, tmp;
142  fk = freqdata[k];
143  fnkc.r = freqdata[ncfft - k].r;
144  fnkc.i = -freqdata[ncfft - k].i;
145  C_FIXDIV( fk , 2 );
146  C_FIXDIV( fnkc , 2 );
147 
148  C_ADD (fek, fk, fnkc);
149  C_SUB (tmp, fk, fnkc);
150  C_MUL (fok, tmp, st->super_twiddles[k-1]);
151  C_ADD (st->tmpbuf[k], fek, fok);
152  C_SUB (st->tmpbuf[ncfft - k], fek, fok);
153 #ifdef USE_SIMD
154  st->tmpbuf[ncfft - k].i *= _mm_set1_ps(-1.0);
155 #else
156  st->tmpbuf[ncfft - k].i *= -1;
157 #endif
158  }
159  kiss_fft (st->substate, st->tmpbuf, (kiss_fft_cpx *) timedata);
