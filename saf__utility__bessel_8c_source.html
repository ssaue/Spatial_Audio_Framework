58 static int MSTA1
59 (
60  double X,
61  int MP
62 )
63 {
64  double A0,F,F0,F1;
65  int IT,NN,N0,N1;
66 
67  NN = 0;
68  A0=fabs(X);
69  N0=(int)(floor(1.1*A0)+1.0);
70  F0=ENVJ(N0,A0)-MP;
71  N1=N0+5;
72  F1=ENVJ(N1,A0)-MP;
73  for (IT=1; IT<=20; IT++) {
74  NN=N1-(N1-N0)/(1.0-F0/F1);
75  F=ENVJ(NN,A0)-MP;
76  if (abs(NN-N1) < 1) goto e20;
77  N0=N1;
78  F0=F1;
79  N1=NN;
80  F1=F;
81  }
82 e20: return NN;
83 }

91 static int MSTA2
92 (
93  double X,
94  int N,
95  int MP
96 )
97 {
98  double A0,EJN,F,F0,F1,HMP,OBJ;
99  int IT,N0,N1,NN;
100 
101  NN = 0;
102  A0=fabs(X);
103  HMP=0.5*MP;
104  EJN=ENVJ(N,A0);
105  if (EJN <= HMP) {
106  OBJ=MP;
107  N0=(int)floor(1.1*A0);
108  }
109  else {
110  OBJ=HMP+EJN;
111  N0=N;
112  }
113  F0=ENVJ(N0,A0)-OBJ;
114  N1=N0+5;
115  F1=ENVJ(N1,A0)-OBJ;
116  for (IT=1; IT<=20; IT++) {
117  NN=N1-(N1-N0)/(1.0-F0/F1);
118  F=ENVJ(NN,A0)-OBJ;
119  if (abs(NN-N1) < 1) goto e20;
120  N0=N1;
121  F0=F1;
122  N1=NN;
123  F1=F;
124  }
125 e20: return NN+10;
126 }

139 static void SPHI
140 (
141  int N,
142  double X,
143  int *NM,
144  double *SI,
145  double *DI
146 )
147 {
148  int K, M, i;
149  double CS, F, F0, F1, SI0;
150 
151  *NM=N;
152  if (fabs(X) < 1e-20) {
153  for (K=0; K<=N; K++) {
154  SI[K]=0.0;
155  DI[K]=0.0;
156  }
157  SI[0]=1.0;
158  DI[1]=0.333333333333333;
159  return;
160  }
161  SI[0]=sinh(X)/X;
162  SI[1]=-(sinh(X)/X-cosh(X))/X;
163  SI0=SI[0];
164  if (N >= 2) {
165  M=MSTA1(X,200);
166  if (M < N)
167  *NM=M;
168  else
169  M=MSTA2(X,N,15);
170  /* I had to add this while loop to avoid NaNs and sacrifice some precision, but only when needed */
171  i=0;
172  while (M < 0) {
173  M=MSTA2(X,N,14-i);
174  i++;
175  if(i==14)
176  M=0;
177  }
178  F0=0.0;
179  F1=1.0-100;
180  F=1;
181  for (K=M; K>-1; K--) {
182  F=(2.0*K+3.0)*F1/X+F0;
183  if (K <= *NM) SI[K]=F;
184  F0=F1;
185  F1=F;
186  }
187  CS=SI0/F;
188  for (K=0; K<=*NM; K++) SI[K] *= CS;
189  }
190  DI[0]=SI[1];
191  for (K=1; K<=*NM; K++)
192  DI[K]=SI[K-1]-(K+1.0)/X*SI[K];
193 }

202 static void SPHK
203 (
204  int N,
205  double X,
206  int *NM,
207  double *SK,
208  double *DK
209 )
210 {
211  int K;
212  double F, F0, F1;
213 
214  *NM=N;
215  if (X < 1e-20) {
216  for (K=0; K<=N; K++) {
217  SK[K]=1.0e+300;
218  DK[K]=-1.0e+300;
219  }
220  return;
221  }
222  SK[0]=0.5*SAF_PId/X*exp(-X);
223  SK[1]=SK[0]*(1.0+1.0/X);
224  F0=SK[0];
225  F1=SK[1];
226  for (K=2; K<=N; K++) {
227  F=(2.0*K-1.0)*F1/X+F0;
228  SK[K]=F;
229  if (fabs(F) > 1.0e+300) goto e20;
230  F0=F1;
231  F1=F;
232  }
233 e20: *NM=K-1;
234  DK[0]=-SK[1];
235  for (K=1; K<=*NM; K++)
236  DK[K]=-SK[K-1]-(K+1.0)/X*SK[K];
237 }

250 static void SPHJ
251 (
252  int N,
253  double X,
254  int *NM,
255  double *SJ,
256  double *DJ
257 )
258 {
259  int K, M, i;
260  double CS, F, F0, F1, SA, SB;
261 
262  *NM=N;
263  if (fabs(X) < 1e-80) {
264  for (K=0; K<=N; K++) {
265  SJ[K]=0.0;
266  DJ[K]=0.0;
267  }
268  SJ[0]=1.0;
269  DJ[1]=0.333333333333333;
270  return;
271  }
272  SJ[0]=sin(X)/X;
273  SJ[1]=(SJ[0]-cos(X))/X;
274  if (N >= 2) {
275  SA=SJ[0];
276  SB=SJ[1];
277  M=MSTA1(X,200);
278  if (M < N)
279  *NM=M;
280  else
281  M=MSTA2(X,N,15);
282  /* I had to add this while loop to avoid NaNs and sacrifice some precision, but only when needed */
283  i=0;
284  while (M < 0) {
285  M=MSTA2(X,N,14-i);
286  i++;
287  if(i==14)
288  M=0;
289  }
290  F0=0.0;
291  F1=1.0-100;
292  F=1;
293  CS=1;
294  for (K=M; K>-1; K--) {
295  F=(2.0*K+3.0)*F1/X-F0;
296  if (K <= *NM) SJ[K]=F;
297  F0=F1;
298  F1=F;
299  }
300  if (fabs(SA) > fabs(SB)) CS=SA/F;
301  if (fabs(SA) <= fabs(SB)) CS=SB/F0;
302  for (K=0; K<=*NM; K++) SJ[K] *= CS;
303  }
304  DJ[0]=(cos(X)-sin(X)/X)/X;
305  for (K=1; K<=*NM; K++)
306  DJ[K]=SJ[K-1]-(K+1.0)*SJ[K]/X;
307 }

315 static void SPHY
316 (
317  int N,
318  double X,
319  int *NM,
320  double *SY,
321  double *DY
322 )
323 {
324  int K;
325  double F, F0, F1;
326 
327  *NM=N;
328  if (X < 1e-20) {
329  for (K=0; K<=N; K++) {
330  SY[K]=-1.0e+300;
331  DY[K]=1e+300;
332  }
333  return;
334  }
335  SY[0]=-cos(X)/X;
336  SY[1]=(SY[0]-sin(X))/X;
337  F0=SY[0];
338  F1=SY[1];
339  for (K=2; K<=N; K++) {
340  F=(2.0*K-1.0)*F1/X-F0;
341  SY[K]=F;
342  if (fabs(F) >= 1e+300) goto e20;
343  F0=F1;
344  F1=F;
345  }
346 e20: *NM=K-1;
347  DY[0]=(sin(X)+cos(X)/X)/X;
348  for (K=1; K<=*NM; K++)
349  DY[K]=SY[K-1]-(K+1.0)*SY[K]/X;
350 }
