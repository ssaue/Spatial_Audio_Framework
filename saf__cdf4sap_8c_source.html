257 void formulate_M_and_Cr
258 (
259  void * const hCdf,
260  float* Cx,
261  float* Cy,
262  float* Q,
263  int useEnergyFLAG,
264  float reg,
265  float* M,
266  float* Cr
267 )
268 {
269  cdf4sap_data *h = (cdf4sap_data*)(hCdf);
270  int i, j, nXcols, nYcols;
271 
272  nXcols = h->nXcols;
273  nYcols = h->nYcols;
274 
275  memset(h->lambda, 0, nYcols * nXcols * sizeof(float));
276  for(i = 0; i<MIN(nXcols,nYcols); i++)
277  h->lambda[i*nXcols + i] = 1.0f;
278 
279  /* Decomposition of Cy */
280  utility_ssvd(Cy, nYcols, nYcols, h->U_Cy, h->S_Cy, NULL, NULL);
281  for(i=0; i< nYcols; i++)
282  h->S_Cy[i*nYcols+i] = sqrtf(MAX(h->S_Cy[i*nYcols+i], 2.23e-20f));
283  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nYcols, nYcols, nYcols, 1.0f,
284  h->U_Cy, nYcols,
285  h->S_Cy, nYcols, 0.0f,
286  h->Ky, nYcols);
287 
288  /* Decomposition of Cx */
289  utility_ssvd(Cx, nXcols, nXcols, h->U_Cx, h->S_Cx, NULL, h->s_Cx);
290  for(i=0; i< nXcols; i++){
291  h->S_Cx[i*nXcols+i] = sqrtf(MAX(h->S_Cx[i*nXcols+i], 2.23e-20f));
292  h->s_Cx[i] = sqrtf(MAX(h->s_Cx[i], 2.23e-20f));
293  }
294  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nXcols, nXcols, nXcols, 1.0f,
295  h->U_Cx, nXcols,
296  h->S_Cx, nXcols, 0.0f,
297  h->Kx, nXcols);
298 
299  /* Regularisation of S_Cx */
300  int ind;
301  float limit, maxVal;
302  utility_simaxv(h->s_Cx, nXcols, &ind);
303  limit = h->s_Cx[ind] * reg + 2.23e-13f;
304  for(i=0; i < nXcols; i++)
305  h->S_Cx[i*nXcols+i] = 1.0f / MAX(h->S_Cx[i*nXcols+i], limit);
306 
307  /* Formulate regularised Kx^-1 */
308  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasTrans, nXcols, nXcols, nXcols, 1.0f,
309  h->S_Cx, nXcols,
310  h->U_Cx, nXcols, 0.0f,
311  h->Kx_reg_inverse, nXcols);
312 
313  /* Formulate normalisation matrix G_hat */
314  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasTrans, nXcols, nYcols, nXcols, 1.0f,
315  Cx, nXcols,
316  Q, nXcols, 0.0f,
317  h->Cx_QH, nYcols);
318  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nYcols, nYcols, nXcols, 1.0f,
319  Q, nXcols,
320  h->Cx_QH, nYcols, 0.0f,
321  h->G_hat, nYcols);
322  maxVal = -2.23e13f;
323  for(i=0; i< nYcols; i++)
324  maxVal = h->G_hat[i*nYcols+i] > maxVal ? h->G_hat[i*nYcols+i] : maxVal;
325  limit = maxVal * 0.001f + 2.23e-13f;
326  for(i=0; i < nYcols; i++)
327  for(j=0; j < nYcols; j++)
328  h->G_hat[i*nYcols+j] = i==j ? sqrtf(MAX(Cy[i*nYcols+j], 2.23e-13f) / MAX(h->G_hat[i*nYcols+j], limit)) : 0.0f;
329 
330  /* Formulate optimal P */
331  cblas_sgemm(CblasRowMajor, CblasTrans, CblasNoTrans, nYcols, nYcols, nYcols, 1.0f,
332  h->G_hat, nYcols,
333  h->Ky, nYcols, 0.0f,
334  h->GhatH_Ky, nYcols);
335  cblas_sgemm(CblasRowMajor, CblasTrans, CblasNoTrans, nXcols, nYcols, nYcols, 1.0f,
336  Q, nXcols,
337  h->GhatH_Ky, nYcols, 0.0f,
338  h->QH_GhatH_Ky, nYcols);
339  cblas_sgemm(CblasRowMajor, CblasTrans, CblasNoTrans, nXcols, nYcols, nXcols, 1.0f,
340  h->Kx, nXcols,
341  h->QH_GhatH_Ky, nYcols, 0.0f,
342  h->KxH_QH_GhatH_Ky, nYcols);
343  utility_ssvd(h->KxH_QH_GhatH_Ky, nXcols, nYcols, h->U, NULL, h->V, NULL);
344  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasTrans, nYcols, nXcols, nXcols, 1.0f,
345  h->lambda, nXcols,
346  h->U, nXcols, 0.0f,
347  h->lambda_UH, nXcols);
348  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nYcols, nXcols, nYcols, 1.0f,
349  h->V, nYcols,
350  h->lambda_UH, nXcols, 0.0f,
351  h->P, nXcols);
352 
353  /* Formulate M */
354  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nYcols, nXcols, nXcols, 1.0f,
355  h->P, nXcols,
356  h->Kx_reg_inverse, nXcols, 0.0f,
357  h->P_Kxreginverse, nXcols);
358  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nYcols, nXcols, nYcols, 1.0f,
359  h->Ky, nYcols,
360  h->P_Kxreginverse, nXcols, 0.0f,
361  M, nXcols);
362 
363  /* Formulate residual covariance matrix */
364  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasTrans, nXcols, nYcols, nXcols, 1.0f,
365  Cx, nXcols,
366  M, nXcols, 0.0f,
367  h->Cx_MH, nYcols);
368  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nYcols, nYcols, nXcols, 1.0f,
369  M, nXcols,
370  h->Cx_MH, nYcols, 0.0f,
371  h->Cy_tilde, nYcols);
372  if(Cr != NULL)
373  for(i=0; i < nYcols*nYcols; i++)
374  Cr[i] = Cy[i] - h->Cy_tilde[i];
375 
376  /* Use energy compensation instead of residuals */
377  if(useEnergyFLAG){
378  for(i=0; i < nYcols; i++)
379  for(j=0; j < nYcols; j++)
380  h->G_hat[i*nYcols+j] = i==j ? sqrtf( MAX(Cy[i*nYcols+j], 2.23e-20f) / (h->Cy_tilde[i*nYcols+j]+2.23e-7f)) : 0.0f;
381  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nYcols, nXcols, nYcols, 1.0f,
382  h->G_hat, nYcols,
383  M, nXcols, 0.0f,
384  h->G_M, nXcols);
385  memcpy(M, h->G_M, nYcols*nXcols*sizeof(float));
386  if(Cr != NULL)
387  memset(Cr, 0, nYcols*nYcols*sizeof(float));
388  }
389 }

391 void formulate_M_and_Cr_cmplx
392 (
393  void * const hCdf,
394  float_complex* Cx,
395  float_complex* Cy,
396  float_complex* Q,
397  int useEnergyFLAG,
398  float reg,
399  float_complex* M,
400  float_complex* Cr
401 )
402 {
403  cdf4sap_cmplx_data *h = (cdf4sap_cmplx_data*)(hCdf);
404  int i, j, nXcols, nYcols;
405  const float_complex calpha = cmplxf(1.0f, 0.0f); const float_complex cbeta = cmplxf(0.0f, 0.0f);
406 
407  nXcols = h->nXcols;
408  nYcols = h->nYcols;
409 
410  memset(h->lambda, 0, nYcols * nXcols * sizeof(float_complex));
411  for(i = 0; i<MIN(nXcols,nYcols); i++)
412  h->lambda[i*nXcols + i] = cmplxf(1.0f, 0.0f);
413 
414  /* Decomposition of Cy */
415  utility_csvd(Cy, nYcols, nYcols, h->U_Cy, h->S_Cy, NULL, NULL);
416  for(i=0; i< nYcols; i++)
417  h->S_Cy[i*nYcols+i] = cmplxf(sqrtf(MAX(crealf(h->S_Cy[i*nYcols+i]), 2.23e-20f)), 0.0f);
418  cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nYcols, nYcols, nYcols, &calpha,
419  h->U_Cy, nYcols,
420  h->S_Cy, nYcols, &cbeta,
421  h->Ky, nYcols);
422 
423  /* Decomposition of Cx */
424  utility_csvd(Cx, nXcols, nXcols, h->U_Cx, h->S_Cx, NULL, h->s_Cx);
425  for(i=0; i< nXcols; i++){
426  h->s_Cx[i] = sqrtf(MAX(h->s_Cx[i], 2.23e-13f));
427  h->S_Cx[i*nXcols+i] = cmplxf(h->s_Cx[i], 0.0f);
428  }
429  cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nXcols, nXcols, nXcols, &calpha,
430  h->U_Cx, nXcols,
431  h->S_Cx, nXcols, &cbeta,
432  h->Kx, nXcols);
433 
434  /* Regularisation of S_Cx */
435  int ind;
436  float limit, maxVal;
437  //utility_simaxv(h->s_Cx, nXcols, &ind);
438  ind = 0; /* utility_csvd returns the singular values in decending order */
439  limit = h->s_Cx[ind] * reg + 2.23e-13f;
440  for(i=0; i < nXcols; i++)
441  h->S_Cx[i*nXcols+i] = cmplxf(1.0f / MAX(h->s_Cx[i], limit), 0.0f);
442 
443  /* Formulate regularised Kx^-1 */
444  cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasConjTrans, nXcols, nXcols, nXcols, &calpha,
445  h->S_Cx, nXcols,
446  h->U_Cx, nXcols, &cbeta,
447  h->Kx_reg_inverse, nXcols);
448 
449  /* Formulate normalisation matrix G_hat */
450  cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasConjTrans, nXcols, nYcols, nXcols, &calpha,
451  Cx, nXcols,
452  Q, nXcols, &cbeta,
453  h->Cx_QH, nYcols);
454  cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nYcols, nYcols, nXcols, &calpha,
455  Q, nXcols,
456  h->Cx_QH, nYcols, &cbeta,
457  h->G_hat, nYcols);
458  /* h->G_hat: imaginary parts along the diagonal are ~0, so it's OK to take
459  * the real instead below: */
460  maxVal = -2.23e13f;
461  for(i=0; i< nYcols; i++)
462  maxVal = cabsf(h->G_hat[i*nYcols+i]) > maxVal ? cabsf(h->G_hat[i*nYcols+i]) : maxVal; // crealf->cabsf
463  limit = maxVal * 0.001f + 2.23e-13f;
464  for(i=0; i < nYcols; i++)
465  for(j=0; j < nYcols; j++)
466  h->G_hat[i*nYcols+j] = i==j ? cmplxf(crealf(csqrtf( ccdivf(Cy[i*nYcols+j], cmplxf(MAX(cabsf(h->G_hat[i*nYcols+j]), limit), 0.0f)))), 0.0f) : cmplxf(0.0f, 0.0f); // changed crealf->cabsf
467 
468  /* Formulate optimal P */
469  cblas_cgemm(CblasRowMajor, CblasConjTrans, CblasNoTrans, nYcols, nYcols, nYcols, &calpha,
470  h->G_hat, nYcols,
471  h->Ky, nYcols, &cbeta,
472  h->GhatH_Ky, nYcols);
473  cblas_cgemm(CblasRowMajor, CblasConjTrans, CblasNoTrans, nXcols, nYcols, nYcols, &calpha,
474  Q, nXcols,
475  h->GhatH_Ky, nYcols, &cbeta,
476  h->QH_GhatH_Ky, nYcols);
477  cblas_cgemm(CblasRowMajor, CblasConjTrans, CblasNoTrans, nXcols, nYcols, nXcols, &calpha,
478  h->Kx, nXcols,
479  h->QH_GhatH_Ky, nYcols, &cbeta,
480  h->KxH_QH_GhatH_Ky, nYcols);
481  utility_csvd(h->KxH_QH_GhatH_Ky, nXcols, nYcols, h->U, NULL, h->V, NULL);
482  cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasConjTrans, nYcols, nXcols, nXcols, &calpha,
483  h->lambda, nXcols,
484  h->U, nXcols, &cbeta,
485  h->lambda_UH, nXcols);
486  cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nYcols, nXcols, nYcols, &calpha,
487  h->V, nYcols,
488  h->lambda_UH, nXcols, &cbeta,
489  h->P, nXcols);
490 
491  /* Formulate M */
492  cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nYcols, nXcols, nXcols, &calpha,
493  h->P, nXcols,
