35 void ims_shoebox_create
36 (
37  void** phIms,
38  float roomDimensions[3],
39  float* abs_wall,
40  float lowestOctaveBand,
41  int nOctBands,
42  float c_ms,
43  float fs
44 )
45 {
46  *phIms = malloc1d(sizeof(ims_scene_data));
47  ims_scene_data *sc = (ims_scene_data*)(*phIms);
48  int i,j,band,wall;
49 
50  /* Shoebox dimensions */
51  sc->room_dims[0] = roomDimensions[0];
52  sc->room_dims[1] = roomDimensions[1];
53  sc->room_dims[2] = roomDimensions[2];
54  sc->c_ms = c_ms;
55 
56  /* Octave band centre frequencies */
57  if(nOctBands>1){
58  sc->nBands = nOctBands;
59  sc->band_centerfreqs = malloc1d(nOctBands*sizeof(float));
60  sc->band_centerfreqs[0] = lowestOctaveBand;
61  for(band=1; band<nOctBands; band++)
62  sc->band_centerfreqs[band] = sc->band_centerfreqs[band-1]*2.0f;
63  sc->band_cutofffreqs = malloc1d((sc->nBands-1)*sizeof(float));
64  getOctaveBandCutoffFreqs(sc->band_centerfreqs, sc->nBands, sc->band_cutofffreqs);
65  }
66  else { /* Broad-band operation */
67  sc->nBands = 1;
68  sc->band_centerfreqs = NULL;
69  sc->band_cutofffreqs = NULL;
70  }
71 
72  /* Samplerate */
73  sc->fs = fs;
74 
75  /* Absorption coeffients per wall and octave band */
76  sc->abs_wall = (float**)malloc2d(sc->nBands, IMS_NUM_WALLS_SHOEBOX, sizeof(float));
77  for(band=0; band<sc->nBands; band++)
78  for(wall=0; wall<IMS_NUM_WALLS_SHOEBOX; wall++)
79  sc->abs_wall[band][wall] = abs_wall[band*IMS_NUM_WALLS_SHOEBOX+wall];
80 
81  /* Default are no sources or receivers in the room */
82  for(i=0; i<IMS_MAX_NUM_SOURCES; i++)
83  sc->srcs[i].ID = IMS_UNASSIGNED; /* -1 indicates not in use */
84  for(i=0; i<IMS_MAX_NUM_RECEIVERS; i++)
85  sc->recs[i].ID = IMS_UNASSIGNED;
86  sc->nSources = 0;
87  sc->nReceivers = 0;
88 
89  /* ims_core_workspace per source / receiver combination */
90  sc->hCoreWrkSpc = (voidPtr**)malloc2d(IMS_MAX_NUM_RECEIVERS, IMS_MAX_NUM_SOURCES, sizeof(voidPtr));
91  for(i=0; i<IMS_MAX_NUM_RECEIVERS; i++)
92  for(j=0; j<IMS_MAX_NUM_SOURCES; j++)
93  sc->hCoreWrkSpc[i][j] = NULL;
94 
95  /* FIR Fiterbank */
96  sc->H_filt = NULL;
97 
98  /* RIRs per source / receiver combination */
99  sc->rirs = (ims_rir**)malloc2d(IMS_MAX_NUM_RECEIVERS, IMS_MAX_NUM_SOURCES, sizeof(ims_rir));
100  for(i=0; i<IMS_MAX_NUM_RECEIVERS; i++){
101  for(j=0; j<IMS_MAX_NUM_SOURCES; j++){
102  sc->rirs[i][j].data = NULL;
103  sc->rirs[i][j].length = sc->rirs[i][j].nChannels = 0;
104  }
105  }
106 
107  /* Circular buffers (only used/allocated when applyEchogramTD() function is called for the first time) */
108  memset(sc->wIdx, 0, IMS_MAX_NUM_RECEIVERS*IMS_MAX_NUM_SOURCES*2*sizeof(unsigned long));
109  sc->circ_buffer[0] = NULL;
110  sc->circ_buffer[1] = NULL;
111 
112  /* IIR Filterbank per source (only used/allocated when applyEchogramTD() function is called for the first time) */
113  sc->hFaFbank = malloc1d(IMS_MAX_NUM_SOURCES*sizeof(voidPtr));
114  sc->src_sigs_bands = malloc1d(IMS_MAX_NUM_SOURCES*sizeof(float**));
115  for(j=0; j<IMS_MAX_NUM_SOURCES; j++){
116  sc->hFaFbank[j] = NULL;
117  sc->src_sigs_bands[j] = NULL;
118  }
119 
120  /* Temp buffers for cross-fading (only used/allocated when applyEchogramTD() function is called for the first time) */
121  sc->rec_sig_tmp[IMS_EG_CURRENT] = malloc1d(IMS_MAX_NUM_RECEIVERS*sizeof(float**));
122  sc->rec_sig_tmp[IMS_EG_PREV] = malloc1d(IMS_MAX_NUM_RECEIVERS*sizeof(float**));
123  for(j=0; j<IMS_MAX_NUM_RECEIVERS; j++){
124  sc->rec_sig_tmp[IMS_EG_CURRENT][j] = NULL;
125  sc->rec_sig_tmp[IMS_EG_PREV][j] = NULL;
126  }
127  memset(sc->applyCrossFadeFLAG, 0, IMS_MAX_NUM_RECEIVERS*IMS_MAX_NUM_SOURCES*sizeof(int));
128  sc->interpolator_fIn = NULL;
129  sc->interpolator_fOut = NULL;
130  sc->framesize = -1;
131 
132  /* Lagrange interpolator look-up table */
133  for(i=0; i<IMS_LAGRANGE_LOOKUP_TABLE_SIZE; i++)
134  sc->lookup_fractions[i] = 1.0f/IMS_LAGRANGE_LOOKUP_TABLE_SIZE * (float)i;
135  lagrangeWeights(IMS_LAGRANGE_ORDER, sc->lookup_fractions, IMS_LAGRANGE_LOOKUP_TABLE_SIZE, (float*)sc->lookup_H_frac);
136 }

138 void ims_shoebox_destroy
139 (
140  void** phIms
141 )
142 {
143  ims_scene_data *sc = (ims_scene_data*)(*phIms);
144  int i,j;
145 
146  if(sc!=NULL){
147  free(sc->band_centerfreqs);
148  free(sc->band_cutofffreqs);
149  free(sc->abs_wall);
150  for(i=0; i<IMS_MAX_NUM_RECEIVERS; i++)
151  for(j=0; j<IMS_MAX_NUM_SOURCES; j++)
152  ims_shoebox_coreWorkspaceDestroy(&(sc->hCoreWrkSpc[i][j]));
153  free(sc->hCoreWrkSpc);
154  free(sc->H_filt);
155  for(i=0; i<IMS_MAX_NUM_RECEIVERS; i++)
156  for(j=0; j<IMS_MAX_NUM_SOURCES; j++)
157  free(sc->rirs[i][j].data);
158  free(sc->rirs);
159  free(sc->circ_buffer[IMS_EG_CURRENT]);
160  free(sc->circ_buffer[IMS_EG_PREV]);
161  for(j=0; j<IMS_MAX_NUM_SOURCES; j++){
162  faf_IIRFilterbank_destroy(&(sc->hFaFbank[j]));
163  free(sc->src_sigs_bands[j]);
164  }
165  free(sc->hFaFbank);
166  free(sc->src_sigs_bands);
167  for(i=0; i<IMS_MAX_NUM_RECEIVERS; i++){
168  free(sc->rec_sig_tmp[IMS_EG_CURRENT][i]);
169  free(sc->rec_sig_tmp[IMS_EG_PREV][i]);
170  }
171  free(sc->rec_sig_tmp[IMS_EG_CURRENT]);
172  free(sc->rec_sig_tmp[IMS_EG_PREV]);
173  free(sc->interpolator_fIn);
174  free(sc->interpolator_fOut);
175  free(sc);
176  sc=NULL;
177  *phIms = NULL;
178  }

181 void ims_shoebox_computeEchograms
182 (
183  void* hIms,
184  int maxN,
185  float maxTime_ms
186 )
187 {
188  ims_scene_data *sc = (ims_scene_data*)(hIms);
189  ims_core_workspace* workspace;
190  ims_pos_xyz src2, rec2;
191  int src_idx, rec_idx, band;
192 
193  assert(maxN<0 || maxTime_ms<0.0f); /* one must be more than 0, and one less */
194 
195  /* Compute echograms for active source/receiver combinations */
196  for(rec_idx = 0; rec_idx < IMS_MAX_NUM_RECEIVERS; rec_idx++){
197  for(src_idx = 0; src_idx < IMS_MAX_NUM_SOURCES; src_idx++){
198  if( (sc->srcs[src_idx].ID != IMS_UNASSIGNED) && (sc->recs[rec_idx].ID != IMS_UNASSIGNED) ){
199  /* Change y coord for Receiver and Source to match convention
200  * used inside the coreInit function */
201  rec2.x = sc->recs[rec_idx].pos.x;
202  rec2.y = sc->room_dims[1] - sc->recs[rec_idx].pos.y;
203  rec2.z = sc->recs[rec_idx].pos.z;
204  src2.x = sc->srcs[src_idx].pos.x;
205  src2.y = sc->room_dims[1] - sc->srcs[src_idx].pos.y;
206  src2.z = sc->srcs[src_idx].pos.z;
207 
208  /* Workspace handle for this source/receiver combination */
209  workspace = sc->hCoreWrkSpc[rec_idx][src_idx];
210 
211  /* Copy previous echograms */
212  for(band=0; band<workspace->nBands; band++)
213  ims_shoebox_echogramCopy(workspace->hEchogram_abs[band], workspace->hPrevEchogram_abs[band]);
214 
215  /* Force refresh if target RIR length or max reflection order has changed */
216  if(maxTime_ms>0.0f){
217  if(workspace->d_max != maxTime_ms)
218  workspace->refreshEchogramFLAG = 1;
219  }
220  else{
221  if(workspace->N_max != maxN)
222  workspace->refreshEchogramFLAG = 1;
223  }
224 
225  /* Only update if it is required */
226  if(workspace->refreshEchogramFLAG){
227  /* Compute echogram due to pure propagation (frequency-independent, omni-directional) */
228  if(maxTime_ms>0.0f)
229  ims_shoebox_coreInitT(workspace, sc->room_dims, src2, rec2, maxTime_ms, sc->c_ms);
230  else
231  ims_shoebox_coreInitN(workspace, sc->room_dims, src2, rec2, maxN, sc->c_ms);
232 
233  /* Apply receiver directivities */
234  switch(sc->recs[rec_idx].type){
235  case RECEIVER_SH:
236  ims_shoebox_coreRecModuleSH(workspace, NSH2ORDER(sc->recs[rec_idx].nChannels));
237  break;
238  }
239 
240  /* Apply boundary absorption per frequency band */
241  ims_shoebox_coreAbsorptionModule(workspace, sc->abs_wall);
242 
243  /* Indicate that the echogram is now up to date, and that the RIR should be updated */
244  workspace->refreshEchogramFLAG = 0;
245  workspace->refreshRIRFLAG = 1;
246 
247  /* Also indicate that applyTD() should cross-fade the next frame to void clicks */
248  sc->applyCrossFadeFLAG[rec_idx][src_idx] = 1;
249  }
250  }
251  }
252  }
253 }

255 void ims_shoebox_renderRIRs
256 (
257  void* hIms,
258  int fractionalDelayFLAG
259 )
260 {
261  ims_scene_data *sc = (ims_scene_data*)(hIms);
262  ims_core_workspace* wrk;
263  int src_idx, rec_idx;
264 
265  /* Compute FIR Filterbank coefficients (if this is the first time this
266  * function is being called) */
267  if(sc->H_filt==NULL){
268  sc->H_filt = (float**)realloc2d((void**)sc->H_filt, sc->nBands, (IMS_FIR_FILTERBANK_ORDER+1), sizeof(float));
269  FIRFilterbank(IMS_FIR_FILTERBANK_ORDER, sc->band_cutofffreqs, sc->nBands-1,
270  sc->fs, WINDOWING_FUNCTION_HAMMING, 1, FLATTEN2D(sc->H_filt));
271  }
272 
273  /* Render RIRs for all active source/receiver combinations */
274  for(rec_idx = 0; rec_idx < IMS_MAX_NUM_RECEIVERS; rec_idx++){
275  for(src_idx = 0; src_idx < IMS_MAX_NUM_SOURCES; src_idx++){
276  if( (sc->srcs[src_idx].ID!=IMS_UNASSIGNED) && (sc->recs[rec_idx].ID!=IMS_UNASSIGNED) ){
277 
278  /* Workspace handle for this source/receiver combination */
279  wrk = sc->hCoreWrkSpc[rec_idx][src_idx];
280 
281  /* Only update if it is required */
282  if(wrk->refreshRIRFLAG){
283  /* Render the RIRs for each band */
284  ims_shoebox_renderRIR(wrk, fractionalDelayFLAG, sc->fs, sc->H_filt, &(sc->rirs[rec_idx][src_idx]));
285 
286  wrk->refreshRIRFLAG = 0;
287  }
288  }
289  }
290  }

293 void ims_shoebox_applyEchogramTD
294 (
295  void* hIms,
296  long receiverID,
297  int nSamples,
298  int fractionalDelaysFLAG
299 )
300 {
301  ims_scene_data *sc = (ims_scene_data*)(hIms);
302  ims_core_workspace* wrk;
303  echogram_data *echogram_abs, *echogram_abs_0;
304  int k, i, n, im, band, ch, rec_idx, src_idx, time_samples, wIdx_n;
305  unsigned long rIdx;
306 
307  assert(fractionalDelaysFLAG==0); /* Not tested yet... */
308  assert(nSamples <= IMS_MAX_NSAMPLES_PER_FRAME);
309 
310  /* Allocate circular buffers (if this is the first time this function is being called) */
311  if(sc->circ_buffer[0] == NULL)
312  sc->circ_buffer[0] = (float***)calloc3d(IMS_MAX_NUM_SOURCES, sc->nBands, IMS_CIRC_BUFFER_LENGTH, sizeof(float));
313  if(sc->circ_buffer[1] == NULL)
314  sc->circ_buffer[1] = (float***)calloc3d(IMS_MAX_NUM_SOURCES, sc->nBands, IMS_CIRC_BUFFER_LENGTH, sizeof(float));
315 
316  /* Also allocate signal buffers and filterbank handles (if this is the first time this function is being called) */
317  for(src_idx = 0; src_idx < IMS_MAX_NUM_SOURCES; src_idx++){
318  /* only allocate if source is active... */
319  if( (sc->srcs[src_idx].ID != IMS_UNASSIGNED) && (sc->src_sigs_bands[src_idx] == NULL) ){
320  if(sc->nBands>1){ /* ... and only create the filterbank if there is more than one band: */
321  faf_IIRFilterbank_create(&(sc->hFaFbank[src_idx]), IMS_IIR_FILTERBANK_ORDER, sc->band_cutofffreqs,
322  sc->nBands-1, sc->fs, IMS_MAX_NSAMPLES_PER_FRAME);
323  }
324  sc->src_sigs_bands[src_idx] = (float**)malloc2d(sc->nBands, IMS_MAX_NSAMPLES_PER_FRAME, sizeof(float));
325  }
326  }
327 
328  /* Find index corresponding to this receiver ID */
329  rec_idx = IMS_UNASSIGNED;
330  for(i=0; i<IMS_MAX_NUM_RECEIVERS; i++){
331  if(sc->recs[i].ID == receiverID){
332  rec_idx = i;
333  break;
334  }
335  }
336  assert(rec_idx != IMS_UNASSIGNED);
337 
338  /* Allocate temporary buffer (if this is the first time this function is being called) */
339  if( (sc->recs[rec_idx].ID != IMS_UNASSIGNED) && (sc->rec_sig_tmp[IMS_EG_CURRENT][rec_idx] == NULL) ){
340  sc->rec_sig_tmp[IMS_EG_CURRENT][rec_idx] = (float**)malloc2d(sc->recs[rec_idx].nChannels, IMS_MAX_NSAMPLES_PER_FRAME, sizeof(float));
341  sc->rec_sig_tmp[IMS_EG_PREV][rec_idx] = (float**)malloc2d(sc->recs[rec_idx].nChannels, IMS_MAX_NSAMPLES_PER_FRAME, sizeof(float));
342  }
343  if(sc->framesize!=nSamples){
344  sc->framesize = nSamples;
345  sc->interpolator_fIn = realloc1d(sc->interpolator_fIn, nSamples*sizeof(float));
346  sc->interpolator_fOut = realloc1d(sc->interpolator_fOut, nSamples*sizeof(float));
347  for(i=0; i<nSamples; i++){
348  sc->interpolator_fIn[i] = (i+1)*1.0f/(float)nSamples;
349  sc->interpolator_fOut[i] = 1.0f-sc->interpolator_fIn[i];
350  }
351  }
352 
353  /* Initialise all receiver channels with zeros */
354  for(ch=0; ch<sc->recs[rec_idx].nChannels; ch++) /* (looping over channels since this array is not guaranteed to be contiguous) */
355  memset(sc->recs[rec_idx].sigs[ch], 0, nSamples * sizeof(float));
356 
357  /* Process all active sources for this specific receiver, directly in the time-domain */
358  for(src_idx = 0; src_idx < IMS_MAX_NUM_SOURCES; src_idx++){
359  if( (sc->srcs[src_idx].ID!=IMS_UNASSIGNED) && (sc->recs[rec_idx].ID!=IMS_UNASSIGNED) ){
360 
361  /* Broad-band operation */
362  if(sc->nBands==1)
363  memcpy(sc->src_sigs_bands[src_idx][0], sc->srcs[src_idx].sig, nSamples*sizeof(float));
364  else /* OR: Pass source signal through the Favrot & Faller (power-complementary) IIR filterbank */
365  faf_IIRFilterbank_apply(sc->hFaFbank[src_idx], sc->srcs[src_idx].sig, sc->src_sigs_bands[src_idx], nSamples);
366 
367  /* Workspace handle for this source/receiver combination */
368  wrk = sc->hCoreWrkSpc[rec_idx][src_idx];
369 
370  /* k=0 is for the current echogram,
371  * k=1 (when applyCrossFadeFLAG is enabled) is for the previous echogram */
372  for(k=sc->applyCrossFadeFLAG[rec_idx][src_idx]; k>=0; k--){
373 
374  /* Loop over samples */
375  for(n=0; n<nSamples; n++){
376 
377  /* Determine write index */
378  wIdx_n = sc->wIdx[k][rec_idx][src_idx] & IMS_CIRC_BUFFER_LENGTH_MASK;
379 
380  /* Since the time vector is the same across bands, it makes sense to determine the read-indices only once... */
381  if(k==1)
382  echogram_abs_0 = (echogram_data*)wrk->hPrevEchogram_abs[0];
383  else
384  echogram_abs_0 = (echogram_data*)wrk->hEchogram_abs[0];
385 
386  /* Handle the special case of an empty echogram */
387  if(echogram_abs_0->numImageSources==0){
388  /* Set output to 0... */
389  for(ch=0; ch<sc->recs[rec_idx].nChannels; ch++)
390  sc->rec_sig_tmp[k][rec_idx][ch][n] = 0.0f;
391 
392  /* Store current sample (per band) into the circular buffer */
393  for(band=0; band < sc->nBands; band++){
394  sc->circ_buffer[k][src_idx][band][wIdx_n] = sc->src_sigs_bands[src_idx][band][n];
395  if(sc->applyCrossFadeFLAG[rec_idx][src_idx]==0)
396  sc->circ_buffer[IMS_EG_PREV][src_idx][band][wIdx_n] = sc->src_sigs_bands[src_idx][band][n];
397  }
398 
399  /* Increment write index */
400  sc->wIdx[k][rec_idx][src_idx]++;
401  if(sc->applyCrossFadeFLAG[rec_idx][src_idx]==0)
402  sc->wIdx[IMS_EG_PREV][rec_idx][src_idx]++;
403 
404  continue; /* to next sample... */
405  }
406 
407  /* Convert time from seconds to number of samples */
408  memset(echogram_abs_0->tmp1, 0, echogram_abs_0->numImageSources*sizeof(float));
409  cblas_saxpy(echogram_abs_0->numImageSources, (sc->fs), echogram_abs_0->time, 1, echogram_abs_0->tmp1, 1);
410 
411  /* Determine read-indices and (optionally) also the interpolation weights */
412  if(fractionalDelaysFLAG){
413  /* Loop over all image sources, and determine the circular buffer read indices */
414  for(im=0; im <echogram_abs_0->numImageSources; im++){
415  /* Base read-index */
416  time_samples = (int)(echogram_abs_0->tmp1[im]); /* FLOOR */
417  rIdx = IMS_CIRC_BUFFER_LENGTH-time_samples + sc->wIdx[k][rec_idx][src_idx] /* read index for this image source */
418  + (IMS_LAGRANGE_ORDER/2); /* in order to correctly centre the filter */
419  echogram_abs_0->rIdx[im] = rIdx & IMS_CIRC_BUFFER_LENGTH_MASK; /* wrap-around if needed */
420  }
421 
422  /* Find fractional parts */
423  utility_svmod(echogram_abs_0->tmp1, echogram_abs_0->ones_dummy, echogram_abs_0->numImageSources, echogram_abs_0->tmp2);
424 
425  /* Find read-indices for interpolator */
426  for(im=0; im <echogram_abs_0->numImageSources; im++){
427  /* Centre the filter */
428  echogram_abs_0->tmp2[im] += (float)(IMS_LAGRANGE_ORDER/2);
429 
430  /* Read-indices for lagrange interpolation */
431  for(i=1; i<IMS_LAGRANGE_ORDER; i++){
432  echogram_abs_0->rIdx_frac[i-1][im] = echogram_abs_0->rIdx[im] - i;
433  /* Wrap around if needed */
434  if(echogram_abs_0->rIdx_frac[i-1][im]<0)
435  echogram_abs_0->rIdx_frac[i-1][im] += IMS_CIRC_BUFFER_LENGTH;
436  }
437  }
438 
439  /* Compute interpolation weights */ // TODO: This line is around 50% CPU usage of the whole function, a look-up table would be faster...
440  lagrangeWeights(IMS_LAGRANGE_ORDER, echogram_abs_0->tmp2, echogram_abs_0->numImageSources, FLATTEN2D(echogram_abs_0->h_frac));
441  }
442  else{
443  /* Loop over all image sources, and determine the circular buffer read indices based on the nearest sample */
444  for(im=0; im <echogram_abs_0->numImageSources; im++){
445  time_samples = (int)(echogram_abs_0->tmp1[im] + 0.5f); /* ROUND to nearest sample */
446  rIdx = IMS_CIRC_BUFFER_LENGTH-time_samples + sc->wIdx[k][rec_idx][src_idx]; /* read index for this image source */
447  echogram_abs_0->rIdx[im] = rIdx & IMS_CIRC_BUFFER_LENGTH_MASK; /* wrap-around if needed */
448  }
449  }
450 
451  /* Loop over octave bands */
452  for(band=0; band < sc->nBands; band++){
453  /* Echogram for this source/receiver at this band */
454  if(k==1)
455  echogram_abs = (echogram_data*)wrk->hPrevEchogram_abs[band];
456  else
457  echogram_abs = (echogram_data*)wrk->hEchogram_abs[band];
458  assert(echogram_abs_0->numImageSources == echogram_abs->numImageSources);
459 
460  /* Pull values from the circular buffer corresponding to these read indices, and store this sparse vector as a "compressed" vector */
461  utility_ssv2cv_inds(sc->circ_buffer[k][src_idx][band], echogram_abs_0->rIdx, echogram_abs->numImageSources, echogram_abs->cb_vals[0]);
462 
463  /* Apply interpolation if enabled */
464  if(fractionalDelaysFLAG){
465  /* Apply the weights corresponding to the first tap in the filter */
466  utility_svvmul(echogram_abs->cb_vals[0], echogram_abs_0->h_frac[0], echogram_abs->numImageSources, echogram_abs->tmp1);
467 
468  /* Step through the rest of the filter */
469  for(i=1; i<IMS_LAGRANGE_ORDER; i++){
470  /* Pull values from the circular buffer corresponding to the read-indices for the current tap in the filter */
471  utility_ssv2cv_inds(sc->circ_buffer[k][src_idx][band], echogram_abs_0->rIdx_frac[i-1], echogram_abs->numImageSources, echogram_abs->cb_vals[0]);
472 
473  /* Apply interpolation weights */
474  utility_svvmul(echogram_abs->cb_vals[0], echogram_abs_0->h_frac[i], echogram_abs->numImageSources, echogram_abs->cb_vals[0]);
475 
476  /* Sum */
477  utility_svvadd(echogram_abs->tmp1, echogram_abs->cb_vals[0], echogram_abs->numImageSources, echogram_abs->tmp1);
478  }
479 
480  /* Copy result */
481  cblas_scopy(echogram_abs->numImageSources, echogram_abs->tmp1, 1, echogram_abs->cb_vals[0], 1);
482  }
483 
484  /* Replicate these circular buffer values for all output channels */
485  for(ch=1; ch<sc->recs[rec_idx].nChannels; ch++)
486  cblas_scopy(echogram_abs->numImageSources, echogram_abs->cb_vals[0], 1, echogram_abs->cb_vals[ch], 1);
487 
488  /* Apply the echogram scalings to each image source - for all channels */
489  utility_svvmul(FLATTEN2D(echogram_abs->value), FLATTEN2D(echogram_abs->cb_vals),
490  (sc->recs[rec_idx].nChannels)*echogram_abs->numImageSources, FLATTEN2D(echogram_abs->contrib));
491

493  for(ch=0; ch<sc->recs[rec_idx].nChannels; ch++)
494  sc->rec_sig_tmp[k][rec_idx][ch][n] = cblas_sdot(echogram_abs->numImageSources, echogram_abs->ones_dummy,
495  1, echogram_abs->contrib[ch], 1);
496 
497  /* Store current sample into the circular buffer */
498  sc->circ_buffer[k][src_idx][band][wIdx_n] = sc->src_sigs_bands[src_idx][band][n];
499  if(sc->applyCrossFadeFLAG[rec_idx][src_idx]==0)
500  sc->circ_buffer[IMS_EG_PREV][src_idx][band][wIdx_n] = sc->src_sigs_bands[src_idx][band][n];
501  }
502 
503  /* Increment write index */
504  sc->wIdx[k][rec_idx][src_idx]++;
505  if(sc->applyCrossFadeFLAG[rec_idx][src_idx]==0)
506  sc->wIdx[IMS_EG_PREV][rec_idx][src_idx]++;
507 
508  } /* Loop over samples */
509  } /* Loop over slots */
510 
511  /* Cross-fade between the buffers rendered using the previous and current echograms */
512  if(sc->applyCrossFadeFLAG[rec_idx][src_idx]){
513  for(ch=0; ch<sc->recs[rec_idx].nChannels; ch++){
514  /* Apply linear interpolator to fade in with the new echogram and fade out with the previous echogram */
515  utility_svvmul(sc->rec_sig_tmp[IMS_EG_CURRENT][rec_idx][ch], sc->interpolator_fIn, nSamples, sc->rec_sig_tmp[IMS_EG_CURRENT][rec_idx][ch]);
516  utility_svvmul(sc->rec_sig_tmp[IMS_EG_PREV][rec_idx][ch], sc->interpolator_fOut, nSamples, sc->rec_sig_tmp[IMS_EG_PREV][rec_idx][ch]);
517 
518  /* Sum the result */
519  cblas_saxpy(nSamples, 1.0f, sc->rec_sig_tmp[IMS_EG_CURRENT][rec_idx][ch], 1, sc->recs[rec_idx].sigs[ch], 1);
520  cblas_saxpy(nSamples, 1.0f, sc->rec_sig_tmp[IMS_EG_PREV][rec_idx][ch], 1, sc->recs[rec_idx].sigs[ch], 1);
521  }
522 
523  /* No longer need to cross-fade for future frames (unless the echograms change again that is...) */
524  sc->applyCrossFadeFLAG[rec_idx][src_idx] = 0;
525  }
526  else
527  for(ch=0; ch<sc->recs[rec_idx].nChannels; ch++)
528  cblas_saxpy(nSamples, 1.0f, sc->rec_sig_tmp[IMS_EG_CURRENT][rec_idx][ch], 1, sc->recs[rec_idx].sigs[ch], 1);
529 
530  } /* If source active */
531 
532  } /* Loop over sources */
