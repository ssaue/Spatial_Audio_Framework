69 void getDecorrelationDelays
70 (
71  int nChannels, /* number of channels */
72  float* freqs, /* centre frequencies */
73  int nFreqs, /* number of elements in frequency vector */
74  float fs, /* host fs */
75  int maxTFdelay, /* max number of time-slots to delay */
76  int hopSize, /* STFT hop size */
77  int* delayTF /* nFreq x nChannels */
78 )
79 {
80  int band, ch;
81  int* randperm_nCH;
82  float maxMilliseconds, nChannelsf;
83  float* delayRangeMax, *delayRangeMin, *tmp_delays, *delays;
84 
85  nChannelsf = (float)nChannels;
86  randperm_nCH = malloc1d(nChannels*sizeof(int));
87  delayRangeMax = malloc1d(nFreqs*sizeof(float));
88  delayRangeMin = malloc1d(nFreqs*sizeof(float));
89  tmp_delays = malloc1d(nChannels*sizeof(float));
90  delays = malloc1d(nFreqs*nChannels*sizeof(float));
91  maxMilliseconds = MIN(80.0f, (maxTFdelay-1.0f)*(float)hopSize/fs*1000.0f);
92  for(band=0; band<nFreqs; band++){
93  delayRangeMax[band] = MAX(7.0f, MIN(maxMilliseconds, 50.0f*1000.0f/(freqs[band]+2.23e-9f)));
94  delayRangeMin[band] = MAX(3.0f, MIN(20.0f, 10.0f*1000.0f/(freqs[band]+2.23e-9f)));
95  }
96  for(band=0; band<nFreqs; band++)
97  for(ch=0; ch<nChannels; ch++)
98  delays[band*nChannels+ch] = (float)ch/nChannelsf + ((float)rand()/(float)RAND_MAX)/nChannelsf;
99  for(band=0; band<nFreqs; band++){
100  randperm(nChannels, randperm_nCH);
101  memcpy(tmp_delays, &delays[band*nChannels], nChannels*sizeof(float));
102  for(ch=0; ch<nChannels; ch++)
103  delays[band*nChannels+ch] = tmp_delays[randperm_nCH[ch]];
104  }
105  for(band=0; band<nFreqs; band++){
106  for(ch=0; ch<nChannels; ch++){
107  delays[band*nChannels+ch] = delays[band*nChannels+ch]*(delayRangeMax[band]-delayRangeMin[band])+delayRangeMin[band];
108  delayTF[band*nChannels+ch] = MAX((int)(delays[band*nChannels+ch]/1000.0f*fs/(float)hopSize + 0.5f)-1,0);
109  }
110  }
111 
112  free(randperm_nCH);
113  free(delayRangeMax);
114  free(delayRangeMin);
115  free(tmp_delays);
116  free(delays);
117 }

119 void synthesiseNoiseReverb
120 (
121  int nCH,
122  float fs,
123  float* t60,
124  float* fcen_oct,
125  int nBands,
126  int flattenFLAG,
127  float** rir_filt,
128  int* rir_len
129 )
130 {
131  int i, j, k, rir_filt_len, rir_filt_lout, filterOrder;
132  float alpha, max_t60, t;
133  float *rir, *fcut, *h_filt, *rir_filt_tmp;
134 
135  filterOrder = 800;
136 
137  /* find rir length */
138  max_t60 = 0.0f;
139  for(i=0; i<nBands; i++)
140  max_t60 = max_t60 < t60[i] ? t60[i] : max_t60;
141  rir_filt_len = (int)(max_t60*fs+0.5f); /* length of RIRs */
142  rir_filt_lout = rir_filt_len + filterOrder/2; /* truncated output length */
143 
144  /* Generate noise and shape with exponentially decaying envelopes */
145  rir = calloc1d(nCH*nBands*rir_filt_lout, sizeof(float));
146  for(i=0; i<nCH; i++){
147  for(j=0; j<nBands; j++){
148  /* decay constants for t60 */
149  alpha = 3.0f*logf(10.0f)/t60[j];
150  for(k=0, t=0.0f; k<rir_filt_len; k++, t+=1.0f/fs)
151  rir[i*nBands*rir_filt_lout + j*rir_filt_lout + k] = expf(-t*alpha) * /* envelope */
152  2.0f * ((float)rand()/(float)RAND_MAX-0.5f); /* whitenoise */
153  }
154  }
155 
156  /* get bank of FIRs filters - octave bands */
157  fcut = malloc1d((nBands-1)*sizeof(float));
158  h_filt = malloc1d(nBands*(filterOrder+1)*sizeof(float));
159  getOctaveBandCutoffFreqs(fcen_oct, nBands, fcut);
160  FIRFilterbank(filterOrder, fcut, (nBands-1), fs, WINDOWING_FUNCTION_HAMMING, 1, h_filt);
161 
162  /* filter RIRs with filterbank */
163  (*rir_filt) = realloc1d((*rir_filt), nCH*rir_filt_lout*sizeof(float));
164  memset((*rir_filt), 0, nCH*rir_filt_lout*sizeof(float));
165  rir_filt_tmp = malloc1d(nBands*rir_filt_lout*sizeof(float));
166  for(i=0; i<nCH; i++){
167  fftfilt(&rir[i*nBands*rir_filt_lout], h_filt, rir_filt_lout, filterOrder+1, nBands, rir_filt_tmp);
168  /* sum over bands */
169  for(j=0; j<nBands; j++){
170  for(k=0; k<rir_filt_lout; k++)
171  (*rir_filt)[i*rir_filt_lout+k] += rir_filt_tmp[j*rir_filt_lout+k];
172  }
173  }
174 
175  /* equalise, to force flat magnitude response */
176  if(flattenFLAG)
177  for(i=0; i<nCH; i++)
178  flattenMinphase(&((*rir_filt)[i*rir_filt_lout]), rir_filt_lout);
179 
180  /* remove filterbank delay */
181  for(i=0; i<nCH; i++)
182  memcpy(&((*rir_filt)[i*rir_filt_len]), &((*rir_filt)[i*rir_filt_lout + filterOrder/2]), rir_filt_len*sizeof(float));
183  (*rir_len) = rir_filt_len;
184 
185  /* clean-up */
186  free(rir);
187  free(fcut);
188  free(h_filt);
189  free(rir_filt_tmp);
190 }

192 void latticeDecorrelator_create
193 (
194  void** phDecor,
195  float fs,
196  int hopsize,
197  float* freqVector,
198  int nBands,
199  int nCH,
200  int* orders,
201  float* freqCutoffs,
202  int nCutoffs,
203  int maxDelay,
204  int lookupOffset,
205  float enComp_coeff
206 )
207 {
208  *phDecor = malloc1d(sizeof(latticeDecor_data));
209  latticeDecor_data *h = (latticeDecor_data*)(*phDecor);
210  int i, band, ch, o, filterIdx;
211 
212  h->nCH = nCH;
213  h->nCutoffs = nCutoffs;
214  h->nBands = nBands;
215 
216  /* alloc */
217  h->orders = malloc1d(nCutoffs*sizeof(int));
218  memcpy(h->orders, orders, nCutoffs*sizeof(int));
219  h->TF_delays = malloc1d(nBands * nCH * sizeof(int));
220  h->lttc_apf = (latticeAPF**)malloc2d(nBands, nCH, sizeof(latticeAPF));
221  h->enComp_coeff = enComp_coeff;
222  h->in_energy = (float**)calloc2d(nBands, nCH, sizeof(float));
223  h->decor_energy = (float**)calloc2d(nBands, nCH, sizeof(float));
224 
225  /* Static delays */
226  getDecorrelationDelays(h->nCH, freqVector, h->nBands, fs, maxDelay, hopsize, h->TF_delays);
227 
228  /* set-up allpass filters per band and channel */
229  for(band=0; band<nBands; band++){
230  filterIdx = -1;
231 
232  /* Find filter index for this band: */
233  for(o=0; o<nCutoffs; o++){
234  if(freqVector[band]<freqCutoffs[o]){
235  filterIdx = o;
236  break;
237  }
238  }
239 
240  /* Pull lattice allpass filter coefficients from database */
241  for(ch=0; ch<nCH; ch++){
242  if(filterIdx == -1){
243  /* Not needed... */
244  h->lttc_apf[band][ch].order = -1;
245  h->lttc_apf[band][ch].filterLength = 0;
246  h->lttc_apf[band][ch].coeffs = NULL;
247  h->lttc_apf[band][ch].buffer = NULL;
248  }
249  else{
250  h->lttc_apf[band][ch].order = h->orders[filterIdx];
251  h->lttc_apf[band][ch].coeffs = (float**)malloc2d(2, h->orders[filterIdx], sizeof(float));
252  h->lttc_apf[band][ch].buffer = calloc1d(h->lttc_apf[band][ch].order, sizeof(float_complex));
253 
254  /* numerator coefficients */
255  switch(orders[filterIdx]){
256  case 20: memcpy(h->lttc_apf[band][ch].coeffs[0], __lattice_coeffs_o20[ch+lookupOffset], (h->orders[filterIdx])*sizeof(float)); break;
257  case 18: memcpy(h->lttc_apf[band][ch].coeffs[0], __lattice_coeffs_o18[ch+lookupOffset], (h->orders[filterIdx])*sizeof(float)); break;
258  case 16: memcpy(h->lttc_apf[band][ch].coeffs[0], __lattice_coeffs_o16[ch+lookupOffset], (h->orders[filterIdx])*sizeof(float)); break;
259  case 15: memcpy(h->lttc_apf[band][ch].coeffs[0], __lattice_coeffs_o15[ch+lookupOffset], (h->orders[filterIdx])*sizeof(float)); break;
260  case 14: memcpy(h->lttc_apf[band][ch].coeffs[0], __lattice_coeffs_o14[ch+lookupOffset], (h->orders[filterIdx])*sizeof(float)); break;
261  case 12: memcpy(h->lttc_apf[band][ch].coeffs[0], __lattice_coeffs_o12[ch+lookupOffset], (h->orders[filterIdx])*sizeof(float)); break;
262  case 10: memcpy(h->lttc_apf[band][ch].coeffs[0], __lattice_coeffs_o10[ch+lookupOffset], (h->orders[filterIdx])*sizeof(float)); break;
263  case 8: memcpy(h->lttc_apf[band][ch].coeffs[0], __lattice_coeffs_o8[ch+lookupOffset], (h->orders[filterIdx])*sizeof(float)); break;
264  case 6: memcpy(h->lttc_apf[band][ch].coeffs[0], __lattice_coeffs_o6[ch+lookupOffset], (h->orders[filterIdx])*sizeof(float)); break;
265  case 4: memcpy(h->lttc_apf[band][ch].coeffs[0], __lattice_coeffs_o4[ch+lookupOffset], (h->orders[filterIdx])*sizeof(float)); break;
266  case 3: memcpy(h->lttc_apf[band][ch].coeffs[0], __lattice_coeffs_o3[ch+lookupOffset], (h->orders[filterIdx])*sizeof(float)); break;
267  case 2: memcpy(h->lttc_apf[band][ch].coeffs[0], __lattice_coeffs_o2[ch+lookupOffset], (h->orders[filterIdx])*sizeof(float)); break;
268  default: assert(0); /* Unsupported filter order specified */
269  }
270 
271  /* denominator coefficients */
272  for(i=0; i<orders[filterIdx]; i++)
273  h->lttc_apf[band][ch].coeffs[1][i] = h->lttc_apf[band][ch].coeffs[0][orders[filterIdx]-i-1];
274  }
275  }
276  }
277 
278  /* Run-time */
279  h->maxBufferLen = maxDelay+1;
280  h->delayBuffers = (float_complex***)calloc3d(nBands, nCH, h->maxBufferLen, sizeof(float_complex));
281  h->rIdx = malloc1d(nBands*nCH*sizeof(int));
282  for(band=0; band<nBands; band++)
283  for(ch=0; ch<nCH; ch++)
284  h->rIdx[band*nCH+ch] = h->TF_delays[band*nCH+ch];
285  h->wIdx = calloc1d(nBands*nCH,sizeof(int));
286 }

333 void latticeDecorrelator_apply
334 (
335  void* hDecor,
336  float_complex*** inFrame,
337  int nTimeSlots,
338  float_complex*** decorFrame
339 )
340 {
341  latticeDecor_data *h = (latticeDecor_data*)(hDecor);
342  int band, ch, t, i;
343  float_complex ytmp, xtmp;
344 
345  for(t=0; t<nTimeSlots; t++){
346  for(band=0; band <h->nBands; band++){
347  for(ch=0; ch<h->nCH; ch++){
348  /* Apply fixed delay */
349  h->delayBuffers[band][ch][h->wIdx[band*(h->nCH) + ch]] = inFrame[band][ch][t];
350  decorFrame[band][ch][t] = h->delayBuffers[band][ch][h->rIdx[band*(h->nCH) + ch]];
351 
352  /* increment and wrap-around as needed */
353  h->rIdx[band*(h->nCH) + ch]++;
354  h->wIdx[band*(h->nCH) + ch]++;
355  if( h->rIdx[band*(h->nCH) + ch] >= h->TF_delays[band*(h->nCH) + ch] )
356  h->rIdx[band*(h->nCH) + ch] = 0;
357  if( h->wIdx[band*(h->nCH) + ch] >= h->TF_delays[band*(h->nCH) + ch] )
358  h->wIdx[band*(h->nCH) + ch] = 0;
359  }
360  }
361  }
362 
363  /* Apply lattice allpass filters */
364 #if _MSC_VER >= 1900
365  for(band=0; band <h->nBands; band++){
366  for(ch=0; ch<h->nCH; ch++){
367  if(h->lttc_apf[band][ch].buffer!=NULL){ /* only if filter is defined... */
368  for(t=0; t<nTimeSlots; t++){
369  /* Compute energy of the input */
370  h->in_energy[band][ch] = (1.0f-h->enComp_coeff)*powf(cabsf(inFrame[band][ch][t]),2.0f) + (h->enComp_coeff)*h->in_energy[band][ch];
371 
372  /* First tap in filter */
373  xtmp = decorFrame[band][ch][t];
374  ytmp = ccaddf(h->lttc_apf[band][ch].buffer[0], crmulf(xtmp, (h->lttc_apf[band][ch].coeffs[0][0])));
375  decorFrame[band][ch][t] = ytmp;
376 
377  /* Energy compensation */
378  h->decor_energy[band][ch] = (1.0f-h->enComp_coeff)*powf(cabsf(decorFrame[band][ch][t]),2.0f) + (h->enComp_coeff)*h->decor_energy[band][ch];
379  decorFrame[band][ch][t] = crmulf(decorFrame[band][ch][t], MIN(sqrtf(h->in_energy[band][ch]/(h->decor_energy[band][ch]+2.23e-9f)), 1.0f));
380 
381  /* propagate through the rest of the lattice filter structure */
382  for(i=0; i<h->lttc_apf[band][ch].order-1; i++){
383  h->lttc_apf[band][ch].buffer[i] = ccaddf(h->lttc_apf[band][ch].buffer[i+1],
384  ccsubf(crmulf(xtmp, h->lttc_apf[band][ch].coeffs[0][i+1]), /* numerator */
385  crmulf(ytmp, h->lttc_apf[band][ch].coeffs[1][i+1]))); /* denominator */
386  }
387  }
388  }
389  }
390  }
391 #else
392  for(band=0; band <h->nBands; band++){
393  for(ch=0; ch<h->nCH; ch++){
394  if(h->lttc_apf[band][ch].buffer!=NULL){ /* only if filter is defined */
395  for(t=0; t<nTimeSlots; t++){
396  /* Compute energy of the input */
397  h->in_energy[band][ch] = (1.0f-h->enComp_coeff)*powf(cabsf(inFrame[band][ch][t]),2.0f) + (h->enComp_coeff)*h->in_energy[band][ch];
398 
399  /* First tap in filter */
400  xtmp = decorFrame[band][ch][t];
401  ytmp = h->lttc_apf[band][ch].buffer[0] + xtmp * (h->lttc_apf[band][ch].coeffs[0][0]);
402  decorFrame[band][ch][t] = ytmp;
403 
404  /* Energy compensation */
405  h->decor_energy[band][ch] = (1.0f-h->enComp_coeff)*powf(cabsf(decorFrame[band][ch][t]), 2.0f) + (h->enComp_coeff)*(h->decor_energy[band][ch]);
406  decorFrame[band][ch][t] *= MIN(sqrtf(h->in_energy[band][ch]/(h->decor_energy[band][ch]+2.23e-9f)), 1.0f);
407 
408  /* propagate through the rest of the lattice filter structure */
409  for(i=0; i<h->lttc_apf[band][ch].order-1; i++){
410  h->lttc_apf[band][ch].buffer[i] = h->lttc_apf[band][ch].buffer[i+1] +
411  h->lttc_apf[band][ch].coeffs[0][i+1] * xtmp - /* numerator */
412  h->lttc_apf[band][ch].coeffs[1][i+1] * ytmp; /* denominator */
413  }
414  }
415  }
416  }
417  }
418 #endif
419 }

454 void transientDucker_apply
455 (
456  void* hDucker,
457  float_complex*** inFrame,
458  int nTimeSlots,
459  float alpha,
460  float beta,
461  float_complex*** residualFrame,
462  float_complex*** transientFrame
463 )
464 {
465  transientDucker_data *h = (transientDucker_data*)(hDucker);
466  int band, i, t;
467  float detectorEne, transientEQ;
468  //const float alpha = 0.95f;
469  //const float beta = 0.995f;
470 
471  for(band=0; band<h->nBands; band++){
472  for(i=0; i<h->nCH; i++){
473  for(t=0; t<nTimeSlots; t++){
474  detectorEne = cabsf(inFrame[band][i][t]);
475  detectorEne = detectorEne*detectorEne;
476  h->transientDetector1[band][i] *= alpha;
477  if(h->transientDetector1[band][i]<detectorEne)
478  h->transientDetector1[band][i] = detectorEne;
479  h->transientDetector2[band][i] = h->transientDetector2[band][i]*beta + (1.0f-beta)*(h->transientDetector1[band][i]);
480  if(h->transientDetector2[band][i] > h->transientDetector1[band][i])
481  h->transientDetector2[band][i] = h->transientDetector1[band][i];
482  transientEQ = MIN(1.0f, 4.0f * (h->transientDetector2[band][i])/(h->transientDetector1[band][i]+2.23e-9f));
483 #ifdef _MSC_VER
484  if(residualFrame!=NULL)
485  residualFrame[band][i][t] = crmulf(inFrame[band][i][t], transientEQ);
486  if(transientFrame!=NULL)
487  transientFrame[band][i][t] = crmulf(inFrame[band][i][t], 1.0f-transientEQ);
488 #else
489  if(residualFrame!=NULL)
490  residualFrame[band][i][t] = inFrame[band][i][t] * transientEQ;
491  if(transientFrame!=NULL)
492  transientFrame[band][i][t] = inFrame[band][i][t] * (1.0f-transientEQ);
493 #endif
494  }
495  }
496  }
