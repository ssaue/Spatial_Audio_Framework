48 void saf_matrixConv_create
49 (
50  void ** const phMC,
51  int hopSize,
52  float* H, /* nCHout x nCHin x length_h */
53  int length_h,
54  int nCHin,
55  int nCHout,
56  int usePartFLAG
57 )
58 {
59  *phMC = malloc1d(sizeof(safMatConv_data));
60  safMatConv_data *h = (safMatConv_data*)(*phMC);
61  int no, ni, nb;
62  float* h_pad, *h_pad_2hops;
63 
64  h->hopSize = hopSize;
65  h->length_h = length_h;
66  h->nCHin = nCHin;
67  h->nCHout = nCHout;
68  h->usePartFLAG = usePartFLAG;
69 
70  if(!h->usePartFLAG){
71  /* intialise non-partitioned convolution mode */
72  h->numOvrlpAddBlocks = (int)(ceilf((float)(hopSize+length_h-1)/(float)hopSize)+0.1f);
73  //h->numOvrlpAddBlocks = nextpow2((int)(ceilf((float)(hopSize+length_h-1)/(float)hopSize)+0.1f));
74  h->fftSize = (h->numOvrlpAddBlocks)*hopSize;
75  h->nBins = h->fftSize/2 + 1;
76 
77  /* Allocate memory for buffers and perform fft on H */
78  h->ovrlpAddBuffer = calloc1d(nCHout*(h->fftSize), sizeof(float));
79  h->x_pad = calloc1d((h->nCHin)*(h->fftSize), sizeof(float)); // CALLOC
80  h->y_pad = malloc1d((h->nCHout)*(h->fftSize)*sizeof(float));
81  h->hx_n = malloc1d((h->fftSize) * sizeof(float));
82  h->H_f = malloc1d((h->nCHout)*(h->nCHin)*(h->nBins)*sizeof(float_complex));
83  h->X_n = malloc1d((h->nCHout)*(h->nCHin)*(h->nBins)*sizeof(float_complex));
84  h->HX_n = malloc1d((h->nCHout)*(h->nCHin)*(h->nBins)*sizeof(float_complex));
85  h->z_n = malloc1d((h->fftSize) * sizeof(float));
86  saf_rfft_create(&(h->hFFT), h->fftSize);
87  h_pad = calloc1d(h->fftSize, sizeof(float));
88  for(no=0; no<nCHout; no++){
89  for(ni=0; ni<nCHin; ni++){
90  memcpy(h_pad, &(H[no*nCHin*length_h+ni*length_h]), length_h*sizeof(float));
91  saf_rfft_forward(h->hFFT, h_pad, &(h->H_f[no*nCHin*(h->nBins)+ni*(h->nBins)]));
92  }
93  }
94  free(h_pad);
95  }
96  else{
97  /* intialise partitioned convolution mode */
98  h->length_h = length_h;
99  h->fftSize = 2*(h->hopSize);
100  h->nBins = hopSize+1;
101  h->numFilterBlocks = (int)ceilf((float)length_h/(float)hopSize); /* number of partitions */
102  assert(h->numFilterBlocks>=1);
103 
104  /* Allocate memory for buffers and perform fft on partitioned H */
105  h_pad = calloc1d(h->numFilterBlocks * hopSize, sizeof(float));
106  h_pad_2hops = calloc1d(2 * hopSize, sizeof(float));
107  h->Hpart_f = malloc1d(nCHout*sizeof(float_complex*));
108  h->X_n = calloc1d(h->numFilterBlocks * nCHin * (h->nBins), sizeof(float_complex));
109  h->HX_n = malloc1d(h->numFilterBlocks * nCHin * (h->nBins) * sizeof(float_complex));
110  h->x_pad = calloc1d(2 * hopSize, sizeof(float));
111  h->hx_n = malloc1d(h->numFilterBlocks*nCHin*(h->fftSize)*sizeof(float));
112  h->y_n_overlap = calloc1d(nCHout*hopSize, sizeof(float));
113  h->z_n = malloc1d((h->fftSize) * sizeof(float));
114  saf_rfft_create(&(h->hFFT), h->fftSize);
115  for(no=0; no<nCHout; no++){
116  h->Hpart_f[no] = malloc1d(h->numFilterBlocks*nCHin*(h->nBins)*sizeof(float_complex));
117  for(ni=0; ni<nCHin; ni++){
118  memcpy(h_pad, &H[no*nCHin*length_h+ni*length_h], length_h*sizeof(float)); /* zero pad filter, to be multiple of hopsize */
119  for (nb=0; nb<h->numFilterBlocks; nb++){
120  memcpy(h_pad_2hops, &(h_pad[nb*hopSize]), hopSize*sizeof(float));
121  saf_rfft_forward(h->hFFT, h_pad_2hops, &(h->Hpart_f[no][nb*nCHin*(h->nBins)+ni*(h->nBins)]));
122  }
123  }
124  }
125 
126  free(h_pad);
127  free(h_pad_2hops);
128  }
129 }

162 void saf_matrixConv_apply
163 (
164  void * const hMC,
165  float* inputSig,
166  float* outputSig
167 )
168 {
169  safMatConv_data *h = (safMatConv_data*)(hMC);
170  int ni, no, nb;
171 
172  /* apply non-partitioned convolution */
173  if(!h->usePartFLAG){
174  /* zero-pad input signals and perform fft */
175  for(ni=0; ni<h->nCHin; ni++){
176  cblas_scopy(h->hopSize, &inputSig[ni*(h->hopSize)], 1, &(h->x_pad[ni*(h->fftSize)]), 1);
177  saf_rfft_forward(h->hFFT, &(h->x_pad[ni*(h->fftSize)]), &(h->X_n[ni*(h->nBins)]));
178  }
179 
180  /* Replicate for all outputs */
181  for(no=1; no<h->nCHout; no++)
182  cblas_ccopy(h->nCHin*(h->nBins), h->X_n, 1, &(h->X_n[no*ni*(h->nBins)]), 1);
183 
184  /* Multiply spectra together */
185  utility_cvvmul(h->H_f, h->X_n, (h->nCHout)*(h->nCHin)*h->nBins, h->HX_n);
186 
187  /* Loop over outputs */
188  for(no=0; no<h->nCHout; no++){
189  /* ifft and sum */
190  memset(h->z_n, 0, (h->fftSize) * sizeof(float));
191  for(ni=0; ni<h->nCHin; ni++){
192  saf_rfft_backward(h->hFFT, &(h->HX_n[no*(h->nCHin)*(h->nBins)+ni*(h->nBins)]), h->hx_n);
193  cblas_saxpy(h->fftSize, 1.0f, h->hx_n, 1, h->z_n, 1);
194  }
195 
196  /* shuffle the over-lap add buffer */
197  memmove(&(h->ovrlpAddBuffer[no*(h->fftSize)]), &(h->ovrlpAddBuffer[no*(h->fftSize)+(h->hopSize)]), (h->numOvrlpAddBlocks-1)*(h->hopSize)*sizeof(float));
198  memset(&(h->ovrlpAddBuffer[no*(h->fftSize)+(h->numOvrlpAddBlocks-1)*(h->hopSize)]), 0, (h->hopSize)*sizeof(float));
199 
200  /* sum with overlap-add buffer */
201  cblas_saxpy(h->fftSize, 1.0f, h->z_n, 1, &(h->ovrlpAddBuffer[no*(h->fftSize)]), 1);
202 
203  /* truncate buffer and output */
204  cblas_scopy(h->hopSize, &(h->ovrlpAddBuffer[no*(h->fftSize)]), 1, &(outputSig[no*(h->hopSize)]), 1);
205  }
206  }
207  /* apply partitioned convolution */
208  else{
209  /* zero-pad input signals and perform fft. Store in partition slot 1. */
210  memmove(&(h->X_n[1*(h->nCHin)*(h->nBins)]), h->X_n, (h->numFilterBlocks-1)*(h->nCHin)*(h->nBins)*sizeof(float_complex)); /* shuffle */
211  for(ni=0; ni<h->nCHin; ni++){
212  cblas_scopy(h->hopSize, &(inputSig[ni*(h->hopSize)]), 1, h->x_pad, 1);
213  saf_rfft_forward(h->hFFT, h->x_pad, &(h->X_n[0*(h->nCHin)*(h->nBins)+ni*(h->nBins)]));
214  }
215 
216  /* apply convolution and inverse fft */
217  for(no=0; no<h->nCHout; no++){
218  utility_cvvmul(h->Hpart_f[no], h->X_n, h->numFilterBlocks * (h->nCHin) * (h->nBins), h->HX_n); /* This is the bulk of the CPU work */
219  for(nb=0; nb<h->numFilterBlocks; nb++)
220  for(ni=0; ni<h->nCHin; ni++)
221  saf_rfft_backward(h->hFFT, &(h->HX_n[nb*(h->nCHin)*(h->nBins)+ni*(h->nBins)]), &(h->hx_n[nb*(h->nCHin)*(h->fftSize)+ni*(h->fftSize)]));
222 
223  /* output frame for this channel is the sum over all partitions and input channels */
224  memset(h->z_n, 0, (h->fftSize) * sizeof(float));
225  for(nb=0; nb<h->numFilterBlocks*(h->nCHin); nb++)
226  cblas_saxpy(h->fftSize, 1.0f, &(h->hx_n[nb*(h->fftSize)]), 1, h->z_n, 1);
227 
228  /* sum with overlap buffer and copy the result to the output buffer */
229  utility_svvadd(h->z_n, (const float*)&(h->y_n_overlap[no*(h->hopSize)]), h->hopSize, &(outputSig[no*(h->hopSize)]));
230 
231  /* for next iteration: */
232  cblas_scopy(h->hopSize, &(h->z_n[h->hopSize]), 1, &(h->y_n_overlap[no*(h->hopSize)]), 1);
233  }
234  }
235 }

356 void saf_multiConv_apply
357 (
358  void * const hMC,
359  float* inputSig,
360  float* outputSig
361 )
362 {
363  safMulConv_data *h = (safMulConv_data*)(hMC);
364  int nc, nb;
365 
366  /* apply non-partitioned convolution */
367  if(!h->usePartFLAG){
368  /* zero-pad input signals and perform fft. */
369  for(nc=0; nc<h->nCH; nc++){
370  memcpy(h->x_pad, &(inputSig[nc*(h->hopSize)]), h->hopSize *sizeof(float));
371  saf_rfft_forward(h->hFFT, h->x_pad, &(h->X_n[nc*(h->nBins)]));
372  }
373 
374  /* apply convolution and inverse fft */
375  utility_cvvmul(h->H_f, h->X_n, (h->nCH) * (h->nBins), h->Z_n); /* This is the bulk of the CPU work */
376  for(nc=0; nc<h->nCH; nc++){
377  saf_rfft_backward(h->hFFT, &(h->Z_n[nc*(h->nBins)]), &(h->z_n[nc*(h->fftSize)]));
378 
379  /* sum with overlap buffer and copy the result to the output buffer */
380  utility_svvcopy(&(h->ovrlpAddBuffer[nc*(h->fftSize)+(h->hopSize)]), (h->numOvrlpAddBlocks-1)*(h->hopSize), &(h->ovrlpAddBuffer[nc*(h->fftSize)]));
381  memset(&(h->ovrlpAddBuffer[nc*(h->fftSize)+(h->numOvrlpAddBlocks-1)*(h->hopSize)]), 0, (h->hopSize)*sizeof(float));
382  utility_svvadd(&(h->ovrlpAddBuffer[nc*(h->fftSize)]), &(h->z_n[nc*(h->fftSize)]), (h->fftSize), &(h->ovrlpAddBuffer[nc*(h->fftSize)]));
383  utility_svvcopy(&(h->ovrlpAddBuffer[nc*(h->fftSize)]), h->hopSize, &(outputSig[nc*(h->hopSize)]));
384  }
385  }
386  /* apply partitioned convolution */
387  else{
388  /* zero-pad input signals and perform fft. Store in partition slot 1. */
389  memcpy(&(h->X_n[1*(h->nCH)*(h->nBins)]), h->X_n, (h->numFilterBlocks-1)*(h->nCH)*(h->nBins)*sizeof(float_complex));
390  for(nc=0; nc<h->nCH; nc++){
391  memcpy(h->x_pad, &(inputSig[nc*(h->hopSize)]), h->hopSize * sizeof(float));
392  saf_rfft_forward(h->hFFT, h->x_pad, &(h->X_n[0*(h->nCH)*(h->nBins)+nc*(h->nBins)]));
393  }
394 
395  /* apply convolution and inverse fft */
396  utility_cvvmul(h->Hpart_f, h->X_n, h->numFilterBlocks * (h->nCH) * (h->nBins), h->HX_n); /* This is the bulk of the CPU work */
397  for(nc=0; nc<h->nCH; nc++){
398  for(nb=0; nb<h->numFilterBlocks; nb++)
399  saf_rfft_backward(h->hFFT, &(h->HX_n[nb*(h->nCH)*(h->nBins)+nc*(h->nBins)]), &(h->hx_n[nb*(h->nCH)*(h->fftSize)+nc*(h->fftSize)]));
400 
401  /* output frame for this channel is the sum over all partitions */
402  memset(h->z_n, 0, h->fftSize*sizeof(float));
403  for(nb=0; nb<h->numFilterBlocks; nb++)
404  utility_svvadd(h->z_n, (const float*)&(h->hx_n[nb*(h->nCH)*(h->fftSize)+nc*(h->fftSize)]), h->fftSize, h->z_n);
405 
406  /* sum with overlap buffer and copy the result to the output buffer */
407  utility_svvadd(h->z_n, (const float*)&(h->y_n_overlap[nc*(h->hopSize)]), h->hopSize, &(outputSig[nc* (h->hopSize)]));
408 
409  /* for next iteration: */
410  memcpy(&(h->y_n_overlap[nc*(h->hopSize)]), &(h->z_n[h->hopSize]), h->hopSize*sizeof(float));
411  }
412  }
