38 void convertHOAChannelConvention
39 (
40  float* insig,
41  int order,
42  int signalLength,
43  HOA_CH_ORDER inConvention,
44  HOA_CH_ORDER outConvention
45 )
46 {
47  int i, nSH;
48  float** tmp;
49 
50  nSH = ORDER2NSH(order);
51 
52  /* bypass, if 0th order, or no conversion required */
53  if(order==0 || inConvention == outConvention)
54  return;
55 #if 0 /* Now changed to just zeroing the remaining channels */
56  /* Assert that FuMa is not being used for anything other than 1st order */
57  if(order!=1){
58  /* FuMa not supported for more than 1st order */
59  assert(inConvention!=HOA_CH_ORDER_FUMA);
60  assert(outConvention!=HOA_CH_ORDER_FUMA);
61  }
62 #endif
63  /* Temp frame */
64  tmp = (float**)malloc2d(3, signalLength, sizeof(float));
65 
66  /* Convert FUMA to ACN */
67  if(inConvention==HOA_CH_ORDER_FUMA && outConvention==HOA_CH_ORDER_ACN){
68  memcpy(tmp[0], &insig[2*signalLength], signalLength*sizeof(float));
69  memcpy(tmp[1], &insig[3*signalLength], signalLength*sizeof(float));
70  memcpy(tmp[2], &insig[1*signalLength], signalLength*sizeof(float));
71 
72  memcpy(&insig[1*signalLength], tmp[0], signalLength*sizeof(float));
73  memcpy(&insig[2*signalLength], tmp[1], signalLength*sizeof(float));
74  memcpy(&insig[3*signalLength], tmp[2], signalLength*sizeof(float));
75  }
76  /* Convert ACN to FUMA */
77  else if(inConvention==HOA_CH_ORDER_ACN && outConvention==HOA_CH_ORDER_FUMA){
78  memcpy(tmp[0], &insig[3*signalLength], signalLength*sizeof(float));
79  memcpy(tmp[1], &insig[1*signalLength], signalLength*sizeof(float));
80  memcpy(tmp[2], &insig[2*signalLength], signalLength*sizeof(float));
81 
82  memcpy(&insig[1*signalLength], tmp[0], signalLength*sizeof(float));
83  memcpy(&insig[2*signalLength], tmp[1], signalLength*sizeof(float));
84  memcpy(&insig[3*signalLength], tmp[2], signalLength*sizeof(float));
85  }
86  /* Fill any remaining channels with zeros */
87  for(i=4; i<nSH; i++)
88  memset(&insig[i*signalLength], 0, signalLength * sizeof(float));
89 
90  /* clean-up */
91  free(tmp);
92 }

94 void convertHOANormConvention
95 (
96  float* insig,
97  int order,
98  int signalLength,
99  HOA_NORM inConvention,
100  HOA_NORM outConvention
101 )
102 {
103  int n, ch, i;
104 
105  if(order==0 || inConvention == outConvention)
106  return; /* Nothing to do */
107 
108  if(inConvention==HOA_NORM_N3D){
109  if(outConvention == HOA_NORM_SN3D){
110  for (n = 0; n<order+1; n++)
111  for (ch = ORDER2NSH(n-1); ch < ORDER2NSH(n); ch++)
112  for(i = 0; i<signalLength; i++)
113  insig[ch*signalLength+i] /= sqrtf(2.0f*(float)n+1.0f);
114  }
115  else if(outConvention==HOA_NORM_FUMA){
116  for(i = 0; i<signalLength; i++)
117  insig[i] /= sqrtf(2.0f);
118  for (ch = 1; ch<4 /* 1st order only */; ch++)
119  for(i = 0; i<signalLength; i++)
120  insig[ch*signalLength+i] /= sqrtf(3.0f);
121  }
122  }
123  else if(inConvention==HOA_NORM_SN3D){
124  if(outConvention == HOA_NORM_N3D){
125  for (n = 0; n<order+1; n++)
126  for (ch = ORDER2NSH(n-1); ch<ORDER2NSH(n); ch++)
127  for(i = 0; i<signalLength; i++)
128  insig[ch*signalLength+i] *= sqrtf(2.0f*(float)n+1.0f);
129  }
130  else if(outConvention==HOA_NORM_FUMA){
131  for(i = 0; i<signalLength; i++)
132  insig[i] /= sqrtf(2.0f);
133  }
134  }
135  else if(inConvention==HOA_NORM_FUMA){
136  if(outConvention == HOA_NORM_N3D){
137  for(i = 0; i<signalLength; i++)
138  insig[i] *= sqrtf(2.0f);
139  for (ch = 1; ch<4 /* 1st order only */; ch++)
140  for(i = 0; i<signalLength; i++)
141  insig[ch*signalLength+i] *= sqrtf(3.0f);
142  }
143  else if(outConvention == HOA_NORM_SN3D){
144  for(i = 0; i<signalLength; i++)
145  insig[i] *= sqrtf(2.0f);
146  }
147  }
148 }

150 void getRSH
151 (
152  int N,
153  float* dirs_deg,
154  int nDirs,
155  float* Y
156 )
157 {
158  int i, nSH;
159  float scale;
160  float* dirs_rad;
161 
162  if(nDirs<1)
163  return;
164 
165  nSH = (N+1)*(N+1);
166  scale = sqrtf(4.0f*SAF_PI);
167 
168  /* convert [azi, elev] in degrees, to [azi, inclination] in radians */
169  dirs_rad = malloc1d(nDirs*2*sizeof(float));
170  for(i=0; i<nDirs; i++){
171  dirs_rad[i*2+0] = dirs_deg[i*2+0] * SAF_PI/180.0f;
172  dirs_rad[i*2+1] = SAF_PI/2.0f - (dirs_deg[i*2+1] * SAF_PI/180.0f);
173  }
174 
175  /* get real-valued spherical harmonics */
176  getSHreal(N, dirs_rad, nDirs, Y);
177 
178  /* remove sqrt(4*pi) term */
179  utility_svsmul(Y, &scale, nSH*nDirs, NULL);
180 
181  free(dirs_rad);
182 }

184 void getRSH_recur
185 (
186  int N,
187  float* dirs_deg,
188  int nDirs,
189  float* Y
190 )
191 {
192  int n, m, i, dir, index_n;
193  float Nn0, Nnm;
194  float sleg_n[8], sleg_n_1[8], sleg_n_2[8], ssin_el, sfactorials_n[15];
195  float* leg_n, *leg_n_1, *leg_n_2, *sin_el, *factorials_n;
196 
197  if(nDirs<1)
198  return;
199 
200  if(N<=7 && nDirs == 1){
201  /* Single direction optimisation for up to 7th order */
202  leg_n = sleg_n;
203  leg_n_1 = sleg_n_1;
204  leg_n_2 = sleg_n_2;
205  sin_el = &ssin_el;
206  factorials_n = sfactorials_n;
207  }
208  else{
209  factorials_n = malloc1d((2*N+1)*sizeof(float));
210  leg_n = malloc1d((N+1)*nDirs * sizeof(float));
211  leg_n_1 = malloc1d((N+1)*nDirs * sizeof(float));
212  leg_n_2 = malloc1d((N+1)*nDirs * sizeof(float));
213  sin_el = malloc1d(nDirs * sizeof(float));
214  }
215  index_n = 0;
216 
217  /* precompute factorials */
218  for (i = 0; i < 2*N+1; i++)
219  factorials_n[i] = (float)factorial(i);
220 
221  /* cos(inclination) = sin(elevation) */
222  for (dir = 0; dir<nDirs; dir++)
223  sin_el[dir] = sinf(dirs_deg[dir*2+1] * SAF_PI/180.0f);
224 
225  /* compute SHs with the recursive Legendre function */
226  for (n = 0; n<N+1; n++) {
227  if (n==0) {
228  for (dir = 0; dir<nDirs; dir++)
229  Y[n*nDirs+dir] = 1.0f;
230  index_n = 1;
231  }
232  else {
233  unnorm_legendreP_recur(n, sin_el, nDirs, leg_n_1, leg_n_2, leg_n); /* does NOT include Condon-Shortley phase term */
234 
235  Nn0 = sqrtf(2.0f*(float)n+1.0f);
236  for (dir = 0; dir<nDirs; dir++){
237  for (m = 0; m<n+1; m++) {
238  if (m==0)
239  Y[(index_n+n)*nDirs+dir] = Nn0 * leg_n[m*nDirs+dir];
240  else {
241  Nnm = Nn0* sqrtf( 2.0f * factorials_n[n-m]/factorials_n[n+m] );
242  Y[(index_n+n-m)*nDirs+dir] = Nnm * leg_n[m*nDirs+dir] * sinf((float)m * (dirs_deg[dir*2])*SAF_PI/180.0f);
243  Y[(index_n+n+m)*nDirs+dir] = Nnm * leg_n[m*nDirs+dir] * cosf((float)m * (dirs_deg[dir*2])*SAF_PI/180.0f);
244  }
245  }
246  }
247  index_n += 2*n+1;
248  }
249  utility_svvcopy(leg_n_1, (N+1)*nDirs, leg_n_2);
250  utility_svvcopy(leg_n, (N+1)*nDirs, leg_n_1);
251  }
252 
253  if(N>7 || nDirs > 1){
254  free(factorials_n);
255  free(leg_n);
256  free(leg_n_1);
257  free(leg_n_2);
258  free(sin_el);
259  }
260 }

267 void getMaxREweights
268 (
269  int order,
270  int diagMtxFlag,
271  float* a_n
272 )
273 {
274  int n, i, idx, nSH;
275  double x;
276  double* ppm;
277 
278  x = cosf(137.9f*(SAF_PI/180.0f)/((float)order+1.51f));
279  nSH = ORDER2NSH(order);
280  if(diagMtxFlag)
281  memset(a_n, 0, nSH*nSH*sizeof(float));
282  else
283  memset(a_n, 0, nSH*sizeof(float));
284  ppm = calloc1d((order+1),sizeof(double));
285  idx = 0;
286  for(n=0; n<=order; n++){
287  unnorm_legendreP(n, &x, 1, ppm);
288  /* store the first Legendre polynomial value for each order along the
289  * diagonal of a_n */
290  for(i = 0; i<2*n+1; i++){
291  if(diagMtxFlag)
292  a_n[(idx+i)*nSH + (idx+i)] = (float)ppm[0];
293  else
294  a_n[(idx+i)] = (float)ppm[0];
295  }
296  idx += 2*n+1;
297  }
298  free(ppm);
299 }

301 void truncationEQ
302 (
303  float* w_n,
304  int order_truncated,
305  int order_target,
306  double* kr,
307  int nBands,
308  float softThreshold,
309  float* gain
310 )
311 {
312  /* For more information refer to:
313  * Hold, C., Gamper, H., Pulkki, V., Raghuvanshi, N., & Tashev, I. J. (2019).
314  * Improving Binaural Ambisonics Decoding by Spherical Harmonics Domain Tapering
315  * and Coloration Compensation. ICASSP, IEEE International Conference on Acoustics,
316  * Speech and Signal Processing - Proceedings.
317  */
318  double_complex* b_n_target = calloc1d(nBands*(order_target+1), sizeof(double_complex));
319  double_complex* b_n_truncated = calloc1d(nBands*(order_truncated+1), sizeof(double_complex));
320  double* p_target = calloc1d(nBands, sizeof(double));
321  double* p_truncated = calloc1d(nBands, sizeof(double));
322 
323  sphModalCoeffs(order_target, kr, nBands, ARRAY_CONSTRUCTION_RIGID, 0., b_n_target);
324  sphModalCoeffs(order_truncated, kr, nBands, ARRAY_CONSTRUCTION_RIGID, 0., b_n_truncated);
325 
326  /* p_full */
327  for (int idxBand=0; idxBand<nBands; idxBand++)
328  for (int idxOrder=0; idxOrder<=order_target; idxOrder++)
329  p_target[idxBand] += (2.0*idxOrder + 1.0) * pow(cabs(b_n_target[idxBand*(order_target+1) + idxOrder]), 2.0);
330  /* p_truncated from (tapered) modal weighting */
331  for (int idxBand=0; idxBand<nBands; idxBand++)
332  for (int idxOrder=0; idxOrder<=order_truncated; idxOrder++)
333  p_truncated[idxBand] += w_n[idxOrder] * (2.0*idxOrder + 1.0) * pow(cabs(b_n_truncated[idxBand*(order_truncated+1) + idxOrder]), 2.0);
334 
335  /* inverse ratio is filter gain */
336  for (int idxBand=0; idxBand < nBands; idxBand++) {
337  p_target[idxBand] = 1.0/(4.0*SAF_PI) * sqrt(p_target[idxBand]);
338  p_truncated[idxBand] = 1.0/(4.0*SAF_PI) * sqrt(p_truncated[idxBand]);
339  gain[idxBand] = (float)(p_target[idxBand] / (p_truncated[idxBand]+2.23e-13));
340  }
341 
342  /* soft clip to limit maximum gain */
343  float clipFactor = powf(10.0f, softThreshold/20.0f);
344  for (int idxBand=0; idxBand<nBands; idxBand++){
345  gain[idxBand] = gain[idxBand]/clipFactor; /* norm by threshold */
346  if (gain[idxBand] > 1.0f)
347  gain[idxBand] = 1.0f + tanhf(gain[idxBand] - 1.0f); /* soft clip to 2 */
348  gain[idxBand] = gain[idxBand] * clipFactor; /* de-norm */
349  }
350 
351  /* clean-up */
352  free(b_n_target);
353  free(b_n_truncated);
354  free(p_target);
355  free(p_truncated);
356 }

358 void getLoudspeakerDecoderMtx
359 (
360  float* ls_dirs_deg,
361  int nLS,
362  LOUDSPEAKER_AMBI_DECODER_METHODS method,
363  int order,
364  int enableMaxReWeighting,
365  float* decMtx
366 )
367 {
368  int i, j, nSH;
369  float scale;
370  float* Y_ls, *a_n, *decMtx_maxrE;
371 
372  nSH = ORDER2NSH(order);
373  scale = 1.0f/SQRT4PI;
374 
375  switch(method){
376  default:
377  case LOUDSPEAKER_DECODER_DEFAULT:
378  case LOUDSPEAKER_DECODER_SAD:
379  /* Sampling Ambisonic Decoder (SAD) is simply the loudspeaker
380  * spherical harmonic matrix scaled by the number of loudspeakers.
381  */
382  Y_ls = malloc1d(nSH*nLS*sizeof(float));
383  getRSH(order, ls_dirs_deg, nLS, Y_ls);
384  utility_svsmul(Y_ls, &scale, nLS*nSH, Y_ls);
385  for(i=0; i<nLS; i++)
386  for(j=0; j<nSH; j++)
387  decMtx[i*nSH+j] = (4.0f*SAF_PI) * Y_ls[j*nLS + i]/(float)nLS;
388  free(Y_ls);
389  break;
390 
391  case LOUDSPEAKER_DECODER_MMD:
392  /* Mode-Matching Decoder (MMD) is simply the psuedo inverse of the
393  * loudspeaker spherical harmonic matrix. */
394  Y_ls = malloc1d(nSH*nLS*sizeof(float));
395  getRSH(order, ls_dirs_deg, nLS, Y_ls);
396  utility_svsmul(Y_ls, &scale, nLS*nSH, Y_ls);
397  utility_spinv(Y_ls, nSH, nLS, decMtx);
398  free(Y_ls);
399  break;
400 
401  case LOUDSPEAKER_DECODER_EPAD:
402  getEPAD(order, ls_dirs_deg, nLS, decMtx);
403  break;
404 
405  case LOUDSPEAKER_DECODER_ALLRAD:
406  getAllRAD(order, ls_dirs_deg, nLS, decMtx);
407  break;
408  }
409 
410  /* Apply maxRE weighting */
411  if(enableMaxReWeighting){
412  a_n = malloc1d(nSH*nSH*sizeof(float));
413  getMaxREweights(order, 1, a_n); /* 1: weights returned as diagonal matrix */
414  decMtx_maxrE = malloc1d(nLS * nSH * sizeof(float));
415  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nLS, nSH, nSH, 1.0f,
416  decMtx, nSH,
417  a_n, nSH, 0.0f,
418  decMtx_maxrE, nSH);
419  memcpy(decMtx, decMtx_maxrE, nLS*nSH*sizeof(float));
420 
421  free(a_n);
422  free(decMtx_maxrE);
423  }
424 }

426 void getBinauralAmbiDecoderMtx
427 (
428  float_complex* hrtfs,
429  float* hrtf_dirs_deg,
430  int N_dirs,
431  int N_bands,
432  BINAURAL_AMBI_DECODER_METHODS method,
433  int order,
434  float* freqVector,
435  float* itd_s,
436  float* weights,
437  int enableDiffCovMatching,
438  int enableMaxReWeighting,
439  float_complex* decMtx
440 )
441 {
442  int i, k, nSH;
443  float *tmp;
444  float_complex* a_n, *decMtx_rE;
445  const float_complex calpha = cmplxf(1.0f, 0.0f), cbeta = cmplxf(0.0f, 0.0f);
446 
447  nSH = ORDER2NSH(order);
448 
449  switch(method){
450  default:
451  case BINAURAL_DECODER_DEFAULT:
452  case BINAURAL_DECODER_LS: getBinDecoder_LS(hrtfs, hrtf_dirs_deg, N_dirs, N_bands, order, weights, decMtx); break;
453  case BINAURAL_DECODER_LSDIFFEQ: getBinDecoder_LSDIFFEQ(hrtfs, hrtf_dirs_deg, N_dirs, N_bands, order, weights, decMtx); break;
454  case BINAURAL_DECODER_SPR: getBinDecoder_SPR(hrtfs, hrtf_dirs_deg, N_dirs, N_bands, order, weights, decMtx); break;
455  case BINAURAL_DECODER_TA: getBinDecoder_TA(hrtfs, hrtf_dirs_deg, N_dirs, N_bands, order, freqVector, itd_s, weights, decMtx); break;
456  case BINAURAL_DECODER_MAGLS: getBinDecoder_MAGLS(hrtfs, hrtf_dirs_deg, N_dirs, N_bands, order, freqVector, weights, decMtx); break;
457  }
458 
459  /* apply Max RE weighting per bin */
460  if(enableMaxReWeighting){
461  tmp = malloc1d(nSH*nSH*sizeof(float));
462  a_n = malloc1d(nSH*nSH*sizeof(float_complex));
463  decMtx_rE = malloc1d(NUM_EARS*nSH*sizeof(float_complex));
464  getMaxREweights(order, 1, tmp);
465  for(i=0; i<nSH*nSH; i++)
466  a_n[i] = cmplxf(tmp[i], 0.0f);
467  for(k=0; k<N_bands; k++){
468  cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, NUM_EARS, nSH, nSH, &calpha,
469  &decMtx[k*NUM_EARS*nSH], nSH,
470  a_n, nSH, &cbeta,
471  decMtx_rE, nSH);
472  memcpy(&decMtx[k*NUM_EARS*nSH], decMtx_rE, NUM_EARS*nSH*sizeof(float_complex));
473  }
474  free(tmp);
475  free(a_n);
476  free(decMtx_rE);
477  }
478 
479  /* apply diffuse-field coherence matching per bin */
480  if(enableDiffCovMatching)
481  applyDiffCovMatching(hrtfs, hrtf_dirs_deg, N_dirs, N_bands, order, weights, decMtx);
482 }

484 void getBinauralAmbiDecoderFilters
485 (
486  float_complex* hrtfs,
487  float* hrtf_dirs_deg,
488  int N_dirs,
489  int fftSize,
490  float fs,
491  BINAURAL_AMBI_DECODER_METHODS method,
492  int order,
493  float* itd_s,
494  float* weights,
495  int enableDiffCovMatching,
496  int enableMaxReWeighting,
497  float* decFilters
498 )
499 {
500  int i, j, k, nBins, nSH;
501  float* freqVector;
502  float_complex* decMtx, *decMtx_bins;
503  void* hSafFFT;
504 
505  /* frequency-vector */
506  nBins = fftSize/2 + 1;
507  freqVector = malloc1d(nBins*sizeof(float));
508  getUniformFreqVector(fftSize, fs, freqVector);
509 
510  /* compute decoding matrix per bin */
511  nSH = ORDER2NSH(order);
512  decMtx = malloc1d(nBins*NUM_EARS*nSH*sizeof(float_complex));
513  getBinauralAmbiDecoderMtx(hrtfs, hrtf_dirs_deg, N_dirs, nBins, method,
514  order, freqVector, itd_s, weights, enableDiffCovMatching,
515  enableMaxReWeighting, decMtx);
516 
517  /* ifft, to obtain time-domain filters */
518  decMtx_bins = malloc1d(nBins*sizeof(float_complex));
519  saf_rfft_create(&hSafFFT, fftSize);
520  for(i=0; i<NUM_EARS; i++){
521  for(j=0; j<nSH; j++){
522  for(k=0; k<nBins; k++)
523  decMtx_bins[k] = decMtx[k*NUM_EARS*nSH + i*nSH + j];
524  saf_rfft_backward(hSafFFT, decMtx_bins, &decFilters[i*nSH*fftSize + j*fftSize]);
525  }
526  }
527 
528  saf_rfft_destroy(&hSafFFT);
529  free(freqVector);
530  free(decMtx);
531  free(decMtx_bins);
532 }

534 void applyDiffCovMatching
535 (
536  float_complex* hrtfs,
537  float* hrtf_dirs_deg,
538  int N_dirs,
539  int N_bands,
540  int order,
541  float* weights,
542  float_complex* decMtx
543 )
544 {
545  int i, nSH, band;
546  float* Y_tmp;
547  float_complex* W, *Y_na, *H_W, *H_ambi, *decMtx_diffMatched;
548  float_complex C_ref[NUM_EARS][NUM_EARS], C_ambi[NUM_EARS][NUM_EARS];
549  float_complex X[NUM_EARS][NUM_EARS], X_ambi[NUM_EARS][NUM_EARS];
550  float_complex XH_Xambi[NUM_EARS][NUM_EARS], U[NUM_EARS][NUM_EARS];
551  float_complex V[NUM_EARS][NUM_EARS], UX[NUM_EARS][NUM_EARS];
552  float_complex VUX[NUM_EARS][NUM_EARS], M[NUM_EARS][NUM_EARS];
553  const float_complex calpha = cmplxf(1.0f, 0.0f), cbeta = cmplxf(0.0f, 0.0f);
554 
555  nSH = ORDER2NSH(order);
556 
557  /* integration weights */
558  W = calloc1d(N_dirs*N_dirs, sizeof(float_complex));
559  if(weights!=NULL)
560  for(i=0; i<N_dirs; i++)
561  W[i*N_dirs+i] = cmplxf(weights[i], 0.0f);
562  else
563  for(i=0; i<N_dirs; i++)

589  C_ref[i][i] = cmplxf(crealf(C_ref[i][i]), 0.0f); /* force diagonal to be real */
590  utility_cchol((float_complex*)C_ref, NUM_EARS, (float_complex*)X);

613

616  (float_complex*)U, NUM_EARS,
617  (float_complex*)X, NUM_EARS, &cbeta,
618  (float_complex*)UX, NUM_EARS);

620  (float_complex*)V, NUM_EARS,
621  (float_complex*)UX, NUM_EARS, &cbeta,
622  (float_complex*)VUX, NUM_EARS);
623  utility_cglslv((float_complex*)X_ambi, NUM_EARS, (float_complex*)VUX, NUM_EARS, (float_complex*)M);
624  cblas_cgemm(CblasRowMajor, CblasConjTrans, CblasNoTrans, NUM_EARS, nSH, NUM_EARS, &calpha,
625  (float_complex*)M, NUM_EARS,
626  &decMtx[band*NUM_EARS*nSH], nSH, &cbeta,
627  decMtx_diffMatched, nSH);
628  memcpy(&decMtx[band*NUM_EARS*nSH], decMtx_diffMatched, NUM_EARS*nSH*sizeof(float_complex));
629  }
630

632  free(Y_na);

634  free(H_ambi);
