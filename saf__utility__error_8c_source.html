33 SAF_ERRORS saf_error_print(SAF_ERRORS err)
34 {
35  switch(err){
36  /* errors */
37  case SAF_ERROR__NO_ERROR: break;
38  case SAF_ERROR__ILLEGAL_INPUT_VALUE:
39  fprintf(stderr, "%s", "SAF Error: One or more input variable was assigned an illegal value.\n");
40  break;
41  case SAF_ERROR__UNALLOCATED_FUNCTION_ARGUMENT:
42  fprintf(stderr, "%s", "SAF Error: Memory for one or more input/output matrix/vector was not allocated.\n");
43  break;
44 
45  /* saf_vbap errors */
46  case SAF_ERROR__FAILED_TO_BUILD_CONVEX_HULL:
47  fprintf(stderr, "%s", "SAF Error: Failed to build Convex Hull.\n");
48  break;
49 
50  /* saf_hrir warnings */
51  case SAF_WARNING__SOFA_FILE_NOT_FOUND:
52  fprintf(stdout, "%s", "SAF Warning: Could not open SOFA file. Loading default HRIR data. \n");
53  break;
54 
55  /* saf_sh warnings */
56  case SAF_WARNING__UNABLE_TO_COMPUTE_BESSEL_FUNCTION_AT_SPECIFIED_ORDER:
57  fprintf(stdout, "%s", "SAF Warning: Could not compute spherical Bessel/Hankel at specified order. \n");
58  break;
59 
60  /* saf_utilites warnings */
61  case SAF_WARNING__FAILED_TO_COMPUTE_SVD:
62  fprintf(stdout, "%s", "SAF Warning: Could not compute SVD. Output matrices/vectors have been zeroed. \n");
63  break;
64  case SAF_WARNING__FAILED_TO_COMPUTE_EVG:
65  fprintf(stdout, "%s", "SAF Warning: Could not compute EVD. Output matrices/vectors have been zeroed. \n");
66  break;
67  case SAF_WARNING__FAILED_TO_SOLVE_LINEAR_EQUATION:
68  fprintf(stdout, "%s", "SAF Warning: Could not solve linear equation. Output matrix has been zeroed. \n");
69  break;
70  case SAF_WARNING__FAILED_TO_COMPUTE_CHOL:
71  fprintf(stdout, "%s", "SAF Warning: Could not compute Choleksy Factorisation. Output matrix has been zeroed. \n");
72  break;
73  }
74 
75  return SAF_ERROR__NO_ERROR;
76 }
