198 void tracker3d_predict
199 (
200  void* const hT3d,
201  int Tinc
202 )
203 {
204  tracker3d_data *pData = (tracker3d_data*)(hT3d);
205  int i, j, k, n, nDead, isDead, ind;
206  int* dead;
207  float dt0, dt1, p_death, rand01, distance_diff;
208  MCS_data* S;
209  tracker3d_config* tpars = &(pData->tpars);
210 #ifdef TRACKER_VERY_VERBOSE
211  char c_event[256], tmp[256];
212  printf("%s\n", "Prediction step");
213 #endif
214 
215  dead = NULL;
216 
217  /* Loop over particles */
218  for (i=0; i<tpars->Np; i++){
219  S = (MCS_data*)pData->SS[i];
220 
221  /* prep */
222  nDead = 0;
223  free(dead);
224  dead = NULL;
225 #ifdef TRACKER_VERY_VERBOSE
226  memset(c_event, 0, 256*sizeof(char));
227 #endif
228 
229  /* Loop over targets */
230  for (j=0; j<S->nTargets; j++){
231 
232  /* No target has died yet or multiple targets are allowed to die in
233  * one prediction step */
234  if (nDead==0 || tpars->ALLOW_MULTI_DEATH){
235  /* Probability of death */
236  dt0 = (float)S->Tcount[j] * S->dt;
237  dt1 = dt0 + S->dt * (float)Tinc;
238  if (dt0 == 0)
239  p_death = gamma_cdf(dt1, tpars->alpha_death, tpars->beta_death, 0.0f);
240  else
241  p_death = 1.0f - (1.0f-gamma_cdf(dt1, tpars->alpha_death,tpars->beta_death, 0.0f)) /
242  (1.0f-gamma_cdf(dt0,tpars->alpha_death,tpars->beta_death, 0.0f));
243 
244  /* Force probability of death to 1, if this target is too close
245  another target that has been alive longer. */
246  if (tpars->FORCE_KILL_TARGETS){
247  for(k=0; k<S->nTargets; k++){
248  if (k!=j){
249  distance_diff = (S->M[j].m0 - S->M[k].m0) * (S->M[j].m0 - S->M[k].m0) +
250  (S->M[j].m1 - S->M[k].m1) * (S->M[j].m1 - S->M[k].m1) +
251  (S->M[j].m2 - S->M[k].m2) * (S->M[j].m2 - S->M[k].m2);
252  distance_diff = sqrtf(distance_diff);
253  if (distance_diff < tpars->forceKillDistance && S->Tcount[j] <= S->Tcount[k])
254  p_death = 1.0f;
255  }
256  }
257  }
258 
259  /* Decide whether target should die */
260  rand_0_1(&rand01, 1);
261  if (rand01 < p_death){
262  nDead++; /* Target dies */
263  dead = realloc1d(dead, nDead*sizeof(float));
264  dead[nDead-1] = j;
265  }
266  }
267 
268  /* Prediction step if target is alive */
269  if( tpars->ALLOW_MULTI_DEATH ){
270  isDead = 0;
271  for(n=0; n<nDead; n++)
272  if(j==dead[n])
273  isDead = 1;
274 
275  /* Kalman Filter prediction for the target if alive */
276  if(!isDead)
277  kf_predict6(S->M[j].M, S->P[j].P, pData->A, pData->Q);
278  }
279  else {
280  /* Kalman Filter prediction for the target if alive */
281  if ( (nDead==0) || (j != dead[0]) )
282  kf_predict6(S->M[j].M, S->P[j].P, pData->A, pData->Q);
283  }
284  }
285 
286  /* Remove the dead target */
287  if (tpars->ALLOW_MULTI_DEATH){
288  for(j=0; j<nDead; j++){
289  /* Find index of the target to remove */
290  ind = -1;
291  for(k=0; k<S->nTargets; k++)
292  if(dead[j]==k)
293  ind = k;
294  assert(ind != -1);
295 
296  /* Shimy target data down by 1... overriding the dead target */
297  S->nTargets--;
298  if(ind!=S->nTargets){
299  memmove(&S->M[ind], &S->M[ind+1], (S->nTargets-ind)*sizeof(M6));
300  memmove(&S->P[ind], &S->P[ind+1], (S->nTargets-ind)*sizeof(P66));
301  memmove(&S->Tcount[ind], &S->Tcount[ind+1], (S->nTargets-ind)*sizeof(int));
302  memmove(&S->targetIDs[ind], &S->targetIDs[ind+1], (S->nTargets-ind)*sizeof(int));
303  }
304 
305  /* resize */
306  S->M = realloc1d(S->M, S->nTargets*sizeof(M6));
307  S->P = realloc1d(S->P, S->nTargets*sizeof(P66));
308  S->Tcount = realloc1d(S->Tcount, S->nTargets*sizeof(int));
309  S->targetIDs = realloc1d(S->targetIDs, S->nTargets*sizeof(int));
310 
311  /* Remove dead index for next iteration */
312  for(k=0; k<nDead; k++)
313  dead[k]--;
314 
315 #ifdef TRACKER_VERY_VERBOSE
316  sprintf(tmp, " Target %d died ", ind);
317  strcat(c_event, tmp);
318 #endif
319  }
320  }
321  else {
322  if (nDead==1){
323  /* Find index of the target to remove */
324  ind = -1;
325  for(k=0; k<S->nTargets; k++)
326  if(dead[0]==k)
327  ind = k;
328  assert(ind != -1);
329 
330  S->nTargets--;
331  if(ind!=S->nTargets){
332  memmove(&S->M[ind], &S->M[ind+1], (S->nTargets-ind)*sizeof(M6));
333  memmove(&S->P[ind], &S->P[ind+1], (S->nTargets-ind)*sizeof(P66));
334  memmove(&S->Tcount[ind], &S->Tcount[ind+1], (S->nTargets-ind)*sizeof(int));
335  memmove(&S->targetIDs[ind], &S->targetIDs[ind+1], (S->nTargets-ind)*sizeof(int));
336  }
337 
338  /* resize */
339  S->M = realloc1d(S->M, S->nTargets * sizeof(M6));
340  S->P = realloc1d(S->P, S->nTargets * sizeof(P66));
341  S->Tcount = realloc1d(S->Tcount, S->nTargets * sizeof(int));
342  S->targetIDs = realloc1d(S->targetIDs, S->nTargets * sizeof(int));
343 
344 #ifdef TRACKER_VERY_VERBOSE
345  sprintf(tmp, ", Target %d died ", ind);
346  strcpy(c_event, tmp);
347 #endif
348  }
349  }
350 
351  /* Print particle state */
352 #ifdef TRACKER_VERY_VERBOSE
353  sprintf(S->evstr, "MCS: %d, W: %.7f, IDs: [", i, S->W);
354  for (j=0; j<S->nTargets; j++){
355  sprintf(tmp, "%d ", S->targetIDs[j]);
356  strcat(S->evstr, tmp);
357  }
358  strcat(S->evstr, "] ");
359  strcat(S->evstr, c_event);
360  printf("%s\n", S->evstr);
361 #endif
362  }
363 }

365 void tracker3d_update
366 (
367  void* const hT3d,
368  float* Y,
369  int Tinc
370 )
371 {
372  tracker3d_data *pData = (tracker3d_data*)(hT3d);
373  int i, j, k, ss, n_events, count, cidx, unique, j_new, ev;
374  float TP0, LH, norm;
375  float M[6], P[6][6];
376  MCS_data* S, *S_event;
377  tracker3d_config* tpars = &(pData->tpars);
378 #ifdef TRACKER_VERY_VERBOSE
379  char tmp[256];
380  printf("%s\n", "Update step");
381 #endif
382 
383  /* Loop over particles */
384  for (i=0; i<tpars->Np; i++){
385  S = (MCS_data*)pData->SS[i];
386 
387  /* Association priors to targets */
388  TP0 = (1.0f-tpars->noiseLikelihood)/(S->nTargets+2.23e-10f);
389 
390  /* Number of possible events: */
391  n_events = S->nTargets + 1; /* clutter (+1) or 1 of the targets is active */
392  if( S->nTargets < tpars->maxNactiveTargets)
393  n_events++; /* Also a chance of a new target */
394  assert(n_events<=TRACKER3D_MAX_NUM_EVENTS);
395 
396  /* Prep */
397 #ifdef TRACKER_VERBOSE
398  memset(pData->evt, 0, n_events*256*sizeof(char)); /* Event descriptions */
399 #endif
400  count = 0; /* Event counter */
401  cidx = 0; /* current index */
402 
403  /* Association to clutter */
404  count++;
405 #ifdef TRACKER_VERBOSE
406  strcpy(pData->evt[cidx], "Clutter");
407 #endif
408  free(pData->evta[cidx]);
409  pData->evta[cidx] = NULL;
410  pData->evp[cidx] = (1.0f-tpars->init_birth)*tpars->noiseLikelihood;
411  pData->evl[cidx] = tpars->cd;
412  tracker3d_particleCopy(pData->SS[i], pData->str[cidx]);
413  cidx++;
414 
415  /* Loop over associations to targets */
416  for (j=0; j<S->nTargets; j++){
417  /* Compute update result and likelihood for association to signal j */
418  kf_update6(S->M[j].M, S->P[j].P, Y, pData->H, pData->R, M, P, &LH);
419  if(pData->tpars.ARE_UNIT_VECTORS)
420  cblas_sscal(3, 1.0f/L2_norm3(M), M, 1);
421 
422  /* Assocation to target j */
423  count++;
424 #ifdef TRACKER_VERBOSE
425  sprintf(pData->evt[cidx], "Target %d ", S->targetIDs[j]);
426 #endif
427  pData->evta[cidx] = realloc1d(pData->evta[cidx], sizeof(int));
428  pData->evta[cidx][0] = S->targetIDs[j];
429  pData->evp[cidx] = (1.0f-tpars->init_birth)*TP0;
430  pData->evl[cidx] = LH;
431  tracker3d_particleCopy(pData->SS[i], pData->str[cidx]);
432  S_event = (MCS_data*)pData->str[cidx];
433  memcpy(S_event->M[j].M, M, sizeof(M6));
434  memcpy(S_event->P[j].P, P, sizeof(P66));
435  for(k=0; k<S->nTargets; k++)
436  S_event->Tcount[k] += Tinc;
437  cidx++;
438  }
439 
440  /* Association to new target */
441  if (S->nTargets < tpars->maxNactiveTargets){
442  /* Initialization of new target */
443  kf_update6(tpars->M0, tpars->P0, Y, pData->H, pData->R, M, P, &LH);
444  if(pData->tpars.ARE_UNIT_VECTORS)
445  cblas_sscal(3, 1.0f/L2_norm3(M), M, 1);
446 
447  /* find an untaken ID */
448  j_new = 0;
449  for (ss = 0; ss<tpars->maxNactiveTargets; ss++){
450  unique = 1;
451  for(j=0; j<S->nTargets; j++){
452  if(ss == S->targetIDs[j]){
453  unique = 0;
454  }
455  };
456  if(unique){
457  j_new = ss;
458  break;
459  }
460  }
461 
462  count++;
463  j = S->nTargets;
464 #ifdef TRACKER_VERBOSE
465  sprintf(pData->evt[cidx], "New Target %d ", j);
466 #endif
467  pData->evta[cidx] = realloc1d(pData->evta[cidx], sizeof(int));
468  pData->evta[cidx][0] = j;
469  pData->evp[cidx] = tpars->init_birth;
470  pData->evl[cidx] = LH;
471  tracker3d_particleCopy(pData->SS[i], pData->str[cidx]);
472  S_event = (MCS_data*)pData->str[cidx];
473  S_event->nTargets = j+1;
474  S_event->M = realloc1d(S_event->M, S_event->nTargets*sizeof(M6));
475  S_event->P = realloc1d(S_event->P, S_event->nTargets*sizeof(P66));
476  memcpy(S_event->M[j].M, M, sizeof(M6));
477  memcpy(S_event->P[j].P, P, sizeof(P66));
478  S_event->Tcount = realloc1d(S_event->Tcount, S_event->nTargets*sizeof(int));
479  S_event->Tcount[j] = 0;
480  S_event->targetIDs = realloc1d(S_event->targetIDs, S_event->nTargets*sizeof(int));
481  S_event->targetIDs[j] = j_new;
482  cidx++;
483  }
484 
485  /* Draw sample from importance distribution */
486  norm = 1.0f/sumf(pData->evp, count);
487  cblas_sscal(count, norm, pData->evp, 1);
488  utility_svvmul(pData->evp, pData->evl, count, pData->imp);
489  norm = 1.0f/sumf(pData->imp, count);
490  cblas_sscal(count, norm, pData->imp, 1);
491  ev = categ_rnd(pData->imp, count); /* Event index */
492  assert(ev!=-1);
493 
494  /* Update particle */
495  tracker3d_particleCopy(pData->str[ev], pData->SS[i]);
496  S->W *= (pData->evl[ev] * pData->evp[ev]/ pData->imp[ev]);
497 
498  /* Print particle state */
499 #ifdef TRACKER_VERY_VERBOSE
500  sprintf(S->evstr, "MCS: %d, W: %.7f, IDs: [", i, S->W);
501  for (j=0; j<S->nTargets; j++){
502  sprintf(tmp, "%d ", S->targetIDs[j]);
503  strcat(S->evstr, tmp);
504  }
505  strcat(S->evstr, "] ");
506  strcat(S->evstr, pData->evt[ev]);
507  printf("%s\n", S->evstr);
508 #endif
509  }
510 
511  normalise_weights(pData->SS, tpars->Np);
512 }

541 void resampstr
542 (
543  voidPtr* SS,
544  int NP,
545  int* s
546 )
547 {
548  int i, j, k, a;
549  float c;
550  float pn[TRACKER3D_MAX_NUM_PARTICLES], r[TRACKER3D_MAX_NUM_PARTICLES];
551 
552  for (i=0; i<NP; i++)
553  pn[i] = ((MCS_data*)SS[i])->W*(float)NP;
554  memset(s, 0, NP*sizeof(int));
555  rand_0_1(r, NP);
556  k=0;
557  c=0.0f;
558  for (i=0; i<NP; i++){
559  c+=pn[i];
560  if (c>=1.0f) {
561  a = floorf(c);
562  c = c-a;
563  for(j=0; j<a; j++)
564  s[k+j]=i;
565  k=k+a;
566  }
567  if (k<NP && c>=r[k]){
568  c=c-1.0f;
569  s[k]=i;
570  k++;
571  }
572  }
573 }

645 /* hard-coded for length(X)=6 ... */
646 void kf_update6
647 (
648  float X[6],
649  float P[6][6],
650  float y[3],
651  float H[3][6],
652  float R[3][3],
653  float X_out[6],
654  float P_out[6][6],
655  float* LH
656 )
657 {
658  int i;
659  float yIM[3], IM[3], IS[3][3], HP[3][6], HPHT[3][3], PHT[6][3], K[6][3], K_yIM[6], KIS[6][3];
660 
661  /* update step */
662  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, 3, 1, 6, 1.0f,
663  (float*)H, 6,
664  (float*)X, 1, 0.0f,
665  (float*)IM, 1);
666  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, 3, 6, 6, 1.0f,
667  (float*)H, 6,
668  (float*)P, 6, 0.0f,
669  (float*)HP, 6);
670  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasTrans, 3, 3, 6, 1.0f,
671  (float*)HP, 6,
672  (float*)H, 6, 0.0f,
673  (float*)HPHT, 3);
674  utility_svvadd((float*)HPHT, (float*)R, 9, (float*)IS);
675  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasTrans, 6, 3, 6, 1.0f,
676  (float*)P, 6,
677  (float*)H, 6, 0.0f,
678  (float*)PHT, 3);
679  utility_sglslvt((float*)PHT, 6, (float*)IS, 3, (float*)K);
