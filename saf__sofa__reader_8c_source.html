568 void loadSofaFile
569 (
570  char* sofa_filepath,
571  float** hrirs,
572  float** hrir_dirs_deg,
573  int* N_hrir_dirs,
574  int* hrir_len,
575  int* hrir_fs
576 )
577 {
578  int i, j, k, retval,dimid[6], *dimids, ndimsp,ncid, varid, is0_360;
579  size_t dimlength[6], IR_dims[3], SourcePosition_dims[2];
580  size_t hrir_dims[3];
581  char dimname[6];
582  //const char* errorMessage;
583  double* IR, *SourcePosition, IR_fs;
584 
585  /* free any existing memory */
586  if(*hrirs!=NULL)
587  free(*hrirs);
588  if(*hrir_dirs_deg!=NULL)
589  free(*hrir_dirs_deg);
590 
591  /* open sofa file */
592  /* (retval is set to error value if sofa_filepath==NULL (intentional), or if the file
593  * path/name was not found (unintentional). */
594  ncid = -1;
595  if(sofa_filepath!=NULL){
596  if ((retval = nc_open(sofa_filepath, NC_NOWRITE, &ncid)))
597  nc_strerror(retval);
598  }
599  else
600  retval = NC_FATAL;
601 
602  /* if error: */
603  if(retval!=NC_NOERR){
604  is0_360 = 0;
605  /* return default HRIR data */
606  (*N_hrir_dirs) = __default_N_hrir_dirs;
607  (*hrir_len) = __default_hrir_len;
608  (*hrir_fs) = __default_hrir_fs;
609  (*hrirs) = malloc1d((*N_hrir_dirs) * 2 * (*hrir_len)*sizeof(float));
610  memcpy((*hrirs), (float*)__default_hrirs, (*N_hrir_dirs) * 2 * (*hrir_len)*sizeof(float));
611  (*hrir_dirs_deg) = malloc1d((*N_hrir_dirs) * 2 * sizeof(float));
612  memcpy((*hrir_dirs_deg), (float*)__default_hrir_dirs_deg, (*N_hrir_dirs) * 2 * sizeof(float));
613  for(i=0; i<(*N_hrir_dirs); i++){
614  if((*hrir_dirs_deg)[2*i+0]>=181.0f)
615  is0_360 = 1;
616  }
617  /* convert to -180..180, if azi is 0..360 */
618  if(is0_360)
619  for(i=0; i<(*N_hrir_dirs); i++)
620  (*hrir_dirs_deg)[2*i+0] = (*hrir_dirs_deg)[2*i+0]>180.0f ? (*hrir_dirs_deg)[2*i+0] -360.0f : (*hrir_dirs_deg)[2*i+0];
621 
622 #ifndef NDEBUG
623  /* also output warning message, if encountering this error value was
624  * unintentional (i.e. sofa_filepath!=NULL) */
625  if(sofa_filepath!=NULL)
626  saf_error_print(SAF_WARNING__SOFA_FILE_NOT_FOUND);
627 #endif
628  return;
629  }
630  assert(ncid!=-1);
631 
632  /* Determine dimension IDs and lengths */
633  /* Note: there are 6 possible dimension lengths in the sofa standard */
634  for (i=0; i<6; i++){
635  retval = nc_inq_dim(ncid, i, &dimname[i], &dimlength[i]);
636  nc_strerror(retval);
637  retval = nc_inq_dimid(ncid, &dimname[i], &dimid[i]);
638  nc_strerror(retval);
639  }
640 
641  /* Extract IR data */
642  if ((retval = nc_inq_varid(ncid, "Data.IR", &varid)))
643  nc_strerror(retval);
644  if ((retval = nc_inq_varndims (ncid, varid, &ndimsp)))
645  nc_strerror(retval);
646  dimids = malloc1d(ndimsp*sizeof(int));
647  if ((retval = nc_inq_vardimid(ncid, varid, dimids)))
648  nc_strerror(retval);
649  for(i=0; i<3; i++)
650  IR_dims[i] = dimlength[dimid[dimids[i]]];
651  free(dimids);
652  IR = malloc1d(IR_dims[0]*IR_dims[1]*IR_dims[2]*sizeof(double));
653  if ((retval = nc_get_var(ncid, varid, IR)))
654  nc_strerror(retval);
655  if ((retval = nc_inq_varid(ncid, "Data.SamplingRate", &varid)))
656  nc_strerror(retval);
657  if ((retval = nc_get_var(ncid, varid, &IR_fs)))
658  nc_strerror(retval);
659 
660  /* Extract positional data */
661  if ((retval = nc_inq_varid(ncid, "SourcePosition", &varid)))
662  nc_strerror(retval);
663  if ((retval = nc_inq_varndims (ncid, varid, &ndimsp)))
664  nc_strerror(retval);
665  dimids = malloc1d(ndimsp*sizeof(int));
666  if ((retval = nc_inq_vardimid(ncid, varid, dimids)))
667  nc_strerror(retval);
668  for(i=0; i<2; i++)
669  SourcePosition_dims[i] = dimlength[dimid[dimids[i]]];
670  free(dimids);
671  SourcePosition = malloc1d(SourcePosition_dims[0]*SourcePosition_dims[1]*sizeof(double));
672  if ((retval = nc_get_var(ncid, varid, SourcePosition)))
673  nc_strerror(retval);
674 
675  /* Close the file, freeing all resources. */
676  if ((retval = nc_close(ncid)))
677  nc_strerror(retval);
678 
679  /* Allocate memory */
680  (*hrir_len) = (int)IR_dims[2];
681  (*hrirs) = malloc1d(IR_dims[0]*IR_dims[1]*(*hrir_len) *sizeof(float));
682  (*hrir_dirs_deg) = malloc1d(SourcePosition_dims[0]*2*sizeof(float));
683 
684  /* Store relevent info in handle */
685  (*hrir_fs) = (int)(IR_fs+0.5);
686  hrir_dims[0] = IR_dims[0];
687  hrir_dims[1] = IR_dims[1];
688  hrir_dims[2] = (size_t)(*hrir_len);
689  (*N_hrir_dirs) = (int)hrir_dims[0];
690  for(i=0; i<(int)hrir_dims[0]; i++){
691  for(j=0; j<(int)hrir_dims[1]; j++){
692  for(k=0; k<(int)hrir_dims[2]; k++){
693  /* trunctate IRs and store in floating point precision */
694  (*hrirs)[i*hrir_dims[1]*hrir_dims[2] + j*hrir_dims[2] + k] = (float)IR[i*IR_dims[1]*IR_dims[2] + j*IR_dims[2] + k];
695  }
696  }
697  }
698 
699  /* store in floating point precision */
700  is0_360 = 0;
701  for(i=0; i<(int)SourcePosition_dims[0]; i++){
702  (*hrir_dirs_deg)[2*i+0] = (float)SourcePosition[i*SourcePosition_dims[1]+0];
703  (*hrir_dirs_deg)[2*i+1] = (float)SourcePosition[i*SourcePosition_dims[1]+1];
704  if((*hrir_dirs_deg)[2*i+0]>=181.0f)
705  is0_360 = 1;
706  }
707 
708  /* convert to -180..180, if azi is 0..360 */
709  if(is0_360)
710  for(i=0; i<(int)SourcePosition_dims[0]; i++)
711  (*hrir_dirs_deg)[2*i+0] = (*hrir_dirs_deg)[2*i+0]>180.0f ? (*hrir_dirs_deg)[2*i+0] -360.0f : (*hrir_dirs_deg)[2*i+0];
712 
713  free(IR);
714  free(SourcePosition);
715 }
