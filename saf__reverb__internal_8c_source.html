70 (
71  void* hEcho,
72  int numImageSources,
73  int nChannels
74 )
75 {
76  echogram_data *ec = (echogram_data*)(hEcho);
77  int i;
78 
79  if(ec->nChannels != nChannels || ec->numImageSources != numImageSources){
80  /* Resize echogram data */
81  ec->nChannels = nChannels;
82  ec->numImageSources = numImageSources;
83  ec->value = (float**)realloc2d((void**)ec->value, nChannels, numImageSources, sizeof(float));
84  ec->time = realloc1d(ec->time, numImageSources*sizeof(float));
85  ec->order = (int**)realloc2d((void**)ec->order, numImageSources, 3, sizeof(int));
86  ec->coords = realloc1d(ec->coords, numImageSources * sizeof(ims_pos_xyz));
87  ec->sortedIdx = realloc1d(ec->sortedIdx, numImageSources*sizeof(int));
88 
89  /* Resize the optional helper variables (used for run-time speed-ups) */
90  if(ec->include_rt_vars){
91  ec->tmp1 = realloc1d(ec->tmp1, numImageSources*sizeof(float));
92  ec->tmp2 = realloc1d(ec->tmp2, numImageSources*sizeof(float));
93  ec->rIdx = realloc1d(ec->rIdx, numImageSources*sizeof(int));
94  for(i=0; i<IMS_LAGRANGE_ORDER; i++)
95  ec->rIdx_frac[i] = realloc1d(ec->rIdx_frac[i], numImageSources*sizeof(int));
96  ec->h_frac = (float**)realloc2d((void**)ec->h_frac, IMS_LAGRANGE_ORDER+1, numImageSources, sizeof(float));
97  ec->cb_vals = (float**)realloc2d((void**)ec->cb_vals, nChannels, numImageSources, sizeof(float));
98  ec->contrib = (float**)realloc2d((void**)ec->contrib, nChannels, numImageSources, sizeof(float));
99  ec->ones_dummy = realloc1d(ec->ones_dummy, numImageSources*sizeof(float));
100  for(i=0; i<numImageSources; i++)
101  ec->ones_dummy[i] = 1.0f;
102  }
103  }

106 void ims_shoebox_echogramCopy
107 (
108  void* hEchoX,
109  void* hEchoY
110 )
111 {
112  echogram_data *ecX = (echogram_data*)(hEchoX);
113  echogram_data *ecY = (echogram_data*)(hEchoY);
114  int nChannels, numImageSources;
115 
116  assert(hEchoX!=NULL && hEchoY!=NULL);
117  if(hEchoX==hEchoY)
118  return; /* no copying required... */
119 
120  if(ecX->nChannels==0 || ecX->numImageSources==0)
121  return;
122 
123  /* Resize container 'Y' to be the same as container 'X' (if needed) */
124  nChannels = ecX->nChannels;
125  numImageSources = ecX->numImageSources;
126  if(ecY->nChannels != nChannels || ecY->numImageSources != numImageSources)
127  ims_shoebox_echogramResize(hEchoY, numImageSources, nChannels);
128 
129  /* Copy echogram data from X to Y */
130  cblas_scopy(nChannels*numImageSources, FLATTEN2D(ecX->value), 1, FLATTEN2D(ecY->value), 1);
131  cblas_scopy(numImageSources, ecX->time, 1, ecY->time, 1);
132  memcpy(FLATTEN2D(ecY->order), FLATTEN2D(ecX->order), numImageSources*3*sizeof(int));
133  memcpy(ecY->coords, ecX->coords, numImageSources*sizeof(ims_pos_xyz));
134  memcpy(ecY->sortedIdx, ecX->sortedIdx, numImageSources*sizeof(int));
135 }

268 void ims_shoebox_coreInitT
269 (
270  void* hWork,
271  float room[3],
272  ims_pos_xyz src,
273  ims_pos_xyz rec,
274  float maxTime_s,
275  float c_ms
276 )
277 {
278  ims_core_workspace *wrk = (ims_core_workspace*)(hWork);
279  echogram_data *echogram = (echogram_data*)(wrk->hEchogram);
280  ims_pos_xyz src_orig, rec_orig;
281  int imsrc, vIdx;
282  int ii, jj, kk;
283  float d_max;
284 
285  d_max = maxTime_s*c_ms;
286 
287  /* move origin to the centre of the room */
288  src_orig.x = src.x - room[0]/2.0f;
289  src_orig.y = room[1]/2.0f - src.y;
290  src_orig.z = src.z - room[2]/2.0f;
291  rec_orig.x = rec.x - room[0]/2.0f;
292  rec_orig.y = room[1]/2.0f - rec.y;
293  rec_orig.z = rec.z - room[2]/2.0f;
294 
295  /* Update indices only if the maximum permitted delay or room dimensions have changed */
296  if( (wrk->d_max != d_max) ||
297  (wrk->room[0] != room[0]) || (wrk->room[1] != room[1]) || (wrk->room[2] != room[2]) )
298  {
299  wrk->Nx = (int)(d_max/room[0] + 1.0f); /* ceil */
300  wrk->Ny = (int)(d_max/room[1] + 1.0f); /* ceil */
301  wrk->Nz = (int)(d_max/room[2] + 1.0f); /* ceil */
302  wrk->lengthVec = (2*(wrk->Nx)+1) * (2*(wrk->Ny)+1) * (2*(wrk->Nz)+1);
303 
304  /* i,j,k indices for calculation in x,y,z respectively */
305  wrk->II = realloc1d(wrk->II, wrk->lengthVec*sizeof(float));
306  wrk->JJ = realloc1d(wrk->JJ, wrk->lengthVec*sizeof(float));
307  wrk->KK = realloc1d(wrk->KK, wrk->lengthVec*sizeof(float));
308  ii = -(wrk->Nx); jj = -(wrk->Ny); kk = -(wrk->Nz);
309  for(imsrc = 0; imsrc<wrk->lengthVec; imsrc++){
310  wrk->II[imsrc] = (float)ii;
311  wrk->JJ[imsrc] = (float)jj;
312  wrk->KK[imsrc] = (float)kk;
313  ii++;
314  if(ii>wrk->Nx){
315  ii = -(wrk->Nx);
316  jj++;
317  }
318  if(jj>wrk->Ny){
319  jj = -(wrk->Ny);
320  kk++;
321  }
322  if(kk>wrk->Nz){
323  kk = -(wrk->Nz);
324  }
325  }
326 
327  /* Re-allocate memory */
328  wrk->validIDs = realloc1d(wrk->validIDs, wrk->lengthVec*sizeof(int));
329  wrk->s_x = realloc1d(wrk->s_x, wrk->lengthVec*sizeof(float));
330  wrk->s_y = realloc1d(wrk->s_y, wrk->lengthVec*sizeof(float));
331  wrk->s_z = realloc1d(wrk->s_z, wrk->lengthVec*sizeof(float));
332  wrk->s_d = realloc1d(wrk->s_d, wrk->lengthVec*sizeof(float));
333  wrk->s_t = realloc1d(wrk->s_t, wrk->lengthVec*sizeof(float));
334  wrk->s_att = realloc1d(wrk->s_att, wrk->lengthVec*sizeof(float));
335  }
336 
337  /* Update echogram only if the source/receiver positions or room dimensions have changed */
338  if( (wrk->d_max != d_max) ||
339  (wrk->rec.x != rec_orig.x) || (wrk->rec.y != rec_orig.y) || (wrk->rec.z != rec_orig.z) ||
340  (wrk->src.x != src_orig.x) || (wrk->src.y != src_orig.y) || (wrk->src.z != src_orig.z) ||
341  (wrk->room[0] != room[0]) || (wrk->room[1] != room[1]) || (wrk->room[2] != room[2]))
342  {
343  wrk->d_max = d_max;
344  wrk->room[0] = room[0];
345  wrk->room[1] = room[1];
346  wrk->room[2] = room[2];
347  memcpy(&(wrk->rec), &rec_orig, sizeof(ims_pos_xyz));
348  memcpy(&(wrk->src), &src_orig, sizeof(ims_pos_xyz));
349 
350  /* image source coordinates with respect to receiver, and distance */
351  for(imsrc = 0; imsrc<wrk->lengthVec; imsrc++){
352  wrk->s_x[imsrc] = wrk->II[imsrc]*room[0] + powf(-1.0f, wrk->II[imsrc])*src_orig.x - rec_orig.x;
353  wrk->s_y[imsrc] = wrk->JJ[imsrc]*room[1] + powf(-1.0f, wrk->JJ[imsrc])*src_orig.y - rec_orig.y;
354  wrk->s_z[imsrc] = wrk->KK[imsrc]*room[2] + powf(-1.0f, wrk->KK[imsrc])*src_orig.z - rec_orig.z;
355  wrk->s_d[imsrc] = sqrtf(powf(wrk->s_x[imsrc], 2.0f) + powf(wrk->s_y[imsrc], 2.0f) + powf(wrk->s_z[imsrc], 2.0f));
356  }
357 
358  /* Determine the indices where the distance is below the specified maximum */
359  for(imsrc = 0, wrk->numImageSources = 0; imsrc<wrk->lengthVec; imsrc++){
360  if(wrk->s_d[imsrc]<d_max){
361  wrk->validIDs[imsrc] = 1;
362  wrk->numImageSources++; /* (within maximum distance) */
363  }
364  else
365  wrk->validIDs[imsrc] = 0;
366  }
367 
368  /* Resize echogram container (only done if needed) */
369  ims_shoebox_echogramResize(wrk->hEchogram, wrk->numImageSources, 1/*omni-pressure*/);
370 
371  /* Copy data into echogram struct */
372  for(imsrc = 0, vIdx = 0; imsrc<wrk->lengthVec; imsrc++){
373  if(wrk->validIDs[imsrc]){
374  echogram->time[vIdx] = wrk->s_d[imsrc]/c_ms;
375 
376  /* reflection propagation attenuation - if distance is <1m set
377  * attenuation to 1 to avoid amplification */
378  echogram->value[0][vIdx] = wrk->s_d[imsrc]<=1 ? 1.0f : 1.0f / wrk->s_d[imsrc];
379 
380  /* Order */
381  echogram->order[vIdx][0] = (int)(wrk->II[imsrc] + (wrk->II[imsrc]>0 ? 0.1f : -0.1f)); /* round */
382  echogram->order[vIdx][1] = (int)(wrk->JJ[imsrc] + (wrk->JJ[imsrc]>0 ? 0.1f : -0.1f));
383  echogram->order[vIdx][2] = (int)(wrk->KK[imsrc] + (wrk->KK[imsrc]>0 ? 0.1f : -0.1f));
384 
385  /* Coordinates */
386  echogram->coords[vIdx].x = wrk->s_x[imsrc];
387  echogram->coords[vIdx].y = wrk->s_y[imsrc];
388  echogram->coords[vIdx].z = wrk->s_z[imsrc];
389  vIdx++;
390  }
391  }
392 
393  /* Find indices to sort reflections according to propagation time (accending order) */
394  sortf(echogram->time, NULL, echogram->sortedIdx, echogram->numImageSources, 0);
395  }
396 }

398 void ims_shoebox_coreInitN
399 (
400  void* hWork,
401  float room[3],
402  ims_pos_xyz src,
403  ims_pos_xyz rec,
404  int maxN,
405  float c_ms
406 )
407 {
408  ims_core_workspace *wrk = (ims_core_workspace*)(hWork);
409  echogram_data *echogram = (echogram_data*)(wrk->hEchogram);
410  ims_pos_xyz src_orig, rec_orig;
411  int imsrc;
412  int ii, jj, kk;
413 
414  /* move origin to the centre of the room */
415  src_orig.x = src.x - room[0]/2.0f;
416  src_orig.y = room[1]/2.0f - src.y;
417  src_orig.z = src.z - room[2]/2.0f;
418  rec_orig.x = rec.x - room[0]/2.0f;
419  rec_orig.y = room[1]/2.0f - rec.y;
420  rec_orig.z = rec.z - room[2]/2.0f;
421 
422  /* Update indices only if the maximum reflection order has changed */
423  if( (wrk->N_max != maxN) )
424  {
425  wrk->lengthVec = (2*(maxN)+1) * (2*(maxN)+1) * (2*(maxN)+1);
426 
427  /* i,j,k indices for calculation in x,y,z respectively */
428  wrk->iII = realloc1d(wrk->iII, wrk->lengthVec*sizeof(int));
429  wrk->iJJ = realloc1d(wrk->iJJ, wrk->lengthVec*sizeof(int));
430  wrk->iKK = realloc1d(wrk->iKK, wrk->lengthVec*sizeof(int));
431  wrk->s_ord = realloc1d(wrk->s_ord, wrk->lengthVec*sizeof(int));
432  ii = -maxN; jj = -maxN; kk = -maxN;
433  for(imsrc = 0; imsrc<wrk->lengthVec; imsrc++){
434  wrk->iII[imsrc] = ii;
435  wrk->iJJ[imsrc] = jj;
436  wrk->iKK[imsrc] = kk;
437  wrk->s_ord[imsrc] = abs(wrk->iII[imsrc]) + abs(wrk->iJJ[imsrc]) + abs(wrk->iKK[imsrc]);
438  ii++;
439  if(ii>maxN){
440  ii = -(maxN);
441  jj++;
442  }
443  if(jj>maxN){
444  jj = -(maxN);
445  kk++;
446  }
447  if(kk>maxN){
448  kk = -(maxN);
449  }
450  }
451 
452  /* Cull the indices where the order is below the specified maximum */
453  wrk->II = realloc1d(wrk->II, wrk->lengthVec*sizeof(float));
454  wrk->JJ = realloc1d(wrk->JJ, wrk->lengthVec*sizeof(float));
455  wrk->KK = realloc1d(wrk->KK, wrk->lengthVec*sizeof(float));
456  for(imsrc = 0, wrk->numImageSources = 0; imsrc<wrk->lengthVec; imsrc++){
457  if(wrk->s_ord[imsrc]<=maxN){
458  wrk->II[wrk->numImageSources] = (float)wrk->iII[imsrc];
459  wrk->JJ[wrk->numImageSources] = (float)wrk->iJJ[imsrc];
460  wrk->KK[wrk->numImageSources] = (float)wrk->iKK[imsrc];
461  wrk->numImageSources++; /* (within maximum order) */
462  }
463  }
464 
465  /* Re-allocate memory */
466  wrk->s_x = realloc1d(wrk->s_x, wrk->numImageSources*sizeof(float));
467  wrk->s_y = realloc1d(wrk->s_y, wrk->numImageSources*sizeof(float));
468  wrk->s_z = realloc1d(wrk->s_z, wrk->numImageSources*sizeof(float));
469  wrk->s_d = realloc1d(wrk->s_d, wrk->numImageSources*sizeof(float));
470  wrk->s_t = realloc1d(wrk->s_t, wrk->numImageSources*sizeof(float));
471  wrk->s_att = realloc1d(wrk->s_att, wrk->numImageSources*sizeof(float));
472  }
473 
474  /* Update echogram only if the maximum reflection order, source/receiver positions or room dimensions have changed */
475  if( (wrk->N_max != maxN) ||
476  (wrk->rec.x != rec_orig.x) || (wrk->rec.y != rec_orig.y) || (wrk->rec.z != rec_orig.z) ||
477  (wrk->src.x != src_orig.x) || (wrk->src.y != src_orig.y) || (wrk->src.z != src_orig.z) ||
478  (wrk->room[0] != room[0]) || (wrk->room[1] != room[1]) || (wrk->room[2] != room[2]))
479  {
480  wrk->N_max = maxN;
481  wrk->room[0] = room[0];
482  wrk->room[1] = room[1];
483  wrk->room[2] = room[2];
484  memcpy(&(wrk->rec), &rec_orig, sizeof(ims_pos_xyz));
485  memcpy(&(wrk->src), &src_orig, sizeof(ims_pos_xyz));
486 
487  /* image source coordinates with respect to receiver, and distance */
488  for(imsrc = 0; imsrc<wrk->numImageSources; imsrc++){
489  wrk->s_x[imsrc] = wrk->II[imsrc]*room[0] + powf(-1.0f, wrk->II[imsrc])*src_orig.x - rec_orig.x;
490  wrk->s_y[imsrc] = wrk->JJ[imsrc]*room[1] + powf(-1.0f, wrk->JJ[imsrc])*src_orig.y - rec_orig.y;
491  wrk->s_z[imsrc] = wrk->KK[imsrc]*room[2] + powf(-1.0f, wrk->KK[imsrc])*src_orig.z - rec_orig.z;
492  wrk->s_d[imsrc] = sqrtf(powf(wrk->s_x[imsrc], 2.0f) + powf(wrk->s_y[imsrc], 2.0f) + powf(wrk->s_z[imsrc], 2.0f));
493  }
494 
495  /* Resize echogram container (only done if needed) */
496  ims_shoebox_echogramResize(wrk->hEchogram, wrk->numImageSources, 1/*omni-pressure*/);
497 
498  /* Copy data into echogram struct */
499  for(imsrc = 0; imsrc<wrk->numImageSources; imsrc++){
500  echogram->time[imsrc] = wrk->s_d[imsrc]/c_ms;
501 
502  /* reflection propagation attenuation - if distance is <1m set
503  * attenuation to 1 to avoid amplification */
504  echogram->value[0][imsrc] = wrk->s_d[imsrc]<=1 ? 1.0f : 1.0f / wrk->s_d[imsrc];
505 
506  /* Order */
507  echogram->order[imsrc][0] = (int)(wrk->II[imsrc] + (wrk->II[imsrc]>0 ? 0.1f : -0.1f)); /* round */
508  echogram->order[imsrc][1] = (int)(wrk->JJ[imsrc] + (wrk->JJ[imsrc]>0 ? 0.1f : -0.1f));
509  echogram->order[imsrc][2] = (int)(wrk->KK[imsrc] + (wrk->KK[imsrc]>0 ? 0.1f : -0.1f));
510 
511  /* Coordinates */
512  echogram->coords[imsrc].x = wrk->s_x[imsrc];
513  echogram->coords[imsrc].y = wrk->s_y[imsrc];
514  echogram->coords[imsrc].z = wrk->s_z[imsrc];
515  }
516 
517  /* Find indices to sort reflections according to propagation time (accending order) */
518  sortf(echogram->time, NULL, echogram->sortedIdx, echogram->numImageSources, 0);

522 void ims_shoebox_coreRecModuleSH
523 (
524  void* hWork,
525  int sh_order
526 )
527 {
528  ims_core_workspace *wrk = (ims_core_workspace*)(hWork);
529  echogram_data *echogram = (echogram_data*)(wrk->hEchogram);
530  echogram_data *echogram_rec = (echogram_data*)(wrk->hEchogram_rec);
531  int i, j, nSH;
532  float aziElev_rad[2];
533  float* sh_gains;
534 
535  nSH = ORDER2NSH(sh_order);
536 
537  /* Resize container (only done if needed) */
538  ims_shoebox_echogramResize(wrk->hEchogram_rec, echogram->numImageSources, nSH);
539 
540  /* Copy 'time', 'coord', 'order', except in ascending order w.r.t propagation time */
541  for(i=0; i<echogram_rec->numImageSources; i++){
542  echogram_rec->time[i] = echogram->time[echogram->sortedIdx[i]];
543  echogram_rec->order[i][0] = echogram->order[echogram->sortedIdx[i]][0];
544  echogram_rec->order[i][1] = echogram->order[echogram->sortedIdx[i]][1];
545  echogram_rec->order[i][2] = echogram->order[echogram->sortedIdx[i]][2];
546  echogram_rec->coords[i].v[0] = echogram->coords[echogram->sortedIdx[i]].v[0];
547  echogram_rec->coords[i].v[1] = echogram->coords[echogram->sortedIdx[i]].v[1];
548  echogram_rec->coords[i].v[2] = echogram->coords[echogram->sortedIdx[i]].v[2];
549 
550  echogram_rec->sortedIdx[i] = i; /* Should already sorted in ims_shoebox_coreInit() */
551  }
552 
553  /* Copy 'value' (the core omni-pressure), except also in ascending order w.r.t propagation time */
554  if(sh_order==0){
555  for(i=0; i<echogram_rec->numImageSources; i++)
556  echogram_rec->value[0][i] = echogram->value[0][echogram->sortedIdx[i]];
557  }
558  /* Impose spherical harmonic directivities onto 'value', and store in ascending order w.r.t propagation time */
559  else{
560  sh_gains = malloc1d(nSH*sizeof(float));
561  for(i=0; i<echogram_rec->numImageSources; i++){
562  /* Cartesian coordinates to spherical coordinates */
563  unitCart2sph(echogram_rec->coords[i].v, 1, 0, (float*)aziElev_rad);
564  aziElev_rad[1] = SAF_PI/2.0f-aziElev_rad[1]; /* AziElev to AziInclination conversion */
565 
566  /* Apply spherical harmonic weights */
567  getSHreal_recur(sh_order, (float*)aziElev_rad, 1, sh_gains);
568  for(j=0; j<nSH; j++)
569  echogram_rec->value[j][i] = sh_gains[j] * (echogram->value[0][echogram->sortedIdx[i]]);
570  }
571  free(sh_gains);
572  }

575 void ims_shoebox_coreAbsorptionModule
576 (
577  void* hWork,
578  float** abs_wall
579 )
580 {
581  ims_core_workspace *wrk = (ims_core_workspace*)(hWork);
582  echogram_data *echogram_abs;
583  int i,band,ch;
584  float r_x[2], r_y[2], r_z[2];
585  float abs_x, abs_y, abs_z, s_abs_tot;
586 
587  for(band=0; band < wrk->nBands; band++){
588  echogram_abs = (echogram_data*)wrk->hEchogram_abs[band];
589 
590  /* Copy "receiver" echogram data into "absorption" echogram container */
591  ims_shoebox_echogramCopy(wrk->hEchogram_rec, wrk->hEchogram_abs[band]);
592 
593  /* Then apply the wall absorption onto it, for this band... */
594 
595  /* Reflection coefficients given the absorption coefficients for x, y, z
596  * walls per frequency */
597  r_x[0] = sqrtf(1.0f - abs_wall[band][0]);
598  r_x[1] = sqrtf(1.0f - abs_wall[band][1]);
599  r_y[0] = sqrtf(1.0f - abs_wall[band][2]);
600  r_y[1] = sqrtf(1.0f - abs_wall[band][3]);
601  r_z[0] = sqrtf(1.0f - abs_wall[band][4]);
602  r_z[1] = sqrtf(1.0f - abs_wall[band][5]);
603 
604  /* find total absorption coefficients by calculating the number of hits on
605  * every surface, based on the order per dimension */
606  for(i=0; i<echogram_abs->numImageSources; i++){
607  /* Surfaces intersecting the x-axis */
608  if((echogram_abs->order[i][0]%2)==0) /* ISEVEN */
609  abs_x = powf(r_x[0], (float)abs(echogram_abs->order[i][0])/2.0f) * powf(r_x[1], (float)abs(echogram_abs->order[i][0])/2.0f);
610  else if (echogram_abs->order[i][0]>0) /* ISODD AND POSITIVE */
611  abs_x = powf(r_x[0], ceilf((float)echogram_abs->order[i][0]/2.0f)) * powf(r_x[1], floorf((float)echogram_abs->order[i][0]/2.0f));
612  else /* ISODD AND NEGATIVE */
613  abs_x = powf(r_x[0], floorf((float)abs(echogram_abs->order[i][0])/2.0f)) * powf(r_x[1], ceilf((float)abs(echogram_abs->order[i][0])/2.0f));
614 
615  /* Surfaces intersecting the y-axis */
616  if((echogram_abs->order[i][1]%2)==0) /* ISEVEN */
617  abs_y = powf(r_y[0], (float)abs(echogram_abs->order[i][1])/2.0f) * powf(r_y[1], (float)abs(echogram_abs->order[i][1])/2.0f);
618  else if (echogram_abs->order[i][1]>0) /* ISODD AND POSITIVE */
619  abs_y = powf(r_y[0], ceilf((float)echogram_abs->order[i][1]/2.0f)) * powf(r_y[1], floorf((float)echogram_abs->order[i][1]/2.0f));
620  else /* ISODD AND NEGATIVE */
621  abs_y = powf(r_y[0], floorf((float)abs(echogram_abs->order[i][1])/2.0f)) * powf(r_y[1], ceilf((float)abs(echogram_abs->order[i][1])/2.0f));
622 
623  /* Surfaces intersecting the y-axis */
624  if((echogram_abs->order[i][2]%2)==0) /* ISEVEN */
625  abs_z = powf(r_z[0], (float)abs(echogram_abs->order[i][2])/2.0f) * powf(r_z[1], (float)abs(echogram_abs->order[i][2])/2.0f);
626  else if (echogram_abs->order[i][2]>0) /* ISODD AND POSITIVE */
627  abs_z = powf(r_z[0], ceilf((float)echogram_abs->order[i][2]/2.0f)) * powf(r_z[1], floorf((float)echogram_abs->order[i][2]/2.0f));
628  else /* ISODD AND NEGATIVE */
629  abs_z = powf(r_z[0], floorf((float)abs(echogram_abs->order[i][2])/2.0f)) * powf(r_z[1], ceilf((float)abs(echogram_abs->order[i][2])/2.0f));
630 
631  /* Apply total absorption */
632  s_abs_tot = abs_x * abs_y * abs_z;
633  for(ch=0; ch<echogram_abs->nChannels; ch++)
634  echogram_abs->value[ch][i] *= s_abs_tot;
635  }
636  }
637 }

639 void ims_shoebox_renderRIR
640 (
641  void* hWork,
642  int fractionalDelayFLAG,
643  float fs,
644  float** H_filt,
645  ims_rir* rir
646 )
647 {
648  ims_core_workspace *wrk = (ims_core_workspace*)(hWork);
649  echogram_data *echogram_abs;
650  float* temp;
651  int i, j, refl_idx, band, rir_len_samples;
652  float endtime, rir_len_seconds;
653 
654  /* Render RIR for each octave band */
655  for(band=0; band<wrk->nBands; band++){
656  echogram_abs = (echogram_data*)wrk->hEchogram_abs[band];
657 
658  /* Determine length of rir */
659  endtime = echogram_abs->time[echogram_abs->numImageSources-1];
660  rir_len_samples = (int)(endtime * fs + 1.0f) + 1; /* ceil + 1 */
661  rir_len_seconds = (float)rir_len_samples/fs;
662 
663  /* Render rir */
664  if(fractionalDelayFLAG){
665  // TODO: implement
666  assert(0);
667  }
668  else{
669  /* Resize RIR vector */
670  wrk->rir_bands[band] = (float**)realloc2d((void**)wrk->rir_bands[band], echogram_abs->nChannels, rir_len_samples, sizeof(float));
671  wrk->rir_len_samples = rir_len_samples;
672  wrk->rir_len_seconds = rir_len_seconds;
673  memset(FLATTEN2D(wrk->rir_bands[band]), 0, (echogram_abs->nChannels)*rir_len_samples*sizeof(float)); /* flush */
674 
675  /* Accumulate 'values' for each image source */
676  for(i=0; i<echogram_abs->numImageSources; i++){
677  refl_idx = (int)(echogram_abs->time[i]*fs+0.5f); /* round */
678  for(j=0; j<echogram_abs->nChannels; j++)
679  wrk->rir_bands[band][j][refl_idx] += echogram_abs->value[j][i];
680  }
681  }
682  }
683 
684  temp = malloc1d((wrk->rir_len_samples+IMS_FIR_FILTERBANK_ORDER)*sizeof(float));
685 
686  /* Resize rir->data if needed, then flush with 0s */
687  echogram_abs = (echogram_data*)wrk->hEchogram_abs[0];
688  if( (echogram_abs->nChannels!=rir->nChannels) || (wrk->rir_len_samples !=rir->length) ){
689  rir->data = realloc1d(rir->data, echogram_abs->nChannels * (wrk->rir_len_samples) * sizeof(float));
690  rir->length = wrk->rir_len_samples;
691  rir->nChannels = echogram_abs->nChannels;
692  }
693  memset(rir->data, 0, echogram_abs->nChannels * (wrk->rir_len_samples) * sizeof(float));
694 
695  /* Apply filterbank to rir_bands and sum them up */
696  for(band=0; band<wrk->nBands; band++){
697  echogram_abs = (echogram_data*)wrk->hEchogram_abs[band];
698 
699  /* Apply the LPF (lowest band), HPF (highest band), and BPF (all other bands) */
700  for(j=0; j<echogram_abs->nChannels; j++)
701  fftconv(wrk->rir_bands[band][j], H_filt[band], wrk->rir_len_samples, IMS_FIR_FILTERBANK_ORDER+1, 1, temp);
702 
703  /* Sum */
704  for(i=0; i<echogram_abs->nChannels; i++)
705  utility_svvadd( &(rir->data[i*(wrk->rir_len_samples)]), wrk->rir_bands[band][i], wrk->rir_len_samples, &(rir->data[i*(wrk->rir_len_samples)]));
706  }
707 
708  free(temp);
709 }
